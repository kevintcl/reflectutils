17  import java.text.DateFormat;
18  import java.text.SimpleDateFormat;
19  import java.util.ArrayList;

22  /***
23   * Simple class to add in support for controlling the formats to
24   * use when parsing the various dates
25   * 
26   * @author Aaron Zeckoski (azeckoski @ gmail.com)
27   */
28  public class BaseDateFormatHolder {
29  
30      public BaseDateFormatHolder() {}
31  
32      /***
33       * @param patterns set the patterns to use when parsing strings to dates
34       */
35      public BaseDateFormatHolder(String[] patterns) {
36          setPatterns(patterns);
37      }
38  
39      /***
40       * @param formats the formats to use when parsing strings into dates
41       */
42      public BaseDateFormatHolder(DateFormat[] formats) {
43          setFormats(formats);
44      }
45  
46      protected DateFormat[] formats;
47      /***
48       * @param formats sets the format objects to use for parsing
49       */
50      public void setFormats(DateFormat[] formats) {
51          this.formats = formats;
52      }
53      protected String[] patterns;
54      /***
55       * @param patterns set the patterns to use for parsing
56       */
57      public void setPatterns(String[] patterns) {
58          this.patterns = patterns;
59          getDateFormats();
60      }
61  //    = new String[] {
62  //            "yyyy-MM-dd HH:mm:ss.fffffffff",
63  //            "yyyy-MM-dd",
64  //            "HH:mm:ss"
65  //    };
66      /***
67       * @return the array of date formats currently used by this converter
68       */
69      public DateFormat[] getDateFormats() {
70          if (formats == null) {
71              ArrayList<DateFormat> dateFormats = new ArrayList<DateFormat>();
72              // add the standard short ones
73  //            dateFormats.add( DateFormat.getDateInstance(DateFormat.SHORT) );
74  //            dateFormats.add( DateFormat.getTimeInstance(DateFormat.SHORT) );
75  //            dateFormats.add( DateFormat.getDateTimeInstance(DateFormat.SHORT, DateFormat.SHORT) );
76              if (patterns != null) {
77                  for (int i = 0; i < patterns.length; i++) {
78                      dateFormats.add( new SimpleDateFormat(patterns[i]) );
79                  }
80              }
81              formats = dateFormats.toArray(new DateFormat[dateFormats.size()]);
82          }
83          return formats;
84      }
