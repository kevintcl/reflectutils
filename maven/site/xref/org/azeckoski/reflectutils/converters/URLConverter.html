15  package org.azeckoski.reflectutils.converters;
16  
17  import java.net.MalformedURLException;
18  import java.net.URL;
19  
20  import org.azeckoski.reflectutils.converters.api.Converter;
21  
22  
23  
24  /***
25   * Converts strings into URLs
26   * 
27   * @author Aaron Zeckoski (azeckoski @ gmail.com)
28   */
29  public class URLConverter implements Converter<URL> {
30  
31      public URL convert(Object value) {
32          URL url = null;
33          String s = value.toString();
34          if (s != null && s.length() > 0) {
35              try {
36                  url = new URL(s);
37              } catch (MalformedURLException e) {
38                  throw new UnsupportedOperationException("URL convert failure: cannot convert source ("+value+") and type ("+value.getClass()+") to a URL: " + e.getMessage());
39              }
40          }
41          if (url == null) {
42              throw new UnsupportedOperationException("URL convert failure: cannot convert source ("+value+") and type ("+value.getClass()+") to a URL");
43          }
44          return url;
45      }
46  
47  }
