58      /***
59       * Convert an object to one of the following date types<br/>
60       * <ul>
61       *     <li><code>java.util.Date</code></li>
62       *     <li><code>java.util.Calendar</code></li>
63       *     <li><code>java.sql.Date</code></li>
64       *     <li><code>java.sql.Time</code></li>
65       *     <li><code>java.sql.Timestamp</code></li>
66       * </ul>
67       * 
68       * @param <T>
69       * @param targetType the type to convert to (should be one of the given date types)
70       * @param value any object
71       * @return the converted value
72       */
73      public static <T> T convertToType(Class<T> targetType, Object value) {
74          return convertToType(targetType, value, null);
75      }

77      /***
78       * @param <T>
79       * @param targetType the type to convert to (should be one of the given date types)
80       * @param value any object
81       * @param dateFormats include date formats to use during the string -> date conversion
82       * @return the converted value
83       */
84      @SuppressWarnings("unchecked")
85      public static <T> T convertToType(Class<T> targetType, Object value, DateFormat[] dateFormats) {
86          if (value == null) {
87              throw new IllegalArgumentException("value to convert cannot be null");
88          }
89  
90          // Handle java.sql.Timestamp
91          if (value instanceof java.sql.Timestamp) {
92              long timeInMillis = ((Timestamp)value).getTime();
93              // ---------------------- JDK 1.3 Fix ----------------------
94              // N.B. Prior to JDK 1.4 the Timestamp's getTime() method
95              //      didn't include the milliseconds. The following code
96              //      ensures it works consistently accross JDK versions
97  //            java.sql.Timestamp timestamp = (java.sql.Timestamp)value;
98  //            long timeInMillis = ((timestamp.getTime() / 1000) * 1000);
99  //            timeInMillis += timestamp.getNanos() / 1000000;
100             // ---------------------- JDK 1.3 Fix ----------------------
101             return (T) toDate(targetType, timeInMillis);
102         }
103 
104         // Handle Date (includes java.sql.Date & java.sql.Time)
105         if ( Date.class.isAssignableFrom(value.getClass()) ) {
106             long time = ((Date)value).getTime();
107             return (T) toDate(targetType, time);
108         }
109 
110         // Handle Calendar
111         if (value instanceof Calendar) {
112             long time = ((Calendar)value).getTimeInMillis();
113             return (T) toDate(targetType, time);
114         }
115 
116         // Handle numbers
117         if (value instanceof Number) {
118             long num = ((Number)value).longValue();
119             return (T) toDate(targetType, num);
120         }
121 
122         // Convert all other types to String & handle
123         String stringValue = value.toString().trim();
124         if (stringValue.length() == 0) {
125             Object defaultValue = ConstructorUtils.getDefaultValue(targetType);
126             if (defaultValue == null) {
127                 throw new UnsupportedOperationException("Date convert failure: Could not convert empty string to target ("+targetType+") or get a default value for it");
128             }
129             return (T) defaultValue;
130         } else {
131             // try to get the long value out of the string
132             try {
133                 long num = new Long(stringValue);
134                 if (num > 30000000l) {
135                     // must be a UTC time code, also, we only lost a week since 1970 so whatever
136                     return (T) toDate(targetType, num);
137                 } else if (num > 10000l) {
138                     // probably is a date like: YYYYMMDD so convert it
139                     int date = (int) (num % 100);
140                     int remain = (int) (num / 100);
141                     int month = remain % 100;
142                     int year = remain / 100;
143                     if (date <= 31 && month <= 12) {
144                         if (month > 0) { month -= 1; }
145                         Calendar calendar = Calendar.getInstance();
146                         calendar.clear();
147                         calendar.set(year, month, date);
148                         long time = calendar.getTimeInMillis();
149                         return (T) toDate(targetType, time);
150                     }
151                 }
152             } catch (NumberFormatException e) {
153                 // ok, not a long probably so try parsing
154             }
155         }
156 
157         // Parse the Date/Time
158         Calendar calendar = parse(stringValue, dateFormats);
159         if (calendar != null) {
160             long time = ((Calendar)value).getTimeInMillis();
161             return (T) toDate(targetType, time);
162         }
163 
164         // Default String conversion
165         return (T) toDate(targetType, stringValue);
166     }

168     /***
169      * Convert a long value to the specified Date type for this
170      * <i>Converter</i>.
171      * <p>
172      *
173      * This method handles conversion to the following types:
174      * <ul>
175      *     <li><code>java.util.Date</code></li>
176      *     <li><code>java.util.Calendar</code></li>
177      *     <li><code>java.sql.Date</code></li>
178      *     <li><code>java.sql.Time</code></li>
179      *     <li><code>java.sql.Timestamp</code></li>
180      * </ul>
181      *
182      * @param type The Date type to convert to
183      * @param value The long value to convert.
184      * @return The converted date value.
185      */
186     protected static Object toDate(Class<?> type, long value) {
187 
188         // java.util.Date
189         if (type.equals(Date.class)) {
190             return new Date(value);
191         }
192 
193         // java.sql.Date
194         if (type.equals(java.sql.Date.class)) {
195             return new java.sql.Date(value);
196         }
197 
198         // java.sql.Time
199         if (type.equals(java.sql.Time.class)) {
200             return new java.sql.Time(value);
201         }
202 
203         // java.sql.Timestamp
204         if (type.equals(java.sql.Timestamp.class)) {
205             return new java.sql.Timestamp(value);
206         }
207 
208         // java.util.Calendar
209         if (type.equals(Calendar.class)) {
210             Calendar calendar = Calendar.getInstance();
211             calendar.setTimeInMillis(value);
212 //            calendar.setLenient(false);
213             return calendar;
214         }
215 
216         throw new UnsupportedOperationException("Date conversion failure: cannot convert long value ("+value+") to ("+type+")");
217     }

219     /***
220      * Default String to Date conversion.
221      * <p>
222      * This method handles conversion from a String to the following types:
223      * <ul>
224      *     <li><code>java.sql.Date</code></li>
225      *     <li><code>java.sql.Time</code></li>
226      *     <li><code>java.sql.Timestamp</code></li>
227      * </ul>
228      * <p>
229      * <strong>N.B.</strong> No default String conversion
230      * mechanism is provided for <code>java.util.Date</code>
231      * and <code>java.util.Calendar</code> type.
232      *
233      * @param type The Number type to convert to
234      * @param value The String value to convert.
235      * @return The converted Number value.
236      */
237     protected static Object toDate(Class<?> type, String value) {
238         // java.sql.Date
239         if (type.equals(java.sql.Date.class)) {
240             try {
241                 return java.sql.Date.valueOf(value);
242             } catch (IllegalArgumentException e) {
243                 throw new UnsupportedOperationException("Date conversion failure: " +
244                 "String must be in JDBC format [yyyy-MM-dd] to create a java.sql.Date");
245             }
246         }
247 
248         // java.sql.Time
249         if (type.equals(java.sql.Time.class)) {
250             try {
251                 return java.sql.Time.valueOf(value);
252             } catch (IllegalArgumentException e) {
253                 throw new UnsupportedOperationException("Date conversion failure: " +
254                 "String must be in JDBC format [HH:mm:ss] to create a java.sql.Time");
255             }
256         }
257 
258         // java.sql.Timestamp
259         if (type.equals(java.sql.Timestamp.class)) {
260             try {
261                 return java.sql.Timestamp.valueOf(value);
262             } catch (IllegalArgumentException e) {
263                 throw new UnsupportedOperationException("Date conversion failure: " +
264                         "String must be in JDBC format [yyyy-MM-dd HH:mm:ss.fffffffff] " +
265                 "to create a java.sql.Timestamp");
266             }
267         }
268 
269         throw new UnsupportedOperationException("Date conversion failure: cannot convert string value ("+value+") to ("+type+")");
270     }

272     /***
273      * Parse a String date value using the set of patterns.
274      *
275      * @param value The String date value.
276      * @return The converted Date object.
277      * @throws Exception if an error occurs parsing the date.
278      */
279     protected static Calendar parse(String value, DateFormat[] dateFormats) {
280         Calendar calendar = null;
281         if (dateFormats != null && dateFormats.length > 0) {
282             for (DateFormat format : dateFormats) {
283                 calendar = parse(value, format);
284                 if (calendar != null) {
285                     break;
286                 }
287             }
288         }
289         if (calendar == null) {
290             // just do the simple attempt now
291             DateFormat format = new SimpleDateFormat();
292             try {
293                 Date date = format.parse(value);
294                 if (date != null) {
295                     calendar = format.getCalendar();
296                 }
297             } catch (ParseException e) {
298                 calendar = null;
299             }
300         }
301         return calendar;
302     }

304     /***
305      * Parse a String into a <code>Calendar</code> object using the specified <code>DateFormat</code><br/>
306      * Does exact matching (non-lenient) only on the given pattern<br/>
307      *
308      * @param value The String date value.
309      * @param format The DateFormat to parse the String value.
310      * @return The converted Calendar object OR null if it cannot be converted
311      */
312     protected static Calendar parse(String value, DateFormat format) {
313         format.setLenient(false);
314         ParsePosition pos = new ParsePosition(0);
315         Date parsedDate = format.parse(value, pos); // ignore the result (use the Calendar)
316         Calendar calendar = null;
317         if (pos.getErrorIndex() >= 0 || pos.getIndex() != value.length() || parsedDate == null) {
318             calendar = null;
319         } else {
320             calendar = format.getCalendar();
321         }
322         return calendar;
323     }
