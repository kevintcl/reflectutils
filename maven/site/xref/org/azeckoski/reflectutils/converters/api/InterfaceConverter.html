20  /***
21   * Special converter which is used for converting to an interface when more control
22   * over the type of implementation to create under the interface is desired<br/>
23   * @see Converter for more details about the core converter interface and converters in general
24   * 
25   * @author Aaron Zeckoski (azeckoski @ gmail.com)
26   */
27  public interface InterfaceConverter<T> extends Converter<T> {
28  
29      /***
30       * Convert the value into the implementationType for the interface T
31       * 
32       * @param value the input value to be converted, this will never be null or the type which you are converting to
33       * since these simple cases are already handled in the {@link ConversionUtils}
34       * @param implementationType
35       * @return the converted value (can be null if desired)
36       * @throws UnsupportedOperationException if conversion cannot be performed successfully
37       */
38      public T convertInterface(Object value, Class<? extends T> implementationType);
