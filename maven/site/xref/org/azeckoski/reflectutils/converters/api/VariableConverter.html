17  import org.azeckoski.reflectutils.ConversionUtils;

20  /***
21   * Allows a converter to check to see if it can handle converting something before
22   * any of the converters are actually called, these converters are called before
23   * any of the other converters if they happen to return true, note that the order
24   * of the variable converters will match the order they were registered
25   * 
26   * @author Aaron Zeckoski (azeckoski @ gmail.com)
27   */
28  public interface VariableConverter extends BaseConverter {
29  
30      /***
31       * Check if this value can be converted to this type by this converter
32       * 
33       * @param value the input value to be converted, this will never be null or the type which you are converting to
34       * since these simple cases are already handled in the {@link ConversionUtils}
35       * @param toType the type to convert this value to
36       * @return true if it can handle the conversion, false otherwise
37       */
38      public boolean canConvert(Object value, Class<?> toType);
39  
40      /***
41       * Convert the specified input object into an output object of the type specified
42       *
43       * @param value the input value to be converted, this will never be null or the type which you are converting to
44       * since these simple cases are already handled in the {@link ConversionUtils}
45       * @param toType the type to convert this value to
46       * @return the converted value (can be null if desired)
47       * @throws UnsupportedOperationException if conversion cannot be performed successfully
48       */
49      public <T> T convert(Object value, Class<T> toType);
50  
51  }
