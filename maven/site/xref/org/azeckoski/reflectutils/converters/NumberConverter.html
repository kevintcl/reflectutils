26  /***
27   * Handles all the number conversions in a single converter,
28   * the other number converters are simply going to call through to this one<br/>
29   * Special handling for {@link Boolean}, {@link Date}, {@link Calendar}, and {@link String}
30   * <ul>
31   *     <li><code>java.math.BigDecimal</code></li>
32   *     <li><code>java.math.BigInteger</code></li>
33   *     <li><code>java.lang.Byte</code></li>
34   *     <li><code>java.lang.Double</code></li>
35   *     <li><code>java.lang.Float</code></li>
36   *     <li><code>java.lang.Integer</code></li>
37   *     <li><code>java.lang.Long</code></li>
38   *     <li><code>java.lang.Number</code></li>
39   *     <li><code>java.lang.Short</code></li>
40   * </ul>
41   * <br/>
42   * Based on code from commons bean utils but updated for generics and removed complexity and (mis)usage of Throwable<br/>
43   * 
44   * @author Aaron Zeckoski (azeckoski @ gmail.com)
45   */
46  public class NumberConverter implements Converter<Number> {
47  
48      public Number convert(Object value) {
49          return NumberConverter.convertToType(Number.class, value);
50      }

54      /***
55       * Convert the input object into a Number object of the
56       * specified type.
57       *
58       * @param targetType Data type to which this value should be converted.
59       * @param value The input value to be converted.
60       * @return The converted value.
61       * @throws Throwable if an error occurs converting to the specified type
62       */
63      @SuppressWarnings("unchecked")
64      public static <T> T convertToType(Class<T> targetType, Object value) {
65          if (value == null) {
66              throw new IllegalArgumentException("value to convert cannot be null");
67          }
68          Class<?> sourceType = value.getClass();
69  
70          // Handle Number
71          if (value instanceof Number) {
72              return (T) toNumber(sourceType, targetType, (Number)value);
73          }
74  
75          // Handle Boolean
76          if (value instanceof Boolean) {
77              return (T) toNumber(sourceType, targetType, ((Boolean)value).booleanValue() ? 1 : 0);
78          }
79  
80          // Handle Date
81          if (value instanceof Date) {
82              long time = ((Date)value).getTime();
83              return (T) toNumber(Long.class, targetType, time);
84          }
85  
86          // Handle Calendar
87          if (value instanceof Calendar) {
88              long time = ((Calendar)value).getTime().getTime();
89              return (T) toNumber(Long.class, targetType, time);
90          }
91  
92          // Convert all other types to String & handle
93          String stringValue = value.toString().trim();
94          if (stringValue.length() == 0) {
95              Object defaultValue = ConstructorUtils.getDefaultValue(targetType);
96              if (defaultValue == null) {
97                  throw new UnsupportedOperationException("Number convert failure: Could not convert empty string to target ("+targetType+") or get a default value for it");
98              }
99              return (T) defaultValue;
100         }
101 
102         // Convert/Parse a String
103         Number number = toNumber(sourceType, targetType, stringValue);
104         return (T) number;
105     }

107     /***
108      * Convert any Number object to the specified type for this <i>Converter</i>.
109      * <p>
110      * This method handles conversion to the following types:
111      * <ul>
112      *     <li><code>java.math.BigDecimal</code></li>
113      *     <li><code>java.math.BigInteger</code></li>
114      *     <li><code>java.lang.Byte</code></li>
115      *     <li><code>java.lang.Double</code></li>
116      *     <li><code>java.lang.Float</code></li>
117      *     <li><code>java.lang.Integer</code></li>
118      *     <li><code>java.lang.Long</code></li>
119      *     <li><code>java.lang.Number</code></li>
120      *     <li><code>java.lang.Short</code></li>
121      * </ul>
122      * @param sourceType The type being converted from
123      * @param targetType The Number type to convert to
124      * @param value The Number to convert.
125      * @return The converted value.
126      */
127     private static Number toNumber(Class<?> sourceType, Class<?> targetType, Number value) {
128 
129         // Correct Number type already
130         if ( ConstructorUtils.classEquals(targetType, value.getClass()) ) {
131             return value;
132         }
133 
134         // Byte
135         if (targetType.equals(Byte.class)) {
136             long longValue = value.longValue();
137             if (longValue > Byte.MAX_VALUE) {
138                 throw new UnsupportedOperationException("source (" + sourceType + ") value (" 
139                         + value + ") is too large for target (" + targetType + ")");
140             }
141             if (longValue < Byte.MIN_VALUE) {
142                 throw new UnsupportedOperationException("source (" + sourceType + ") value (" 
143                         + value + ") is too small for target (" + targetType + ")");
144             }
145             return new Byte(value.byteValue());
146         }
147 
148         // Short
149         if (targetType.equals(Short.class)) {
150             long longValue = value.longValue();
151             if (longValue > Short.MAX_VALUE) {
152                 throw new UnsupportedOperationException("source (" + sourceType + ") value (" 
153                         + value + ") is too large for target (" + targetType + ")");
154             }
155             if (longValue < Short.MIN_VALUE) {
156                 throw new UnsupportedOperationException("source (" + sourceType + ") value (" 
157                         + value + ") is too small for target (" + targetType + ")");
158             }
159             return new Short(value.shortValue());
160         }
161 
162         // Integer
163         if (targetType.equals(Integer.class)) {
164             long longValue = value.longValue();
165             if (longValue > Integer.MAX_VALUE) {
166                 throw new UnsupportedOperationException("source (" + sourceType + ") value (" 
167                         + value + ") is too large for target (" + targetType + ")");
168             }
169             if (longValue < Integer.MIN_VALUE) {
170                 throw new UnsupportedOperationException("source (" + sourceType + ") value (" 
171                         + value + ") is too small for target (" + targetType + ")");
172             }
173             return new Integer(value.intValue());
174         }
175 
176         // Long
177         if (targetType.equals(Long.class)) {
178             return new Long(value.longValue());
179         }
180 
181         // Float
182         if (targetType.equals(Float.class)) {
183             if (value.doubleValue() > Float.MAX_VALUE) {
184                 throw new UnsupportedOperationException("source (" + sourceType + ") value (" 
185                         + value + ") is too large for target (" + targetType + ")");
186             }
187             return new Float(value.floatValue());
188         }
189 
190         // Double
191         if (targetType.equals(Double.class)) {
192             return new Double(value.doubleValue());
193         }
194 
195         // BigDecimal
196         if (targetType.equals(BigDecimal.class)) {
197             if (value instanceof Float || value instanceof Double) {
198                 return new BigDecimal(value.toString());
199             } else if (value instanceof BigInteger) {
200                 return new BigDecimal((BigInteger)value);
201             } else {
202                 return BigDecimal.valueOf(value.longValue());
203             }
204         }
205 
206         // BigInteger
207         if (targetType.equals(BigInteger.class)) {
208             if (value instanceof BigDecimal) {
209                 return ((BigDecimal)value).toBigInteger();
210             } else {
211                 return BigInteger.valueOf(value.longValue());
212             }
213         }
214 
215         throw new UnsupportedOperationException("Number convert failure: cannot convert source ("+sourceType+") to target ("+targetType+")");
216     }

218     /***
219      * Default String to Number conversion.
220      * <p>
221      * This method handles conversion from a String to the following types:
222      * <ul>
223      *     <li><code>java.lang.Byte</code></li>
224      *     <li><code>java.lang.Short</code></li>
225      *     <li><code>java.lang.Integer</code></li>
226      *     <li><code>java.lang.Long</code></li>
227      *     <li><code>java.lang.Float</code></li>
228      *     <li><code>java.lang.Double</code></li>
229      *     <li><code>java.math.BigDecimal</code></li>
230      *     <li><code>java.math.BigInteger</code></li>
231      * </ul>
232      * @param sourceType The type being converted from
233      * @param targetType The Number type to convert to
234      * @param value The String value to convert.
235      *
236      * @return The converted Number value.
237      */
238     private static Number toNumber(Class<?> sourceType, Class<?> targetType, String value) {
239 
240         // Byte
241         if (targetType.equals(Byte.class)) {
242             return Byte.valueOf(value);
243         }
244 
245         // Short
246         if (targetType.equals(Short.class)) {
247             return Short.valueOf(value);
248         }
249 
250         // Integer
251         if (targetType.equals(Integer.class)) {
252             return Integer.valueOf(value);
253         }
254 
255         // Long
256         if (targetType.equals(Long.class)) {
257             return Long.valueOf(value);
258         }
259 
260         // Float
261         if (targetType.equals(Float.class)) {
262             return Float.valueOf(value);
263         }
264 
265         // Double
266         if (targetType.equals(Double.class)) {
267             return Double.valueOf(value);
268         }
269 
270         // BigDecimal
271         if (targetType.equals(BigDecimal.class)) {
272             return new BigDecimal(value);
273         }
274 
275         // BigInteger
276         if (targetType.equals(BigInteger.class)) {
277             return new BigInteger(value);
278         }
279 
280         throw new UnsupportedOperationException("Number convert failure: cannot convert string source ("+sourceType+") to target ("+targetType+")");
281     }
