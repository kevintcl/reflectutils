15  package org.azeckoski.reflectutils.converters;

17  import org.azeckoski.reflectutils.ClassLoaderUtils;
18  import org.azeckoski.reflectutils.converters.api.Converter;
19  
20  
21  /***
22   * Converts a string to a class (this is pretty much the only conversion supported)
23   * 
24   * @author Aaron Zeckoski (azeckoski @ gmail.com)
25   */
26  public class ClassConverter implements Converter<Class<?>> {
27  
28      public Class<?> convert(Object value) {
29          String className = value.toString();
30          Class<?> c = ClassLoaderUtils.getClassFromString(className);
31          if (c == null) {
32              throw new UnsupportedOperationException("Class convert failure: cannot convert source ("+value+") and type ("+value.getClass()+") to a Class");
33          }
34          return c;
35      }
36  
37  }
