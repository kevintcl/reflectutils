17  import java.lang.reflect.Array;
18  import java.util.Collection;
19  import java.util.Map;
20  
21  import org.azeckoski.reflectutils.ConstructorUtils;
22  import org.azeckoski.reflectutils.ConversionUtils;
23  import org.azeckoski.reflectutils.converters.api.VariableConverter;
24  
25  
26  /***
27   * Handles conversions to arrays from various types (including other arrays),
28   * also handles the special case of a comma separated list of strings which
29   * it will attempt to convert into an array of strings or whatever was requested
30   * 
31   * @author Aaron Zeckoski (azeckoski @ gmail.com)
32   */
33  public class ArrayConverter implements VariableConverter {
34  
35      /* (non-Javadoc)
36       * @see org.azeckoski.reflectutils.converters.api.VariableConverter#canConvert(java.lang.Object, java.lang.Class)
37       */
38      public boolean canConvert(Object value, Class<?> toType) {
39          if ( ConstructorUtils.isClassArray(toType)) {
40              return true;
41          }
42          return false;
43      }
44  
45      protected ConversionUtils getConversionUtils() {
46          return ConversionUtils.getInstance();
47      }
48  
49      /* (non-Javadoc)
50       * @see org.azeckoski.reflectutils.converters.api.VariableConverter#convert(java.lang.Object, java.lang.Class)
51       */
52      @SuppressWarnings("unchecked")
53      public <T> T convert(Object value, Class<T> toType) {
54          T convert = null;
55          Class<?> fromType = value.getClass();
56          Object toConvert = value;
57  
58          Class<?> componentType = toType.getComponentType();
59          if ( ConstructorUtils.isClassArray(fromType) ) {
60              // from array - to different type of array
61              int length = Array.getLength(toConvert);
62              convert = (T) Array.newInstance(componentType, length);
63              for (int i = 0; i < length; i++) {
64                  Object object = Array.get(toConvert, i);
65                  object = getConversionUtils().convert(object, componentType); // convert each value
66                  Array.set(convert, i, object);
67              }
68          } else if ( ConstructorUtils.isClassCollection(fromType) ) {
69              // from collection
70              Collection collection = (Collection) toConvert;
71              int length = collection.size();
72              convert = (T) Array.newInstance(componentType, length);
73              int i = 0;
74              for (Object object : collection) {
75                  object = getConversionUtils().convert(object, componentType); // convert each value
76                  Array.set(convert, i, object);
77                  i++;
78              }
79          } else if ( ConstructorUtils.isClassMap(fromType) ) {
80              // from map
81              Map map = (Map) toConvert;
82              int length = map.size();
83              convert = (T) Array.newInstance(componentType, length);
84              int i = 0;
85              for (Object object : map.values()) {
86                  object = getConversionUtils().convert(object, componentType); // convert each value
87                  Array.set(convert, i, object);
88                  i++;
89              }
90          } else {
91              // from scalar
92              String valueString = toConvert.toString();
93              if ("".equals(valueString)) {
94                  // empty string becomes empty array
95                  convert = (T) Array.newInstance(componentType, 0);
96              } else if (valueString.indexOf(',') > 0) {
97                  // support comma separated string to array
98                  String[] parts = valueString.split(",");
99                  convert = (T) Array.newInstance(componentType, parts.length);
100                 for (int i = 0; i < parts.length; i++) {
101                     Object object = getConversionUtils().convert(parts[i].trim(), componentType); // convert each value
102                     Array.set(convert, i, object);
103                 }
104             } else {
105                 // just put it in the array
106                 convert = (T) Array.newInstance(componentType, 1);
107                 Object object = getConversionUtils().convert(toConvert, componentType); // convert each value
108                 Array.set(convert, 0, object);
109             }
110         }
111         return convert;
112     }
