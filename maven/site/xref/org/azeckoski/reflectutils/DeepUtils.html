32  /***
33   * Class which provides methods for handling deep operations: <br/>
34   * deep copy - make a copy of one object into another which traverses the entire object and duplicates all data down to immutable/simple fields <br/>
35   * deep clone - clone an object by traversing it and making a copy of every field <br/>
36   * deep map - turn an object into a map with only simple objects in it, all contained objects become nested maps <br/>
37   * populate - turn a map into an object, can handle params maps (from a servlet request) and nested or non-nested string->object maps
38   * 
39   * @author Aaron Zeckoski (azeckoski @ gmail.com)
40   */

72      /***
73       * Deep clone an object and all the values in it into a brand new object of the same type,
74       * this will traverse the bean and will make new objects for all non-null values contained in the object
75       * 
76       * @param <T>
77       * @param object any java object, this can be a list, map, array, or any simple
78       * object, it does not have to be a custom object or even a java bean,
79       * also works with DynaBeans
80       * @param maxDepth the number of objects to follow when traveling through the object and copying
81       * the values from it, 0 means to only copy the simple values in the object, any objects will
82       * be ignored and will end up as nulls, 1 means to follow the first objects found and copy all
83       * of their simple values as well, and so forth
84       * @param fieldNamesToSkip (optional) the names of fields to skip while cloning this object,
85       * this only has an effect on the bottom level of the object, any fields found
86       * on child objects will always be copied (if the maxDepth allows)
87       * @return the cloned object
88       */
89      @SuppressWarnings("unchecked")
90      public <T> T deepClone(T object, int maxDepth, String[] fieldNamesToSkip) {
91          Set<String> skip = ArrayUtils.makeSetFromArray(fieldNamesToSkip);
92          T clone = (T) internalDeepClone(object, CopyDestination.ORIGINAL, maxDepth, skip, 0, true, false);
93          return clone;
94      }

96      /***
97       * Deep copies one object into another, this is primarily for copying between identical types of objects but
98       * it can also handle copying between objects which are quite different, 
99       * this does not just do a reference copy of the values but actually creates new objects in the current classloader
100      * 
101      * @param orig the original object to copy from
102      * @param dest the object to copy the values to (must have the same fields with the same types)
103      * @param maxDepth the number of objects to follow when traveling through the object and copying
104      * the values from it, 0 means to only copy the simple values in the object, any objects will
105      * be ignored and will end up as nulls, 1 means to follow the first objects found and copy all
106      * of their simple values as well, and so forth
107      * @param fieldNamesToSkip (optional) the names of fields to skip while cloning this object,
108      * this only has an effect on the bottom level of the object, any fields found
109      * on child objects will always be copied (if the maxDepth allows)
110      * @param ignoreNulls if true then nulls are not copied and the destination retains the value it has,
111      * if false then nulls are copied and the destination value will become a null if the original value is a null
112      * @throws IllegalArgumentException if the copy cannot be completed because the objects to copy do not have matching fields or types
113      */
114     public void deepCopy(Object orig, Object dest, int maxDepth, String[] fieldNamesToSkip, boolean ignoreNulls) {
115         Set<String> skip = ArrayUtils.makeSetFromArray(fieldNamesToSkip);
116         internalDeepCopy(orig, dest, maxDepth, skip, ignoreNulls, true);
117     }

119     /***
120      * This handles the copying of objects to maps, it is recursive and is a deep operation which
121      * traverses the entire object and clones every part of it. When converting to a map this will ensure
122      * that there are no objects which are not part of java.lang or java.util in the new map<br/>
123      * NOTE: This can handle simple objects (non-maps and non-beans) but will have to make up the initial map key
124      * in the returned map, "data" will be used as the key<br/>
125      * NOTE: Nulls are allowed to pass through this method (i.e. passing in a null object results in a null output)
126      * 
127      * @param bean any java object
128      * @param maxDepth the number of objects to follow when traveling through the object and copying
129      * the values from it, 0 means to only copy the simple values in the object, any objects will
130      * be ignored and will end up as nulls, 1 means to follow the first objects found and copy all
131      * of their simple values as well, and so forth
132      * @param fieldNamesToSkip (optional) the names of fields to skip while cloning this object,
133      * this only has an effect on the bottom level of the object, any fields found
134      * on child objects will always be copied (if the maxDepth allows)
135      * @param ignoreNulls if true then nulls are not copied and the destination retains the value it has,
136      * if false then nulls are copied and the destination value will become a null if the original value is a null
137      * @param ignoreTransient if true then all transient fields will be skipped, useful when serializing
138      * @param initialKey (optional) the initial key to use when simple objects are converted to maps, defaults to "data"
139      * @return the resulting map which contains the cloned data from the object
140      */
141     @SuppressWarnings("unchecked")
142     public Map<String, Object> deepMap(Object object, int maxDepth, String[] fieldNamesToSkip, boolean ignoreNulls, boolean ignoreTransient, String initialKey) {
143         Set<String> skip = ArrayUtils.makeSetFromArray(fieldNamesToSkip);
144         Object clone = internalDeepClone(object, CopyDestination.MAP, maxDepth, skip, 0, ignoreNulls, ignoreTransient);
145         Map<String, Object> map = null;
146         if (clone != null) {
147             if ( ConstructorUtils.isClassMap(clone.getClass())) {
148                 map = (Map<String, Object>) clone;
149             } else {
150                 // handle the case of non-map/bean by wrapping in a map
151                 if (initialKey == null) {
152                     initialKey = "data";
153                 }
154                 map = new HashMap<String, Object>();
155                 map.put(initialKey, clone);
156             }
157         }
158         return map;
159     }

161     /***
162      * Populates an object with the values in the properties map,
163      * this will not fail if the fieldName in the map is not a property on the
164      * object or the fieldName cannot be written to with the value in the object.
165      * This will attempt to convert the provided object values into the right values
166      * to place in the object<br/>
167      * <b>NOTE:</b> simple types like numbers and strings can almost always be converted from
168      * just about anything though they will probably end up as 0 or ""<br/>
169      * Setting fields supports simple, nested, indexed, and mapped values:<br/>
170      * <b>Simple:</b> Get/set a field in a bean (or map), Example: "title", "id"<br/>
171      * <b>Nested:</b> Get/set a field in a bean which is contained in another bean, Example: "someBean.title", "someBean.id"<br/>
172      * <b>Indexed:</b> Get/set a list/array item by index in a bean, Example: "myList[1]", "anArray[2]"<br/>
173      * <b>Mapped:</b> Get/set a map entry by key in a bean, Example: "myMap(key)", "someMap(thing)"<br/>
174      * 
175      * @param object any object
176      * @param properties a map of fieldNames -> Object
177      * @return the list of fieldNames which were successfully written to the object
178      * @throws IllegalArgumentException if the arguments are invalid
179      */
180     public List<String> populate(Object object, Map<String, Object> properties) {
181         if (object == null || properties == null) {
182             throw new IllegalArgumentException("object and properties cannot be null");
183         }
184         Map<String, Class<?>> fieldTypes = getFieldUtils().getFieldTypes(object.getClass(), FieldsFilter.ALL);
185         List<String> writtenFields = new ArrayList<String>();
186         for (Entry<String, Object> entry : properties.entrySet()) {
187             String fieldName = entry.getKey();
188             if (fieldName != null && ! "".equals(fieldName)) {
189                 Class<?> targetType = fieldTypes.get(fieldName);
190                 if (targetType != null) {
191                     Object value = entry.getValue();
192                     try {
193                         if ( ConstructorUtils.isClassSimple(targetType) ) {
194                             // this should write the value or fail
195                             getFieldUtils().setFieldValue(object, fieldName, value, true);
196                         } else {
197                             // value is a bean so we will dig into it further
198                             Object dest = getFieldUtils().getFieldValue(object, fieldName);
199                             if (dest == null) {
200                                 // need to construct the destination
201                                 if (value.getClass().isArray() && targetType.isArray()) {
202                                     // special case for array to array since the destination must match
203                                     dest = ArrayUtils.template((Object[])value);
204                                 } else {
205                                     dest = getConstructorUtils().constructClass(targetType);
206                                 }
207                                 getFieldUtils().setFieldValue(object, fieldName, dest, true);
208                             }
209                             internalDeepCopy(value, dest, 5, null, false, true);
210                         }
211                         writtenFields.add(fieldName); // record that the value was written successfully
212                     } catch (RuntimeException e) {
213                         // this is OK, it just means we did not write the value
214                         continue;
215                     }
216                 }
217             }
218         }
219         return writtenFields;
220     }

222     /***
223      * Populates an object with the String array values in the params map,
224      * this will not fail if the fieldName in the map is not a property on the
225      * object or the fieldName cannot be written to with the value in the object<br/>
226      * Arrays which are length 1 will be converted to a string before they are set on the target field <br/>
227      * Setting fields supports simple, nested, indexed, and mapped values:<br/>
228      * <b>Simple:</b> Get/set a field in a bean (or map), Example: "title", "id"<br/>
229      * <b>Nested:</b> Get/set a field in a bean which is contained in another bean, Example: "someBean.title", "someBean.id"<br/>
230      * <b>Indexed:</b> Get/set a list/array item by index in a bean, Example: "myList[1]", "anArray[2]"<br/>
231      * <b>Mapped:</b> Get/set a map entry by key in a bean, Example: "myMap(key)", "someMap(thing)"<br/>
232      * 
233      * @param object any object
234      * @param params a map of fieldNames -> String[] (normally from an http request)
235      * @return the list of fieldNames which were successfully written to the object
236      * @throws IllegalArgumentException if the arguments are invalid
237      */
238     public List<String> populateFromParams(Object object, Map<String, String[]> params) {
239         if (object == null || params == null) {
240             throw new IllegalArgumentException("object and params cannot be null");
241         }
242         List<String> writtenFields = new ArrayList<String>();
243         for (Entry<String, String[]> entry : params.entrySet()) {
244             String fieldName = entry.getKey();
245             if (fieldName != null && ! "".equals(fieldName)) {
246                 String[] value = entry.getValue();
247                 try {
248                     // this should write the value or fail
249                     if (value != null) {
250                         Object toSet = value;
251                         if (value.length == 0) {
252                             toSet = "";
253                         } else if (value.length == 1) {
254                             toSet = value[0];
255                         }
256                         getFieldUtils().setFieldValue(object, fieldName, toSet, true);
257                         writtenFields.add(fieldName); // record that the value was written successfully
258                     }
259                 } catch (RuntimeException e) {
260                     // this is OK, it just means we did not write the value
261                     continue;
262                 }
263             }
264         }
265         return writtenFields;
266     }

269     public static enum CopyDestination {ORIGINAL, MAP};
270     /***
271      * This handles the cloning of objects to objects or maps, it is recursive and is a deep operation which
272      * traverses the entire object and clones every part of it, when converting to a map this will ensure
273      * that there are no objects which are not part of java.lang or java.util in the new map
274      * @param bean any java object
275      * @param dest the type of destination: ORIGINAL = an object of type matching bean, MAP = an ordered map
276      * @param maxDepth the number of objects to follow when traveling through the object and copying
277      * the values from it, 0 means to only copy the simple values in the object, any objects will
278      * be ignored and will end up as nulls, 1 means to follow the first objects found and copy all
279      * of their simple values as well, and so forth
280      * @param fieldNamesToSkip the names of fields to skip while cloning this object,
281      * this only has an effect on the bottom level of the object, any fields found
282      * on child objects will always be copied (if the maxDepth allows)
283      * @param currentDepth the current depth for recursion and halting when appropriate depth is reached
284      * @param ignoreNulls if true then nulls are not copied and the destination retains the value it has,
285      * if false then nulls are copied and the destination value will become a null if the original value is a null
286      * @param ignoreTransient if true then all transient fields will be skipped, useful when serializing
287      * @return the object or map which has been cloned
288      */
289     @SuppressWarnings("unchecked")
290     protected Object internalDeepClone(Object bean, CopyDestination dest, int maxDepth, Set<String> fieldNamesToSkip, int currentDepth, boolean ignoreNulls, boolean ignoreTransient) {
291         Object copy = null;
292         if ( bean != null ) {
293             Class<?> beanClass = bean.getClass();
294             // always copy the simple types if possible
295             if (ConstructorUtils.isClassSimple(beanClass)) {
296                 copy = bean;
297             } else {
298                 if (currentDepth <= maxDepth) {
299                     currentDepth++;
300                     try {
301                         // now do the cloning based on the thing to clone
302                         if (beanClass.isArray()) {
303                             // special case, use array reflection
304                             // make new array
305                             int length = ArrayUtils.size((Object[]) bean);
306                             Class<?> componentType = beanClass.getComponentType();
307                             if ( ConstructorUtils.isClassBean(componentType) 
308                                     && CopyDestination.MAP.equals(dest)) {
309                                 // special array component type for arrays of objects when map converting
310                                 componentType = ArrayOrderedMap.class;
311                             }
312                             copy = ArrayUtils.create(componentType, length);
313                             // now copy the stuff into it
314                             for (int i = 0; i < length; i++) {
315                                 Object value = Array.get(bean, i);
316                                 Object clone = internalDeepClone(value, dest, maxDepth, null, currentDepth, ignoreNulls, ignoreTransient);
317                                 Array.set(copy, i, clone);
318                             }
319                         } else if (Collection.class.isAssignableFrom(beanClass)) {
320                             // special case, clone everything in the list
321                             copy = getConstructorUtils().constructClass(beanClass); // make new list
322                             for (Object element : (Collection) bean) {
323                                 Object clone = internalDeepClone(element, dest, maxDepth, null, currentDepth, ignoreNulls, ignoreTransient);
324                                 ((Collection) copy).add( clone );
325                             }
326                         } else if (Map.class.isAssignableFrom(beanClass)) {
327                             // special case, clone everything in the map except keys
328                             copy = getConstructorUtils().constructClass(beanClass); // make new map
329                             for (Object key : ((Map) bean).keySet()) {
330                                 if ( fieldNamesToSkip != null
331                                         && fieldNamesToSkip.contains(key) ) {
332                                     continue; // skip to next
333                                 }
334                                 Object value = ((Map) bean).get(key);
335                                 if (value == null && ignoreNulls) {
336                                     continue;
337                                 }
338                                 try {
339                                     ((Map) copy).put(key, 
340                                             internalDeepClone(value, dest, maxDepth, null, currentDepth, ignoreNulls, ignoreTransient) );
341                                 } catch (NullPointerException e) {
342                                     // map does not support nulls so skip and keep going
343                                     continue;
344                                 }                     
345                             }
346                         } else if (getFieldAdapter().isAdaptableClass(beanClass)) {
347                             // special handling for dynabeans
348                             if (CopyDestination.MAP.equals(dest)) {
349                                 copy = new ArrayOrderedMap<String, Object>();
350                             } else {
351                                 copy = getFieldAdapter().newInstance(bean); // make new dynabean
352                             }
353                             List<String> propertyNames = getFieldAdapter().getPropertyNames(bean);
354                             for (String name : propertyNames) {
355                                 if ( fieldNamesToSkip != null
356                                         && fieldNamesToSkip.contains(name) ) {
357                                     continue; // skip to next
358                                 }
359                                 try {
360                                     Object value = getFieldAdapter().getSimpleValue(bean, name);
361                                     if (value == null && ignoreNulls) {
362                                         continue;
363                                     }
364                                     if (CopyDestination.MAP.equals(dest)) {
365                                         ((Map<String, Object>) copy).put(name, 
366                                                 internalDeepClone(value, dest, maxDepth, null, currentDepth, ignoreNulls, ignoreTransient) );
367                                     } else {
368                                         getFieldUtils().setFieldValue(copy, name, 
369                                                 internalDeepClone(value, dest, maxDepth, null, currentDepth, ignoreNulls, ignoreTransient) );
370                                     }
371                                 } catch (IllegalArgumentException e) {
372                                     // this is ok, field might not be readable so we will not clone it, continue on
373                                 }
374                             }
375                         } else {
376                             // regular javabean
377                             FieldsFilter filter = FieldsFilter.COMPLETE;
378                             if (CopyDestination.MAP.equals(dest)) {
379                                 copy = new ArrayOrderedMap<String, Object>();
380                                 // maps should pick up all readable fields
381                                 if (ignoreTransient) {
382                                     filter = FieldsFilter.SERIALIZABLE;
383                                 } else {
384                                     filter = FieldsFilter.READABLE;
385                                 }
386                             } else {
387                                 copy = getConstructorUtils().constructClass(beanClass); // make new bean
388                             }
389                             ClassFields<?> cf = getFieldUtils().analyzeClass(beanClass);
390                             Map<String, Class<?>> types = cf.getFieldTypes(filter);
391                             for (String name : types.keySet()) {
392                                 if (ClassFields.FIELD_CLASS.equals(name)) {
393                                     continue; // No point in trying to set/get an object's class
394                                 }
395                                 if ( fieldNamesToSkip != null
396                                         && fieldNamesToSkip.contains(name) ) {
397                                     continue; // skip to next
398                                 }
399                                 try {
400                                     Object value = getFieldUtils().getFieldValue(bean, name);
401                                     if (value == null && ignoreNulls) {
402                                         continue;
403                                     }
404                                     if (CopyDestination.MAP.equals(dest)) {
405                                         ((Map<String, Object>) copy).put(name, 
406                                                 internalDeepClone(value, dest, maxDepth, null, currentDepth, ignoreNulls, ignoreTransient) );                     
407                                     } else {
408                                         getFieldUtils().setFieldValue(copy, name, 
409                                                 internalDeepClone(value, dest, maxDepth, null, currentDepth, ignoreNulls, ignoreTransient) );
410                                     }
411                                 } catch (FieldnameNotFoundException e) {
412                                     // this is ok, field might not exist or might not be readable so we skip it
413                                 } catch (IllegalArgumentException e) {
414                                     // this is ok, failure should not stop the clone
415                                 }
416                             }
417                         }
418                     } catch (Exception e) {
419                         // catch any possible thrown exception and translate (this is not a mistake so ignore findbugs)
420                         throw new RuntimeException("Failure during deep cloning ("+beanClass+") maxDepth="+maxDepth+", currentDepth="+currentDepth+": " + e.getMessage(), e);
421                     }
422                 }
423             }
424         }
425         return copy;
426     }

428     /***
429      * @param orig the original object to copy from
430      * @param dest the object to copy the values to (must have the same fields with the same types)
431      * @param maxDepth the number of objects to follow when traveling through the object and copying
432      * the values from it, 0 means to only copy the simple values in the object, any objects will
433      * be ignored and will end up as nulls, 1 means to follow the first objects found and copy all
434      * of their simple values as well, and so forth
435      * @param fieldNamesToSkip the names of fields to skip while cloning this object,
436      * this only has an effect on the bottom level of the object, any fields found
437      * on child objects will always be copied (if the maxDepth allows)
438      * @param ignoreNulls if true then nulls are not copied and the destination retains the value it has,
439      * if false then nulls are copied and the destination value will become a null if the original value is a null
440      * @param autoConvert if true the the original will be converted to the dest type if needed
441      * @throws IllegalArgumentException if the copy cannot be completed because the objects to copy do not have matching fields or types
442      */
443     @SuppressWarnings("unchecked")
444     protected void internalDeepCopy(Object orig, Object dest, int maxDepth, Set<String> fieldNamesToSkip, boolean ignoreNulls, boolean autoConvert) {
445         if (orig == null || dest == null) {
446             throw new IllegalArgumentException("original object and destination object must not be null, if you want to clone the object then use the clone method");
447         }
448 
449         int currentDepth = 1;
450         Class<?> origClass = orig.getClass();
451         Class<?> destClass = dest.getClass();
452 
453         // check if copy is possible first
454         if (ConstructorUtils.getImmutableTypes().contains(destClass)) {
455             // cannot copy to an immutable object
456             throw new IllegalArgumentException("Cannot copy to an immutable object");
457         }
458         if (autoConvert && ! ConstructorUtils.isClassBean(origClass) ) {
459             // attempt to convert non-beans into the type we are copying to, fail if it does not work
460             // This has to be done because the copy will simply fail if we try to copy between non-beans which are not equivalent
461             try {
462                 orig = getConversionUtils().convert(orig, destClass);
463                 origClass = destClass;
464             } catch (UnsupportedOperationException e) {
465                 throw new IllegalArgumentException("Cannot copy from an immutable object to a complex object without converter support");
466             }
467         }
468         // copy orig to dest
469         try {
470             if (origClass.isArray()) {
471                 // special case, copy and overwrite existing array values
472                 if (destClass.isArray()) {
473                     // TODO if the dest array is empty then fail
474                     try {
475                         for (int i = 0; i < Array.getLength(orig); i++) {
476                             Object value = Array.get(orig, i);
477                             if (ignoreNulls && value == null) {
478                                 // don't copy this null over the existing value
479                             } else {
480                                 Array.set(dest, i, 
481                                         internalDeepClone(value, CopyDestination.ORIGINAL, maxDepth, null, currentDepth, ignoreNulls, false));
482                             }
483                         }
484                     } catch (ArrayIndexOutOfBoundsException e) {
485                         // partial copy is ok, continue on
486                     }
487                 } else {
488                     throw new IllegalArgumentException("Cannot copy a simple value to a complex object");                  
489                 }
490             } else if (Collection.class.isAssignableFrom(origClass)) {
491                 // special case, copy everything from orig and add to dest
492                 for (Object value : (Collection) orig) {
493                     if (ignoreNulls && value == null) {
494                         // don't copy this null over the existing value
495                     } else {
496                         ((Collection) dest).add( 
497                                 internalDeepClone(value, CopyDestination.ORIGINAL, maxDepth, null, currentDepth, ignoreNulls, false));
498                     }
499                 }
500             } else if (Map.class.isAssignableFrom(origClass)) {
501                 // special case clone everything in the map except keys
502                 for (Object key : ((Map) orig).keySet()) {
503                     if ( fieldNamesToSkip != null
504                             && fieldNamesToSkip.contains(key) ) {
505                         continue; // skip to next
506                     }
507                     Object value = ((Map) orig).get(key);
508                     if (ignoreNulls && value == null) {
509                         // don't copy this null over the existing value
510                     } else {
511                         ((Map) dest).put(key, 
512                                 internalDeepClone(value, CopyDestination.ORIGINAL, maxDepth, null, currentDepth, ignoreNulls, false));
513                     }
514                 }
515             } else if (getFieldAdapter().isAdaptableClass(origClass)) {
516                 List<String> propertyNames = getFieldAdapter().getPropertyNames(orig);
517                 for (String name : propertyNames) {
518                     if ( fieldNamesToSkip != null
519                             && fieldNamesToSkip.contains(name) ) {
520                         continue; // skip to next
521                     }
522                     try {
523                         Object value = getFieldAdapter().getSimpleValue(orig, name);
524                         if (ignoreNulls && value == null) {
525                             // don't copy this null over the existing value
526                         } else {
527                             getFieldUtils().setFieldValue(dest, name, 
528                                     internalDeepClone(value, CopyDestination.ORIGINAL, maxDepth, null, currentDepth, ignoreNulls, false));
529                         }
530                     } catch (FieldnameNotFoundException e) {
531                         // it is ok for the objects to not be the same
532                     } catch (IllegalArgumentException e) {
533                         // it is ok for the objects to not be the same
534                     }
535                 }
536             } else {
537                 // regular javabean
538                 Map<String, Object> values = getFieldUtils().getFieldValues(orig);
539                 for (Entry<String, Object> entry : values.entrySet()) {
540                     String name = entry.getKey();
541                     if (ClassFields.FIELD_CLASS.equals(name)) {
542                         continue; // No point in trying to set an object's class
543                     }
544                     if ( fieldNamesToSkip != null
545                             && fieldNamesToSkip.contains(name) ) {
546                         continue; // skip to next
547                     }
548                     try {
549                         Object value = getFieldUtils().getFieldValue(orig, name);
550                         if (ignoreNulls && value == null) {
551                             // don't copy this null over the existing value
552                         } else {
553                             getFieldUtils().setFieldValue(dest, name, 
554                                     internalDeepClone(value, CopyDestination.ORIGINAL, maxDepth, null, currentDepth, ignoreNulls, false));
555                         }
556                     } catch (FieldnameNotFoundException e) {
557                         // this is ok, field might not exist or might not be readable so we skip it
558                     } catch (IllegalArgumentException e) {
559                         // this is ok, failure should not stop the clone
560                     }
561                 }
562             }
563         } catch (Exception e) {
564             // catch any possible thrown exception and translate (this is not a mistake so ignore findbugs)
565             throw new IllegalArgumentException("Failure copying " + orig + " ("+origClass+") to " 
566                     + dest + " ("+destClass+")" + e.getMessage(), e);
567         }
568     }
