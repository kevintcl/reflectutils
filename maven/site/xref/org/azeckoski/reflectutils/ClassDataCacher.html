50      /***
51       * Construct and specify the field finding mode and your own cache when caching class data, must implement the standard map interface but
52       * only the following methods are required:<br/>
53       * {@link Map#clear()}, {@link Map#size()}, {@link Map#put(Object, Object)}, {@link Map#get(Object)} <br/>

61      public ClassDataCacher(FieldFindMode fieldFindMode, Map<Class<?>, ClassFields> reflectionCache) {
62          setFieldFindMode(fieldFindMode);
63          setReflectionCache(reflectionCache);

65          ClassDataCacher.setInstance(this);
66      }

70      protected FieldFindMode fieldFindMode = FieldFindMode.HYBRID;
71      /***
72       * Set the mode used to find fields on classes (default {@link FieldFindMode#HYBRID}) <br/>
73       * <b>WARNING</b>: changing modes will clear the existing cache
74       * 
75       * @param fieldFindMode see FieldFindMode enum for details
76       * @see FieldFindMode

78      public void setFieldFindMode(FieldFindMode fieldFindMode) {
79          if (fieldFindMode == null) {
80              fieldFindMode = FieldFindMode.HYBRID;
81          }
82          if (! this.fieldFindMode.equals(fieldFindMode)) {
83              // need to clear the cache if we change the mode
84              getReflectionCache().clear();
85          }
86          this.fieldFindMode = fieldFindMode;
87      }

112     public int cacheHits = 0;

115     @SuppressWarnings("unchecked")
116     protected Map<Class<?>, ClassFields> reflectionCache = null;
117     @SuppressWarnings("unchecked")
118     protected Map<Class<?>, ClassFields> getReflectionCache() {
119         if (reflectionCache == null) {
120             // internally we are using the ReferenceMap (from the Guice codebase)
121             // modeled after the groovy reflection caching (weak -> soft)
122             reflectionCache = new ReferenceMap<Class<?>,ClassFields>(ReferenceType.WEAK, ReferenceType.SOFT);
123         }
124         return reflectionCache;
125     }
126     /***
127      * Set the cache to be used for holding the reflection data, 
128      * this allows control over where the reflection caches are stored,
129      * this should store the data in a way that it will not hold open the classloader the class comes from <br/>
130      * Note that you can set this to a map implementation which does not store anything to disable caching if you like
131      * 
132      * @param reflectionCache a cache for holding class cache data (implements map), null to use the default internal cache
133      */
134     @SuppressWarnings("unchecked")
135     public void setReflectionCache(Map<Class<?>, ClassFields> reflectionCache) {
136         if (reflectionCache != null) {
137             this.reflectionCache.clear();
138             this.reflectionCache = reflectionCache;
139         } else {
140             getReflectionCache();
141         }
142     }

144     /***
145      * Get the class fields analysis of a class which contains information about that class and its fields,
146      * includes annotations, fields/properties, etc. packaged in a way which makes the data easy to get to,
147      * use the {@link ClassData} object to get to the more raw data
148      * 
149      * @param <T>
150      * @param cls any {@link Class}
151      * @return the ClassFields analysis object which contains the information about this class
152      */
153     @SuppressWarnings("unchecked")
154     public <T> ClassFields<T> getClassFields(Class<T> cls) {
155         if (cls == null) {
156             throw new IllegalArgumentException("cls (type) cannot be null");
157         }
158         lookups++;
159         ClassFields<T> cf = getReflectionCache().get(cls);
160         if (cf == null) {
161             // make new and put in cache
162             cf = new ClassFields<T>(cls, FieldFindMode.HYBRID, false, includeClassField);
163             getReflectionCache().put(cls, cf);
164             cacheMisses++;
165         } else {
166             cacheHits++;
167         }
168         return cf;
169     }

171     /***
172      * Convenience Method: <br/>
173      * Gets the class data object which contains information about this class,
174      * will retrieve this from the class data cache if available or generate it if not<br/>
175      * This is also available from the {@link ClassFields} object
176      * 
177      * @param <T>
178      * @param cls any {@link Class}
179      * @return the class data cache object (contains reflected data from this class)
180      */
181     public <T> ClassData<T> getClassData(Class<T> cls) {
182         ClassFields<T> cf = getClassFields(cls);
183         return cf.getClassData();
184     }

186     /***
187      * Convenience Method: <br/>
188      * Analyze an object and produce an object which contains information about it and its fields,
189      * see {@link ClassDataCacher#getClassData(Class)}
190      * 
191      * @param <T>
192      * @param obj any {@link Object}
193      * @return the ClassFields analysis object which contains the information about this objects class
194      * @throws IllegalArgumentException if obj is null
195      */
196     @SuppressWarnings("unchecked")
197     public <T> ClassFields<T> getClassFieldsFromObject(Object obj) {
198         if (obj == null) {
199             throw new IllegalArgumentException("obj cannot be null");
200         }
201         Class<T> cls = (Class<T>) obj.getClass();
202         return getClassFields(cls);
203     }
204 
205     /***
206      * Convenience Method: <br/>
207      * Gets the class data object which contains information about this objects class,
208      * will retrieve this from the class data cache if available or generate it if not<br/>
209      * This is also available from the {@link ClassFields} object
210      * 
211      * @param <T>
212      * @param obj any {@link Object}
213      * @return the raw ClassData cache object which contains reflection data about this objects class
214      * @throws IllegalArgumentException if obj is null
215      */
216     public <T> ClassData<T> getClassData(Object obj) {
217         ClassFields<T> cf = getClassFieldsFromObject(obj);
218         return cf.getClassData();
219     }

242     protected static SoftReference<ClassDataCacher> instanceStorage;
243     /***
244      * Get a singleton instance of this class to work with (stored statically) <br/>
245      * <b>WARNING</b>: do not hold onto this object or cache it yourself, call this method again if you need it again
246      * @return a singleton instance of this class
247      */
248     public static ClassDataCacher getInstance() {
249         ClassDataCacher instance = (instanceStorage == null ? null : instanceStorage.get());
250         if (instance == null) {
251             instance = ClassDataCacher.setInstance(null);
252         }
253         return instance;
254     }
255     /***
256      * Set the singleton instance of the class which will be stored statically
257      * @param instance the instance to use as the singleton instance
258      */
259     public static ClassDataCacher setInstance(ClassDataCacher newInstance) {
260         ClassDataCacher instance = newInstance;
261         if (instance == null) {
262             instance = new ClassDataCacher();
263             instance.singleton = true;
264         }
265         ClassDataCacher.timesCreated++;
266         instanceStorage = new SoftReference<ClassDataCacher>(instance);
267         return instance;
268     }
