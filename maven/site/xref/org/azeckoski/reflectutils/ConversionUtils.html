94       * Set the object converters to add to the default converters
95       * @param converters a map of converters to add to the default set
96       */
97      public void setConverters(Map<Class<?>, Converter<?>> converters) {
98          loadDefaultConverters();
99          if (converters != null) {
100             for (Entry<Class<?>, Converter<?>> entry : converters.entrySet()) {
101                 if (entry.getKey() != null && entry.getValue() != null) {
102                     this.converters.put(entry.getKey(), entry.getValue());
103                 }
104             }
105         }
106     }

108         if (converters == null) {
109             loadDefaultConverters();
110         }

216     /***
217      * Added for apache commons beanutils compatibility,
218      * you should probably use {@link #convert(Object, Class)}<br/>
219      * Convert the specified value into a String.  If the specified value
220      * is an array, the first element (converted to a String) will be
221      * returned.  The registered {@link Converter} for the
222      * <code>java.lang.String</code> class will be used, which allows
223      * applications to customize Object->String conversions (the default
224      * implementation simply uses toString()).
225      *  
226      * @param object any object
227      * @return the string OR null if one cannot be found
228      */
229     public String convertToString(Object object) {
230         // code here is basically from ConvertUtilsBeans 1.8.0
231         String convert = null;
232         if (object == null) {
233             convert = null;
234         } else if (object.getClass().isArray()) {
235             if (Array.getLength(object) < 1) {
236                 convert = null;
237             } else {
238                 Object value = Array.get(object, 0);
239                 if (value == null) {
240                     convert = null;
241                 } else {
242                     Converter<String> converter = getConverter(String.class);
243                     return converter.convert(value);
244                 }
245             }
246         } else {
247             Converter<String> converter = getConverter(String.class);
248             return converter.convert(object);
249         }
250         return convert;
251     }

256      * Convert the string value to an object of the specified class (if
257      * possible).  Otherwise, return a String representation of the value.
258      *

274     /***
275      * Converts an object to any other object if possible using the current set of converters,
276      * will allow nulls to pass through unless there is a converter which will handle them <br/>
277      * Includes special handling for primitives, arrays, and collections 
278      * (will take the first value when converting to scalar)
279      * 
280      * @param <T>
281      * @param value any object
282      * @param type any class type that you want to try to convert the object to
283      * @return the converted value (allows null to pass through except in the case of primitives which become the primitive default)
284      * @throws UnsupportedOperationException if the conversion cannot be completed
285      */
286     @SuppressWarnings("unchecked")
287     public <T> T convert(Object value, Class<T> type) {
288         T convert = null;
289         Object toConvert = value;
290         if (toConvert != null) {
291             Class<?> fromType = toConvert.getClass();
292             // first we check to see if we even need to do the conversion
293             if ( ConstructorUtils.classAssignable(fromType, type) ) {
294                 // this is already equivalent so no reason to convert
295                 convert = (T) value;
296             } else {
297                 // needs to be converted
298                 try {
299                     convert = convertWithConverter(toConvert, type);
300                 } catch (RuntimeException e) {
301                     throw new UnsupportedOperationException("Could not convert object ("+toConvert+") from type ("+fromType+") to type ("+type+"): " + e.getMessage(), e);
302                 }
303             }
304         } else {
305             // object is null but type requested may be primitive
306             if ( ConstructorUtils.isClassPrimitive(type) ) {
307                 // for primitives we return the default value
308                 if (ConstructorUtils.getPrimitiveDefaults().containsKey(type)) {
309                     convert = (T) ConstructorUtils.getPrimitiveDefaults().get(type);
310                 }
311             }
312         }
313         return convert;
314     }

316     /***
317      * Use the converters to convert the value to the provided type,
318      * this simply finds the converter and does the conversion,
319      * will convert interface types automatically <br/>
320      * WARNING: you should use {@link #convert(Object, Class)} unless you have a special need
321      * for this, it is primarily for reducing code complexity
322      * 
323      * @param <T>
324      * @param value any object to be converted
325      * @param type the type to convert to
326      * @return the converted object (may be null)
327      * @throws UnsupportedOperationException is the conversion could not be completed
328      */
329     protected <T> T convertWithConverter(Object value, Class<T> type) {
330         T convert = null;
331         // check for a variable converter that says it will handle the conversion first
332         VariableConverter variableConverter = getVariableConverter(value, type);
333         if (variableConverter != null) {
334             // use the variable converter
335             convert = variableConverter.convert(value, type);
336         } else {
337             // use a converter
338             Converter<T> converter = getConverterOrFail(type);
339             if (InterfaceConverter.class.isAssignableFrom(converter.getClass())) {
340                 convert = ((InterfaceConverter<T>)converter).convertInterface(value, type);
341             } else {
342                 // standard converter
343                 convert = converter.convert(value);
344             }
345         }
346         return convert;
347     }

351      * @param <T>
352      * @param type type to convert to
353      * @return the converter or die
354      * @throws UnsupportedOperationException if the converter cannot be found
355      */
356     protected <T> Converter<T> getConverterOrFail(Class<T> type) {
357         Converter<T> converter = getConverter(type);
358         if (converter == null) {
359             throw new UnsupportedOperationException("Conversion failure: No converter available to handle conversions to type ("+type+")");
360         }
361         return converter;

364     /***
365      * Get the converter for the given type if there is one
366      * @param <T>
367      * @param type the type to convert to
368      * @return the converter for this type OR null if there is not one
369      */
370     @SuppressWarnings("unchecked")
371     protected <T> Converter<T> getConverter(Class<T> type) {
372         if (type == null) {
373             throw new IllegalArgumentException("Cannot get a converter for nulls");
374         }
375         // first make sure we are using an actual wrapper class and not the primitive class (int.class)
376         Class<?> toType = ConstructorUtils.getWrapper(type);
377         Converter<T> converter = (Converter<T>) getConverters().get(toType);
378         if (converter == null) {
379             // none found so try not using the interface
380             toType = ConstructorUtils.getClassFromInterface(toType);
381             converter = (Converter<T>) getConverters().get(toType);
382             if (converter == null) {
383                 // still no converter found so try the interfaces
384                 for (Class<?> iface : ConstructorUtils.getExtendAndInterfacesForClass(toType)) {
385                     converter = (Converter<T>) getConverters().get(iface);
386                     if (converter != null) {
387                         // found a converter
388                         break;
389                     }
390                 }
391             }
392         }
393         return converter;
394     }

396     /***
397      * Get the variable converter for this value and type if there is one,
398      * returns null if no converter is available
399      * @param value the value to convert
400      * @param type the type to convert to
401      * @return the variable converter if there is one OR null if none exists
402      */
403     protected VariableConverter getVariableConverter(Object value, Class<?> type) {
404         VariableConverter converter = null;
405         Class<?> toType = ConstructorUtils.getWrapper(type);
406         for (VariableConverter variableConverter : getVariableConverters()) {
407             if (variableConverter.canConvert(value, toType)) {
408                 converter = variableConverter;
409                 break;
410             }
411         }
412         return converter;
413     }

415     @Override
416     public String toString() {
417         StringBuilder sb = new StringBuilder();
418         sb.append("converters=");
419         sb.append(getConverters().size());
420         sb.append(":");
421         for (Entry<Class<?>, Converter<?>> entry : getConverters().entrySet()) {
422             sb.append("[");
423             sb.append(entry.getKey().getName());
424             sb.append("=>");
425             sb.append(entry.getValue().getClass().getName());
426             sb.append("]");
427         }
428         sb.append(":variable=");
429         sb.append(getVariableConverters().size());
430         sb.append(":");
431         for (VariableConverter variableConverter : getVariableConverters()) {
432             sb.append("(");
433             sb.append(variableConverter.getClass().getName());
434             sb.append(")");
435         }        
436         return "Convert::c="+ConversionUtils.timesCreated+":s="+singleton+":" + sb.toString();
437     }
