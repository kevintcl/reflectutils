21  /***

23   * which a background thread invokes after the garbage collector reclaims the
24   * referent. This is a simpler alternative to using a
25   * {@link java.lang.ref.ReferenceQueue}.
26   *
27   * @author crazybob@google.com (Bob Lee)
28   */
29  public abstract class FinalizableWeakReference<T> extends WeakReference<T>
30      implements FinalizableReference {

32    protected FinalizableWeakReference(T referent) {
33      super(referent, FinalizableReferenceQueue.getInstance());
34    }
