42  /***
43   * @param <T> the class type
44   * This is used as a record (cache) of the information about a classes fields, properties, and
45   * annotations among other things (property here means a bean standard property so a public getter
46   * and setter which conform to the bean standard), it may or may not be a complete record of the
47   * class<br/> This is primarily meant to provide easy access to information gathered via reflection
48   * and puts all that information in one convenient package which can be cached (but must be cached
49   * in a ClassLoader safe way)<br/> 
50   * NOTE: It is important that you do not hold onto this object as
51   * it is meant to be cached in a way that it can be collected and will hold open the ClassLoader
52   * that the class came from<br/> 
53   * Terminology:<br/> 
54   * Complete field: the value of the field can be
55   * set (is not final or missing a setter) and retrieved (not missing a getter) <br/> 
56   * Partial field:
57   * the value can either be set or retrieved but not both <br/> 
58   * Settable: the value can be set on this field (final fields cannot be set) <br/> 
59   * Gettable: the value can be retrieved from this field <br/>
60   * <br/>
61   * This object is immutable<br/>
62   * 
63   * @author Aaron Zeckoski (azeckoski@gmail.com)
64   */
65  public class ClassFields<T> {

67      /***
68       * Indicates the fields filtering to use when retrieving fields from types/objects: <br/>
69       * {@link #COMPLETE} (default)
70       */
71      public static enum FieldsFilter {
72          /***
73           * (default) all complete (read and write) fields
74           */
75          COMPLETE, 
76          /***
77           * all readable fields (may not be writeable), includes transient fields
78           */
79          READABLE, 
80          /***
81           * all writeable fields (may not be readable)
82           */
83          WRITEABLE, 
84          /***
85           * all serializable fields (readable fields but skips transient fields)
86           */
87          SERIALIZABLE, 
88          /***
89           * all fields including ones that are partial (read or write only)
90           */
91          ALL
92      };
93  
94      /***
95       * @param cp the {@link ClassProperty} object which represents a field
96       * @param filter (optional) indicates the fields to return the names for, can be null for defaults
97       * @return true if this field is in the given filter
98       */
99      public boolean isFieldInFilter(ClassProperty cp, FieldsFilter filter) {
100         boolean in = false;
101         if (FieldsFilter.ALL.equals(filter)) {
102             in = true;
103         } else if (FieldsFilter.READABLE.equals(filter)) {
104             if ( isGettable(cp) ) {
105                 in = true;
106             }
107         } else if (FieldsFilter.WRITEABLE.equals(filter)) {
108             if ( isSettable(cp) ) {
109                 in = true;
110             }
111         } else if (FieldsFilter.SERIALIZABLE.equals(filter)) {
112             if ( isGettable(cp) ) {
113                 if (!cp.isTransient()) {
114                     in = true;
115                 }
116             }
117         } else {
118             // return complete
119             if ( isComplete(cp) ) {
120                 in = true;
121             }
122         }
123         return in;
124     }

126     /***
127      * Mode for finding the fields on classes/objects:<br/>
128      * {@link #HYBRID} (default)
129      */
130     public static enum FieldFindMode { 
131         /***
132          * (default) finds fields by matched public getters and setters first and then public fields
133          */
134         HYBRID, 
135         /***
136          * finds all fields which are accessible (public, protected, or private), ignores getters and setters
137          */
138         FIELD, 
139         /***
140          * find all matched getters and setters only, ignores fields
141          */
142         PROPERTY
143     };

166      * @param filter (optional) indicates the fields to return the names for, can be null for defaults
167      * @return the list of field names
168      */
169     public List<String> getFieldNames(FieldsFilter filter) {
170         List<String> names = new ArrayList<String>();
171         for (Entry<String, ClassProperty> entry : namesToProperties.getEntries()) {
172             String name = entry.getKey();
173             ClassProperty cp = entry.getValue();
174             if ( isFieldInFilter(cp, filter) ) {
175                 names.add(name);
176             }
177         }
178         return names;

195         OrderedMap<String, Class<?>> fieldTypes = new ArrayOrderedMap<String, Class<?>>();
196         fieldTypes.setName(getStoredClass().getName());
197         for (Entry<String, ClassProperty> entry : namesToProperties.getEntries()) {

199             ClassProperty cp = entry.getValue();
200             if ( isFieldInFilter(cp, filter) ) {
201                 fieldTypes.put(name, cp.getType());
202             }
203         }
204         return fieldTypes;
205     }

253      * Checks if a field is valid for the given filter
254      * 
255      * @param name the fieldName
256      * @param filter (optional) indicates the fields to include in the check, can be null for defaults
257      * @return true if this fieldName is valid, false otherwise
258      */
259     public boolean isFieldNameValid(String name, FieldsFilter filter) {
260         boolean valid = namesToProperties.containsKey(name);
261         if (valid) {
262             // the field is real so check to see if it is in this filter
263             ClassProperty cp = getAnyPropertyOrFail(name);
264             if ( isFieldInFilter(cp, filter) ) {
265                 valid = true;
266             } else {
267                 valid = false;
268             }
269         }
270         return valid;
271     }

321     /***
322      * Will find the first field which has this annotation type
323      * 
324      * @param annotationType
325      *            the annotation type to look for on this field
326      * @return the name of the field which has this annotation OR null if none found
327      */
328     public String getFieldNameByAnnotation(Class<? extends Annotation> annotationType) {
329         if (annotationType == null) {
330             throw new IllegalArgumentException("annotationType must not be null");
331         }
332         String fieldName = null;
333         Collection<ClassProperty> cps = namesToProperties.values();
334         for (ClassProperty classProperty : cps) {
335             if (classProperty.getAnnotation(annotationType) != null) {
336                 fieldName = classProperty.getFieldName();
337                 break;
338             }
339         }
340         return fieldName;
341     }

344      * Finds the names of all the fields with the given annotation type
345      * @param annotationType
346      *            the annotation type to look for on this field
347      * @return the list of all field names sorted OR empty if none found
348      */
349     public List<String> getFieldNamesWithAnnotation(Class<? extends Annotation> annotationType) {
350         if (annotationType == null) {
351             throw new IllegalArgumentException("annotationType must not be null");
352         }
353         HashSet<String> fieldNames = new HashSet<String>();
354         Collection<ClassProperty> cps = namesToProperties.values();
355         for (ClassProperty classProperty : cps) {
356             if (classProperty.getAnnotation(annotationType) != null) {
357                 fieldNames.add( classProperty.getFieldName() );
358             }
359         }
360         ArrayList<String> l = new ArrayList<String>(fieldNames);
361         Collections.sort(l);
362         return l;
363     }

430     /***
431      * Constructor for when you have everything already in nice arrays, the getter and setter method
432      * arrays MUST be the same length AND the same order (i.e. getter[0] and setter[0] must refer to
433      * the same property), no arguments can be null but arrays can be empty, null values in arrays
434      * will be ignored<br/> 
435      * <b>NOTE:</b> validation of the getters and setters is somewhat costly
436      * so you may want to cache the constructed object
437      * 
438      * @param fieldClass
439      *            this is the class whose fields are represented
440      * @param getterMethods
441      *            array of getter methods for properties
442      * @param setterMethods
443      *            array of setter methods for properties
444      * @param publicFields
445      *            array of public fields
446      */
447     public ClassFields(Class<T> fieldClass, Method[] getterMethods, Method[] setterMethods,
448             Field[] publicFields) {
449         if (fieldClass == null || getterMethods == null || setterMethods == null
450                 || publicFields == null) {
451             throw new IllegalArgumentException("None of the params can be null");
452         }
453         if (getterMethods.length != setterMethods.length) {
454             throw new IllegalArgumentException("Getter and setter methods must be the same length");
455         }
456         // set the field class
457         classData = new ClassData<T>(fieldClass);
458         fieldFindMode = FieldFindMode.HYBRID;
459         // set the properties
460         namesToProperties = new ArrayOrderedMap<String, ClassProperty>(getterMethods.length
461                 + publicFields.length);
462         for (int i = 0; i < getterMethods.length; i++) {
463             if (getterMethods[i] != null && setterMethods[i] != null) {
464                 String fieldName = checkPropertyMethods(getterMethods[i], setterMethods[i]);
465                 if (!namesToProperties.containsKey(fieldName)) {
466                     ClassProperty p = new ClassProperty(fieldName, getterMethods[i],
467                             setterMethods[i]);
468                     namesToProperties.put(fieldName, p);
469                 }
470             }
471         }
472         populateFields(publicFields);
473         populateAnnotationsFields();
474     }

476     /***
477      * Constructor for when you have descriptors and fields in arrays (if you have descriptors it is
478      * easy to get the fields from the class using {@link Class#getFields()}), no arguments can be
479      * null but the arrays can be empty
480      * 
481      * @param fieldClass
482      *            this is the class whose fields are represented
483      * @param descriptors
484      *            an array of descriptors, you can get these from
485      *            {@link BeanInfo#getPropertyDescriptors()} and you can get the {@link BeanInfo}
486      *            from {@link Introspector#getBeanInfo(Class)}
487      * @param publicFields
488      *            array of public fields
489      */
490     public ClassFields(Class<T> fieldClass, PropertyDescriptor[] descriptors, Field[] publicFields) {
491         if (fieldClass == null || descriptors == null || publicFields == null) {
492             throw new IllegalArgumentException("None of the params can be null");
493         }
494         classData = new ClassData<T>(fieldClass);
495         fieldFindMode = FieldFindMode.HYBRID;
496         namesToProperties = new ArrayOrderedMap<String, ClassProperty>(descriptors.length + publicFields.length);
497         populateProperties(descriptors);
498         populateFields(publicFields);
499         populateAnnotationsFields();
500     }

517      * Constructor to use when you know nothing about the class at all, this will use very

520      * <b>NOTE:</b> This is fairly expensive so it is recommended that you cache the constructed object
521      * by using {@link FieldUtils}
522      * 
523      * @param fieldClass this is the class whose fields are represented
524      * @param findMode the search mode for fields on the classes, allows the developer to control the way the 
525      * fields are found for later usage<br/>
526      * HYBRID (default): finds fields by matched public getters and setters first and then public fields <br/>
527      * FIELD: finds all fields which are accessible (public, protected, or private), ignores getters and setters <br/>
528      * PROPERTY: find all matched getters and setters only, ignores fields <br/>

534     /***
535      * Constructor to use when you know nothing about the class at all, this will use the java
536      * introspector as long as the param is true, otherwise it will simply call over to
537      * {@link ClassFields#ClassFields(Class)} and use very
538      * straightforward walking over the methods and fields on the class to extract the field
539      * information according to the field finding mode<br/> 
540      * <b>NOTE:</b> This is fairly expensive so it is recommended that you cache the constructed object
541      * by using {@link FieldUtils}
542      * 
543      * @param fieldClass this is the class whose fields are represented
544      * @param findMode the search mode for fields on the classes, allows the developer to control the way the 
545      * fields are found for later usage<br/>
546      * HYBRID (default): finds fields by matched public getters and setters first and then public fields <br/>
547      * FIELD: finds all fields which are accessible (public, protected, or private), ignores getters and setters <br/>
548      * PROPERTY: find all matched getters and setters only, ignores fields <br/>
549      * @param useIntrospector
550      *            if this is true then the {@link Introspector} is used, this is generally the
551      *            slowest way to get information about a class but some people prefer it, if false
552      *            then the internal methods are used
553      * @param includeClassFields if true then the "class" field (the result of getClass()) is included as a read only field
554      */
555     public ClassFields(Class<T> fieldClass, FieldFindMode findMode, boolean useIntrospector, boolean includeClassFields) {
556         // set the field class
557         if (fieldClass == null) {
558             throw new IllegalArgumentException("field class cannot be null");
559         }
560         classData = new ClassData<T>(fieldClass);
561         fieldFindMode = findMode;
562         includeClassField = includeClassFields;
563         namesToProperties = new ArrayOrderedMap<String, ClassProperty>();
564 
565         // check for reflect annotations on the class
566         List<Annotation> annotations = classData.getAnnotations();
567         for (Annotation annotation : annotations) {
568             if (annotation.annotationType().equals(ReflectIncludeStaticFields.class)) {
569                 includeStaticFields = true;
570             } else if (annotation.annotationType().equals(ReflectIgnoreClassFields.class)) {
571                 String[] ignore = ((ReflectIgnoreClassFields)annotation).value();
572                 for (String name : ignore) {
573                     ignoredFieldNames.add(name);
574                 }
575             }
576         }
577 
578         // populate the internal storage with the list of properties
579         if (useIntrospector) {
580             // use the java Introspector
581             BeanInfo bi;
582             try {
583                 bi = Introspector.getBeanInfo(fieldClass);
584                 PropertyDescriptor[] descriptors = bi.getPropertyDescriptors();
585                 if (descriptors != null) {
586                     populateProperties(descriptors);
587                     // fields
588                     populateFields(false);
589                 } else {
590                     useIntrospector = false;
591                 }
592             } catch (IntrospectionException e) {
593                 // no descriptors so fail over to using the internal methods
594                 useIntrospector = false;
595             }
596         }
597 
598         if (!useIntrospector) {
599             // construct using pure reflection
600             if (FieldFindMode.HYBRID.equals(findMode) || FieldFindMode.PROPERTY.equals(findMode)) {
601                 List<ClassProperty> properties = findProperties(true);
602                 for (ClassProperty property : properties) {
603                     String fieldName = property.getFieldName();
604                     namesToProperties.put(fieldName, property);
605                 }
606             }
607             if (FieldFindMode.HYBRID.equals(findMode) || FieldFindMode.FIELD.equals(findMode)) {
608                 if (FieldFindMode.FIELD.equals(findMode)) {
609                     populateFields(true);
610                 } else {
611                     populateFields(false);
612                 }
613             }
614         }
615         populateAnnotationsFields();
616     }

636     /***
637      * Determines if a {@link ClassProperty} is complete based on the {@link FieldFindMode} setting
638      * @param cp a class property setting
639      * @return true if it is complete, false otherwise
640      */
641     private boolean isComplete(ClassProperty cp) {
642         boolean complete = false;
643         if (FieldFindMode.FIELD.equals(fieldFindMode)) {
644             if (cp.isField()) {
645                 complete = true;
646             }
647         } else if (FieldFindMode.PROPERTY.equals(fieldFindMode)) {
648             if (cp.isProperty()) {
649                 complete = true;
650             }
651         } else {
652             // default
653             if (cp.isComplete()) {
654                 complete = true;
655             }
656         }
657         return complete;
658     }

660     /***
661      * Determines if a {@link ClassProperty} is gettable based on the {@link FieldFindMode} setting
662      * @param cp a class property setting
663      * @return true if it is gettable, false otherwise
664      */
665     private boolean isGettable(ClassProperty cp) {
666         boolean gettable = false;
667         if (FieldFindMode.FIELD.equals(fieldFindMode)) {
668             if (cp.isField() && cp.isGettable()) {
669                 gettable = true;
670             }
671         } else if (FieldFindMode.PROPERTY.equals(fieldFindMode)) {
672             if (cp.isProperty() && cp.isGettable()) {
673                 gettable = true;
674             }
675         } else {
676             // default
677             if (cp.isPublicGettable()) {
678                 gettable = true;
679             }
680         }
681         return gettable;
682     }

688      */
689     private boolean isSettable(ClassProperty cp) {
690         boolean settable = false;
691         if (FieldFindMode.FIELD.equals(fieldFindMode)) {
692             if (cp.isField() && cp.isSettable()) {
693                 settable = true;
694             }
695         } else if (FieldFindMode.PROPERTY.equals(fieldFindMode)) {
696             if (cp.isProperty() && cp.isSettable()) {

698             }
699         } else {

701             if (cp.isPublicSettable()) {

703             }
704         }
705         return settable;

741     /***
742      * Checks a getter and setter pair and returns the proper property name
743      * 
744      * @param getter
745      * @param setter
746      * @return the property name for this getter and setter pair
747      */
748     private String checkPropertyMethods(Method getter, Method setter) {
749         String name = null;
750         String gName = getter.getName();
751         Class<?> gType = null;
752         Class<?>[] paramTypes = getter.getParameterTypes();
753         if (paramTypes.length == 0) {
754             if (isGetClassMethod(getter)) {
755                 throw new GetClassMethodException();
756             } else if (gName.startsWith(PREFIX_GET) || gName.startsWith(PREFIX_IS)) {
757                 gType = getter.getReturnType();
758                 if (gType != null) {
759                     name = makeFieldNameFromMethod(gName);
760                 }
761             }
762         }
763         if (name != null) {
764             boolean validSetter = false;
765             String sName = setter.getName();
766             Class<?> sType = null;
767             if (sName.startsWith(PREFIX_SET)) {
768                 Class<?>[] sParamTypes = setter.getParameterTypes();
769                 if (sParamTypes.length == 1) {
770                     sType = sParamTypes[0];
771                     if (sType == null || !sType.isAssignableFrom(gType)) {
772                         throw new IllegalArgumentException("setter (" + sName + ") type (" + sType
773                                 + ") and " + "getter (" + gName + ") type (" + gType
774                                 + ") do not match, they MUST match");
775                     } else {
776                         validSetter = true;
777                     }
778                 }
779             }
780             if (!validSetter) {
781                 throw new IllegalArgumentException("setter (" + sName
782                         + ") does not appear to be a valid setter method "
783                         + "(single param, name starts with set)");
784             }
785         } else {
786             throw new IllegalArgumentException("getter (" + gName
787                     + ") does not appear to be a valid getter method "
788                     + "(0 params, name starts with get or is)");
789         }
790         return name;
791     }

793     /***
794      * Simple manual method for walking over a class and finding the bean properties, generally used
795      * because the sun Introspector is so slow, note that this will also return all the non-matching
796      * getters or setters if requested
797      * 
798      * @param fieldClass
799      *            the class to reflect over
800      * @param includePartial
801      *            if true then includes getters and setters that do not match, if false then ONLY
802      *            include the getters and setters which match exactly
803      * @return a list Property objects
804      */
805     private List<ClassProperty> findProperties(boolean includePartial) {
806         // find the property methods from all public methods
807         Map<String, ClassProperty> propertyMap = new ArrayOrderedMap<String, ClassProperty>();
808         for (Method method : classData.getPublicMethods()) {
809             Class<?>[] paramTypes = method.getParameterTypes();
810             String name = method.getName();
811             if (! includeStaticFields && isStatic(method)) {
812                 continue; // skip statics
813             } else if (isGetClassMethod(method) && ! includeClassField) {
814                 continue; // skip class field
815             } else if (paramTypes.length == 0) {
816                 // no params
817                 if (name.startsWith(PREFIX_GET) || name.startsWith(PREFIX_IS)) {
818                     // getter
819                     Class<?> returnType = method.getReturnType();
820                     if (returnType != null) {
821                         try {
822                             String fieldName = makeFieldNameFromMethod(method.getName());
823                             ClassProperty p = null;
824                             if (propertyMap.containsKey(fieldName)) {
825                                 p = propertyMap.get(fieldName);
826                                 p.setGetter(method);
827                             } else {
828                                 p = new ClassProperty(fieldName, method, null);
829                                 propertyMap.put(fieldName, p);
830                             }
831                         } catch (Exception e) {
832                             // nothing to do here but move on
833                         }
834                     }
835                 }
836             } else if (paramTypes.length == 1) {
837                 if (name.startsWith(PREFIX_SET)) {
838                     // setter (one param)
839                     String fieldName = makeFieldNameFromMethod(method.getName());
840                     ClassProperty p = null;
841                     if (propertyMap.containsKey(fieldName)) {
842                         p = propertyMap.get(fieldName);
843                         p.setSetter(method);
844                     } else {
845                         p = new ClassProperty(fieldName, null, method);
846                         propertyMap.put(fieldName, p);
847                     }
848                 }
849             }
850         }
851         List<ClassProperty> properties = new ArrayList<ClassProperty>();
852         for (Entry<String, ClassProperty> entry : propertyMap.entrySet()) {
853             ClassProperty p = entry.getValue();
854             if (p != null) {
855                 if (! includeStaticFields && p.isStatic()) {
856                     continue; // skip any props with static fields
857                 }
858                 if (ignoredFieldNames.contains(p.getFieldName())) {
859                     continue; // skip ignored names
860                 }
861                 if (includePartial) {
862                     // put all of them in
863                     properties.add(p);
864                 } else {
865                     // filter down to only the one which are complete
866                     if (p.isGettable() && p.isSettable()) {
867                         properties.add(p);
868                     }
869                 }
870             }
871         }
872         return properties;
873     }

930     /***
931      * We need to find all the class, method, and field annotations and not just the ones on public
932      * methods so we are going through everything on this class and all the super classes<br/>
933      * WARNING: this is NOT cheap and the results need to be cached badly<br/>
934      * NOTE: This also populates all the fields in every property and checks and removes
935      * all staics or excluded fields if any slipped through the previous processes
936      * 
937      * @param fieldClass
938      *            the class to analyze
939      */
940     private void populateAnnotationsFields() {
941         // get the annotations from all class methods
942         for (Method method : classData.getMethods()) {
943             try {
944                 String name = makeFieldNameFromMethod(method.getName());
945                 ClassProperty cp = getAnyPropertyOrFail(name);
946                 Annotation[] annotations = method.getAnnotations();
947                 for (Annotation annotation : annotations) {
948                     if (annotation != null) cp.addAnnotation(annotation);
949                 }
950             } catch (FieldnameNotFoundException e) {
951                 // nothing to do but keep going
952             }
953         }
954         // get the annotations from all class fields
955         for (Field field : classData.getFields()) {
956             try {
957                 String fieldName = field.getName();
958                 if (! includeStaticFields && isStatic(field)) {
959                     namesToProperties.remove(fieldName); // as a final check, take this out if it is in there
960                     continue; // skip statics
961                 }
962                 if (ignoredFieldNames.contains(fieldName)) {
963                     namesToProperties.remove(fieldName); // as a final check, take this out if it is in there
964                     continue; // skip ignored names
965                 }
966                 ClassProperty cp = getAnyPropertyOrFail(fieldName);
967                 if (cp.getField() == null) {
968                     cp.setField(field); // ensure there is a field set for all methods
969                 }
970                 Annotation[] annotations = field.getAnnotations();
971                 for (Annotation annotation : annotations) {
972                     if (annotation != null) cp.addAnnotation(annotation);
973                 }
974             } catch (FieldnameNotFoundException e) {
975                 // nothing to do but keep going
976             }
977         }

992     public static String makeFieldNameFromMethod(String methodName) {
