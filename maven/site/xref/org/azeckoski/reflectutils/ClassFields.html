43  /***
44   * @param <T> the class type
45   * This is used as a record (cache) of the information about a classes fields, properties, and
46   * annotations among other things (property here means a bean standard property so a public getter
47   * and setter which conform to the bean standard), it may or may not be a complete record of the
48   * class<br/> This is primarily meant to provide easy access to information gathered via reflection
49   * and puts all that information in one convenient package which can be cached (but must be cached
50   * in a ClassLoader safe way)<br/> 
51   * NOTE: It is important that you do not hold onto this object as
52   * it is meant to be cached in a way that it can be collected and will hold open the ClassLoader
53   * that the class came from<br/> 
54   * Terminology:<br/> 
55   * Complete field: the value of the field can be
56   * set (is not final or missing a setter) and retrieved (not missing a getter) <br/> 
57   * Partial field:
58   * the value can either be set or retrieved but not both <br/> 
59   * Settable: the value can be set on this field (final fields cannot be set) <br/> 
60   * Gettable: the value can be retrieved from this field <br/>
61   * <br/>
62   * This object is immutable<br/>
63   * 
64   * @author Aaron Zeckoski (azeckoski@gmail.com)
65   */
66  public class ClassFields<T> {

68      /***
69       * Indicates the fields filtering to use when retrieving fields from types/objects: <br/>
70       * {@link #COMPLETE} (default)
71       */
72      public static enum FieldsFilter {
73          /***
74           * (default) all complete (read and write) fields
75           */
76          COMPLETE, 
77          /***
78           * all readable fields (may not be writeable), includes transient fields
79           */
80          READABLE, 
81          /***
82           * all writeable fields (may not be readable)
83           */
84          WRITEABLE, 
85          /***
86           * all serializable fields (readable fields but skips transient fields)
87           */
88          SERIALIZABLE, 
89          /***
90           * all fields including ones that are partial (read or write only)
91           */
92          ALL
93      };
94  
95      /***
96       * @param cp the {@link ClassProperty} object which represents a field
97       * @param filter (optional) indicates the fields to return the names for, can be null for defaults
98       * @return true if this field is in the given filter
99       */
100     public boolean isFieldInFilter(ClassProperty cp, FieldsFilter filter) {
101         boolean in = false;
102         if (FieldsFilter.ALL.equals(filter)) {
103             in = true;
104         } else if (FieldsFilter.READABLE.equals(filter)) {
105             if ( isGettable(cp) ) {
106                 in = true;
107             }
108         } else if (FieldsFilter.WRITEABLE.equals(filter)) {
109             if ( isSettable(cp) ) {
110                 in = true;
111             }
112         } else if (FieldsFilter.SERIALIZABLE.equals(filter)) {
113             if ( isGettable(cp) ) {
114                 if (!cp.isTransient()) {
115                     in = true;
116                 }
117             }
118         } else {
119             // return complete
120             if ( isComplete(cp) ) {
121                 in = true;
122             }
123         }
124         return in;
125     }

127     /***
128      * Mode for finding the fields on classes/objects:<br/>
129      * {@link #HYBRID} (default)
130      */
131     public static enum FieldFindMode { 
132         /***
133          * (default) finds fields by matched public getters and setters first and then public fields
134          */
135         HYBRID, 
136         /***
137          * finds all fields which are accessible (public, protected, or private), ignores getters and setters
138          */
139         FIELD, 
140         /***
141          * find all matched getters and setters only, ignores fields
142          */
143         PROPERTY
144     };

165     /***
166      * Get the field names but filter the fields to return
167      * @param filter (optional) indicates the fields to return the names for, can be null for defaults
168      * @return the list of field names
169      */
170     public List<String> getFieldNames(FieldsFilter filter) {
171         List<String> names = new ArrayList<String>();
172         for (Entry<String, ClassProperty> entry : namesToProperties.getEntries()) {
173             String name = entry.getKey();
174             ClassProperty cp = entry.getValue();
175             if ( isFieldInFilter(cp, filter) ) {
176                 names.add(name);
177             }
178         }
179         return names;
180     }

190     /***
191      * Get the types for fields in a class but filter the fields to get the types for
192      * @param filter (optional) indicates the fields to return the types for, can be null for defaults
193      * @return the map of fieldName -> field type
194      */
195     public Map<String, Class<?>> getFieldTypes(FieldsFilter filter) {
196         OrderedMap<String, Class<?>> fieldTypes = new ArrayOrderedMap<String, Class<?>>();
197         fieldTypes.setName(getStoredClass().getName());
198         for (Entry<String, ClassProperty> entry : namesToProperties.getEntries()) {
199             String name = entry.getKey();
200             ClassProperty cp = entry.getValue();
201             if ( isFieldInFilter(cp, filter) ) {
202                 fieldTypes.put(name, cp.getType());
203             }
204         }
205         return fieldTypes;
206     }

253     /***
254      * Checks if a field is valid for the given filter
255      * 
256      * @param name the fieldName
257      * @param filter (optional) indicates the fields to include in the check, can be null for defaults
258      * @return true if this fieldName is valid, false otherwise
259      */
260     public boolean isFieldNameValid(String name, FieldsFilter filter) {
261         boolean valid = namesToProperties.containsKey(name);
262         if (valid) {
263             // the field is real so check to see if it is in this filter
264             ClassProperty cp = getAnyPropertyOrFail(name);
265             if ( isFieldInFilter(cp, filter) ) {
266                 valid = true;
267             } else {
268                 valid = false;
269             }
270         }
271         return valid;
272     }

322     /***
323      * Will find the first field which has this annotation type
324      * 
325      * @param annotationType
326      *            the annotation type to look for on this field
327      * @return the name of the field which has this annotation OR null if none found
328      */
329     public String getFieldNameByAnnotation(Class<? extends Annotation> annotationType) {
330         if (annotationType == null) {
331             throw new IllegalArgumentException("annotationType must not be null");
332         }
333         String fieldName = null;
334         Collection<ClassProperty> cps = namesToProperties.values();
335         for (ClassProperty classProperty : cps) {
336             if (classProperty.getAnnotation(annotationType) != null) {
337                 fieldName = classProperty.getFieldName();
338                 break;
339             }
340         }
341         return fieldName;
342     }

344     /***
345      * Finds the names of all the fields with the given annotation type
346      * @param annotationType
347      *            the annotation type to look for on this field
348      * @return the list of all field names sorted OR empty if none found
349      */
350     public List<String> getFieldNamesWithAnnotation(Class<? extends Annotation> annotationType) {
351         if (annotationType == null) {
352             throw new IllegalArgumentException("annotationType must not be null");
353         }
354         HashSet<String> fieldNames = new HashSet<String>();
355         Collection<ClassProperty> cps = namesToProperties.values();
356         for (ClassProperty classProperty : cps) {
357             if (classProperty.getAnnotation(annotationType) != null) {
358                 fieldNames.add( classProperty.getFieldName() );
359             }
360         }
361         ArrayList<String> l = new ArrayList<String>(fieldNames);
362         Collections.sort(l);
363         return l;
364     }

435     /***
436      * Constructor for when you have everything already in nice arrays, the getter and setter method
437      * arrays MUST be the same length AND the same order (i.e. getter[0] and setter[0] must refer to
438      * the same property), no arguments can be null but arrays can be empty, null values in arrays
439      * will be ignored<br/> 
440      * <b>NOTE:</b> validation of the getters and setters is somewhat costly
441      * so you may want to cache the constructed object
442      * 
443      * @param fieldClass
444      *            this is the class whose fields are represented
445      * @param getterMethods
446      *            array of getter methods for properties
447      * @param setterMethods
448      *            array of setter methods for properties
449      * @param publicFields
450      *            array of public fields
451      */
452     public ClassFields(Class<T> fieldClass, Method[] getterMethods, Method[] setterMethods,
453             Field[] publicFields) {
454         if (fieldClass == null || getterMethods == null || setterMethods == null
455                 || publicFields == null) {
456             throw new IllegalArgumentException("None of the params can be null");
457         }
458         if (getterMethods.length != setterMethods.length) {
459             throw new IllegalArgumentException("Getter and setter methods must be the same length");
460         }
461         // set the field class
462         classData = new ClassData<T>(fieldClass);
463         fieldFindMode = FieldFindMode.HYBRID;
464         // set the properties
465         namesToProperties = new ArrayOrderedMap<String, ClassProperty>(getterMethods.length
466                 + publicFields.length);
467         for (int i = 0; i < getterMethods.length; i++) {
468             if (getterMethods[i] != null && setterMethods[i] != null) {
469                 String fieldName = checkPropertyMethods(getterMethods[i], setterMethods[i]);
470                 if (!namesToProperties.containsKey(fieldName)) {
471                     ClassProperty p = new ClassProperty(fieldName, getterMethods[i],
472                             setterMethods[i]);
473                     namesToProperties.put(fieldName, p);
474                 }
475             }
476         }
477         populateFields(publicFields);
478         populateAnnotationsFields();
479     }

483      * easy to get the fields from the class using {@link Class#getFields()}), no arguments can be

485      * 
486      * @param fieldClass
487      *            this is the class whose fields are represented
488      * @param descriptors
489      *            an array of descriptors, you can get these from
490      *            {@link BeanInfo#getPropertyDescriptors()} and you can get the {@link BeanInfo}

504         populateAnnotationsFields();
505     }

529      * @param findMode the search mode for fields on the classes, allows the developer to control the way the 
530      * fields are found for later usage<br/>
531      * HYBRID (default): finds fields by matched public getters and setters first and then public fields <br/>

539     /***
540      * Constructor to use when you know nothing about the class at all, this will use the java
541      * introspector as long as the param is true, otherwise it will simply call over to
542      * {@link ClassFields#ClassFields(Class)} and use very
543      * straightforward walking over the methods and fields on the class to extract the field
544      * information according to the field finding mode<br/> 
545      * <b>NOTE:</b> This is fairly expensive so it is recommended that you cache the constructed object
546      * by using {@link FieldUtils}
547      * 
548      * @param fieldClass this is the class whose fields are represented
549      * @param findMode the search mode for fields on the classes, allows the developer to control the way the 
550      * fields are found for later usage<br/>
551      * HYBRID (default): finds fields by matched public getters and setters first and then public fields <br/>
552      * FIELD: finds all fields which are accessible (public, protected, or private), ignores getters and setters <br/>
553      * PROPERTY: find all matched getters and setters only, ignores fields <br/>
554      * @param useIntrospector
555      *            if this is true then the {@link Introspector} is used, this is generally the
556      *            slowest way to get information about a class but some people prefer it, if false
557      *            then the internal methods are used
558      * @param includeClassFields if true then the "class" field (the result of getClass()) is included as a read only field
559      */
560     public ClassFields(Class<T> fieldClass, FieldFindMode findMode, boolean useIntrospector, boolean includeClassFields) {
561         // set the field class
562         if (fieldClass == null) {
563             throw new IllegalArgumentException("field class cannot be null");
564         }
565         classData = new ClassData<T>(fieldClass);
566         fieldFindMode = findMode;
567         includeClassField = includeClassFields;
568         namesToProperties = new ArrayOrderedMap<String, ClassProperty>();
569 
570         // check for reflect annotations on the class
571         List<Annotation> annotations = classData.getAnnotations();
572         for (Annotation annotation : annotations) {
573             // note that we compare the simple name to avoid issues with cross classloader types
574             if (ReflectIncludeStaticFields.class.getSimpleName().equals(annotation.annotationType().getSimpleName())) {
575                 includeStaticFields = true;
576             } else if (ReflectIgnoreClassFields.class.getSimpleName().equals(annotation.annotationType().getSimpleName())) {
577                 // this does not work if the classloader of the annotation is not our classloader
578                 String[] ignore = new String[0];
579                 if (ReflectIgnoreClassFields.class.equals(annotation.annotationType())) {
580                     ignore = ((ReflectIgnoreClassFields)annotation).value();
581                 } else {
582                     // get the value out by reflection on the annotation
583                     ClassData<?> cd = ClassDataCacher.getInstance().getClassData(annotation);
584                     List<Method> methods = cd.getMethods();
585                     for (Method method : methods) {
586                         if (method.getName().equals("value")) {
587                             try {
588                                 Object o = method.invoke(annotation, new Object[] {});
589                                 ignore = (String[]) o;
590                             } catch (Exception e) {
591                                 throw new RuntimeException("Annotation of the same name has invalid value() method ("+method+") and is not of the right type: " + ReflectIgnoreClassFields.class);
592                             }
593                             break;
594                         }
595                     }
596                 }
597                 for (String name : ignore) {
598                     ignoredFieldNames.add(name);
599                 }
600             } else if (ReflectTransientClassFields.class.getSimpleName().equals(annotation.annotationType().getSimpleName())) {
601                 // this does not work if the classloader of the annotation is not our classloader
602                 String[] transients = new String[0];
603                 if (ReflectTransientClassFields.class.equals(annotation.annotationType())) {
604                     transients = ((ReflectTransientClassFields)annotation).value();
605                 } else {
606                     // get the value out by reflection on the annotation
607                     ClassData<?> cd = ClassDataCacher.getInstance().getClassData(annotation);
608                     List<Method> methods = cd.getMethods();
609                     for (Method method : methods) {
610                         if (method.getName().equals("value")) {
611                             try {
612                                 Object o = method.invoke(annotation, new Object[] {});
613                                 transients = (String[]) o;
614                             } catch (Exception e) {
615                                 throw new RuntimeException("Annotation of the same name has invalid value() method ("+method+") and is not of the right type: " + ReflectTransientClassFields.class);
616                             }
617                             break;
618                         }
619                     }
620                 }
621                 for (String name : transients) {
622                     transientFieldNames.add(name);
623                 }
624             }
625         }
626 
627         // populate the internal storage with the list of properties
628         if (useIntrospector) {
629             // use the java Introspector
630             BeanInfo bi;
631             try {
632                 bi = Introspector.getBeanInfo(fieldClass);
633                 PropertyDescriptor[] descriptors = bi.getPropertyDescriptors();
634                 if (descriptors != null) {
635                     populateProperties(descriptors);
636                     // fields
637                     populateFields(false);
638                 } else {
639                     useIntrospector = false;
640                 }
641             } catch (IntrospectionException e) {
642                 // no descriptors so fail over to using the internal methods
643                 useIntrospector = false;
644             }
645         }
646 
647         if (!useIntrospector) {
648             // construct using pure reflection
649             if (FieldFindMode.HYBRID.equals(findMode) || FieldFindMode.PROPERTY.equals(findMode)) {
650                 List<ClassProperty> properties = findProperties(true);
651                 for (ClassProperty property : properties) {
652                     String fieldName = property.getFieldName();
653                     namesToProperties.put(fieldName, property);
654                 }
655             }
656             if (FieldFindMode.HYBRID.equals(findMode) || FieldFindMode.FIELD.equals(findMode)) {
657                 if (FieldFindMode.FIELD.equals(findMode)) {
658                     populateFields(true);
659                 } else {
660                     populateFields(false);
661                 }
662             }
663         }
664         populateAnnotationsFields();

790     /***
791      * Checks a getter and setter pair and returns the proper property name
792      * 
793      * @param getter
794      * @param setter
795      * @return the property name for this getter and setter pair
796      */
797     private String checkPropertyMethods(Method getter, Method setter) {
798         String name = null;
799         String gName = getter.getName();
800         Class<?> gType = null;
801         Class<?>[] paramTypes = getter.getParameterTypes();
802         if (paramTypes.length == 0) {
803             if (isGetClassMethod(getter)) {
804                 throw new GetClassMethodException();
805             } else if (gName.startsWith(PREFIX_GET) || gName.startsWith(PREFIX_IS)) {
806                 gType = getter.getReturnType();
807                 if (gType != null) {
808                     name = makeFieldNameFromMethod(gName);
809                 }
810             }
811         }
812         if (name != null) {
813             boolean validSetter = false;
814             String sName = setter.getName();
815             Class<?> sType = null;
816             if (sName.startsWith(PREFIX_SET)) {
817                 Class<?>[] sParamTypes = setter.getParameterTypes();
818                 if (sParamTypes.length == 1) {
819                     sType = sParamTypes[0];
820                     if (sType == null || !sType.isAssignableFrom(gType)) {
821                         throw new IllegalArgumentException("setter (" + sName + ") type (" + sType
822                                 + ") and " + "getter (" + gName + ") type (" + gType
823                                 + ") do not match, they MUST match");
824                     } else {
825                         validSetter = true;
826                     }
827                 }
828             }
829             if (!validSetter) {
830                 throw new IllegalArgumentException("setter (" + sName
831                         + ") does not appear to be a valid setter method "
832                         + "(single param, name starts with set)");
833             }
834         } else {
835             throw new IllegalArgumentException("getter (" + gName
836                     + ") does not appear to be a valid getter method "
837                     + "(0 params, name starts with get or is)");
838         }
839         return name;
840     }

842     /***
843      * Simple manual method for walking over a class and finding the bean properties, generally used
844      * because the sun Introspector is so slow, note that this will also return all the non-matching
845      * getters or setters if requested
846      * 
847      * @param fieldClass
848      *            the class to reflect over
849      * @param includePartial
850      *            if true then includes getters and setters that do not match, if false then ONLY
851      *            include the getters and setters which match exactly
852      * @return a list Property objects
853      */
854     private List<ClassProperty> findProperties(boolean includePartial) {
855         // find the property methods from all public methods
856         Map<String, ClassProperty> propertyMap = new ArrayOrderedMap<String, ClassProperty>();
857         for (Method method : classData.getPublicMethods()) {
858             Class<?>[] paramTypes = method.getParameterTypes();
859             String name = method.getName();
860             if (! includeStaticFields && isStatic(method)) {
861                 continue; // skip statics
862             } else if (isGetClassMethod(method) && ! includeClassField) {
863                 continue; // skip class field
864             } else if (paramTypes.length == 0) {
865                 // no params
866                 if (name.startsWith(PREFIX_GET) || name.startsWith(PREFIX_IS)) {
867                     // getter
868                     Class<?> returnType = method.getReturnType();
869                     if (returnType != null) {
870                         try {
871                             String fieldName = makeFieldNameFromMethod(method.getName());
872                             ClassProperty p = null;
873                             if (propertyMap.containsKey(fieldName)) {
874                                 p = propertyMap.get(fieldName);
875                                 p.setGetter(method);
876                             } else {
877                                 p = new ClassProperty(fieldName, method, null);
878                                 propertyMap.put(fieldName, p);
879                             }
880                         } catch (Exception e) {
881                             // nothing to do here but move on
882                         }
883                     }
884                 }
885             } else if (paramTypes.length == 1) {
886                 if (name.startsWith(PREFIX_SET)) {
887                     // setter (one param)
888                     String fieldName = makeFieldNameFromMethod(method.getName());
889                     ClassProperty p = null;
890                     if (propertyMap.containsKey(fieldName)) {
891                         p = propertyMap.get(fieldName);
892                         p.setSetter(method);
893                     } else {
894                         p = new ClassProperty(fieldName, null, method);
895                         propertyMap.put(fieldName, p);
896                     }
897                 }
898             }
899         }
900         List<ClassProperty> properties = new ArrayList<ClassProperty>();
901         for (Entry<String, ClassProperty> entry : propertyMap.entrySet()) {
902             ClassProperty p = entry.getValue();
903             if (p != null) {
904                 if (! includeStaticFields && p.isStatic()) {
905                     continue; // skip any props with static fields
906                 }
907                 if (ignoredFieldNames.contains(p.getFieldName())) {
908                     continue; // skip ignored names
909                 }
910                 if (includePartial) {
911                     // put all of them in
912                     properties.add(p);
913                 } else {
914                     // filter down to only the one which are complete
915                     if (p.isGettable() && p.isSettable()) {
916                         properties.add(p);
917                     }
918                 }
919             }
920         }
921         return properties;
922     }

979     /***
980      * We need to find all the class, method, and field annotations and not just the ones on public
981      * methods so we are going through everything on this class and all the super classes<br/>
982      * WARNING: this is NOT cheap and the results need to be cached badly<br/>
983      * NOTE: This also populates all the fields in every property and checks and removes
984      * all staics or excluded fields if any slipped through the previous processes
985      * 
986      * @param fieldClass
987      *            the class to analyze
988      */
989     private void populateAnnotationsFields() {
990         // get the annotations from all class methods
991         for (Method method : classData.getMethods()) {
992             int paramCount = method.getParameterTypes().length;
993             // only include annotations on methods which *might* be fields (no args, 1 arg, or 2 args for the special ones)
994             if (paramCount <= 2) {
995                 try {
996                     String name = makeFieldNameFromMethod(method.getName());
997                     ClassProperty cp = getAnyPropertyOrFail(name);
998                     Annotation[] annotations = method.getAnnotations();
999                     for (Annotation annotation : annotations) {
1000                         if (annotation != null) cp.addAnnotation(annotation);
1001                     }
1002                 } catch (FieldnameNotFoundException e) {
1003                     // nothing to do but keep going
1004                 }
1005             }
1006         }
1007         // get the annotations from all class fields
1008         for (Field field : classData.getFields()) {
1009             try {
1010                 String fieldName = field.getName();
1011                 if (! includeStaticFields && isStatic(field)) {
1012                     namesToProperties.remove(fieldName); // as a final check, take this out if it is in there
1013                     continue; // skip statics
1014                 }
1015                 if (ignoredFieldNames.contains(fieldName)) {
1016                     namesToProperties.remove(fieldName); // as a final check, take this out if it is in there
1017                     continue; // skip ignored names
1018                 }
1019                 ClassProperty cp = getAnyPropertyOrFail(fieldName);
1020                 if (cp.getField() == null) {
1021                     cp.setField(field); // ensure there is a field set for all methods
1022                 }
1023                 Annotation[] annotations = field.getAnnotations();
1024                 for (Annotation annotation : annotations) {
1025                     if (annotation != null) cp.addAnnotation(annotation);
1026                 }
1027             } catch (FieldnameNotFoundException e) {
1028                 // nothing to do but keep going
1029             }
1030         }
1031         // now mark the transient fields as such
1032         for (String fieldName : transientFieldNames) {
1033             try {
1034                 ClassProperty cp = getAnyPropertyOrFail(fieldName);
1035                 cp.transientField = true;
1036             } catch (FieldnameNotFoundException e) {
1037                 // nothing to do but keep going
1038             }
1039         }
1040     }
