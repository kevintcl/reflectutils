17  import java.lang.annotation.Documented;
18  import java.lang.annotation.ElementType;
19  import java.lang.annotation.Retention;
20  import java.lang.annotation.RetentionPolicy;
21  import java.lang.annotation.Target;
22

24  /***
25   * Indicate fields on a class which should be ignored when performing class field operations <br/>
26   * This is primarily for indicating that reflection operations should not affect fields in classes
27   * which you are extending or constant fields in interfaces <br/>
28   * For example: <br/>
29   * Given a class with 3 String fields: name, email, phone <br/>
30   * Adding the annotation like so to the class will cause the phone field to not be counted for field reflection operations:
31   * <pre>
32   *    &#064;ReflectIgnoreClassFields("phone")
33   *    public class Person {
34   *        String name;
35   *        String email;
36   *        String phone;
37   *        ... 
38   *    }
39   * </pre>
40   * And this annotation would cause the email and phone fields to be ignored:
41   * <pre>
42   *    &#064;ReflectIgnoreClassFields({"phone","email"})
43   *    public class Person {
44   *        String name;
45   *        String email;
46   *        String phone;
47   *        ... 
48   *    }
49   * </pre>
50   * 
51   * @author Aaron Zeckoski (azeckoski @ gmail.com)
52   */
53  @Documented
54  @Retention(RetentionPolicy.RUNTIME)
55  @Target(ElementType.TYPE)
56  public @interface ReflectIgnoreClassFields {
57      String[] value();
58  }
