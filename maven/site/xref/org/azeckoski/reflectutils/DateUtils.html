15  package org.azeckoski.reflectutils;
16  
17  import java.text.SimpleDateFormat;
18  import java.util.Date;
19  import java.util.Locale;
20  
21  /***
22   * Date conversion utilities, mostly for outputing dates in standard formats
23   * 
24   * @author Aaron Zeckoski (azeckoski @ gmail.com)
25   */
26  public class DateUtils {
27  
28      public static String makeDateRFC2822(Date date) {
29          SimpleDateFormat df = new SimpleDateFormat("EEE', 'dd' 'MMM' 'yyyy' 'HH:mm:ss' 'Z", Locale.US);
30          return df.format(date);
31      }
32  
33      public static String makeDateISO8601(Date date) {
34          SimpleDateFormat df = new SimpleDateFormat("yyyy-MM-dd'T'HH:mm:ssZ");
35          String result = df.format(date);
36          // convert YYYYMMDDTHH:mm:ss+HH00 into YYYYMMDDTHH:mm:ss+HH:00
37          result = result.substring(0, result.length() - 2) + ":" + result.substring(result.length() - 2);
38          return result;
39      }
40  }
