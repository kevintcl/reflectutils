30      /***
31       * Resize an array and return the new one,
32       * this will truncate an array or expand it depending on the size given
33       * 
34       * @param <T>
35       * @param array any array
36       * @param newSize the new size for the array
37       * @return the resized array
38       */
39      @SuppressWarnings("unchecked")
40      public static <T> T[] resize(T[] array, int newSize) {
41          Class<?> type = array.getClass().getComponentType();
42          T[] newArray = (T[]) Array.newInstance(type, newSize);
43          int toCopy = Math.min(array.length, newArray.length);
44          System.arraycopy( array, 0, newArray, 0, toCopy );
45          return newArray;
46      }

106     /***
107      * Checks to see if an array contains a value,
108      * will return false if a null value is supplied
109      * 
110      * @param <T>
111      * @param array any array of objects
112      * @param value the value to check for
113      * @return true if the value is found, false otherwise
114      */
115     public static <T> boolean contains(T[] array, T value) {
116         boolean foundValue = false;
117         if (value != null) {
118             for (int i = 0; i < array.length; i++) {
119                 if (value.equals(array[i])) {
120                     foundValue = true;
121                     break;
122                 }
123             }
124         }
125         return foundValue;
126     }

250     /***
251      * Take an array of anything and turn it into a string
252      * 
253      * @param array any array
254      * @return a string representing that array
255      */
256     public static String arrayToString(Object[] array) {
257         StringBuilder result = new StringBuilder();
258         if (array != null && array.length > 0) {
259             for (int i = 0; i < array.length; i++) {
260                 if (i > 0) {
261                     result.append(",");
262                 }
263                 if (array[i] != null) {
264                     result.append(array[i].toString());
265                 }
266             }
267         }
268         return result.toString();
269     }

285     /***
286      * Remove all duplicate objects from a list
287      * 
288      * @param <T>
289      * @param list any list
290      * @return the original list with the duplicate objects removed
291      */
292     public static <T> List<T> removeDuplicates(List<T> list) {
293         Set<T> s = new HashSet<T>();
294         for (Iterator<T> iter = list.iterator(); iter.hasNext();) {
295             T element = (T) iter.next();
296             if (! s.add(element)) {
297                 iter.remove();
298             }
299         }
300         return list;
301     }

303     /***
304      * Create a set from any array
305      * 
306      * @param array any array (including null or empty)
307      * @return the set with the values from the array
308      */
309     public static <T> Set<T> makeSetFromArray(T[] array) {
310         Set<T> set = new HashSet<T>();
311         if (array != null) {
312             for (int i = 0; i < array.length; i++) {
313                 if (array[i] != null) {
314                     set.add(array[i]);
315                 }
316             }
317         }
