37  /***
38   * Class which provides methods for dealing with the fields in objects and classes,
39   * this provides the core functionality for the reflect util class<br/>
40   * <br/>
41   * Setting and getting fields supports simple, nested, indexed, and mapped values:<br/>
42   * <b>Simple:</b> Get/set a field in a bean (or map), Example: "title", "id"<br/>
43   * <b>Nested:</b> Get/set a field in a bean which is contained in another bean, Example: "someBean.title", "someBean.id"<br/>
44   * <b>Indexed:</b> Get/set a list/array item by index in a bean, Example: "myList[1]", "anArray[2]"<br/>
45   * <b>Mapped:</b> Get/set a map entry by key in a bean, Example: "myMap(key)", "someMap(thing)"<br/>
46   * 
47   * @author Aaron Zeckoski (azeckoski @ gmail.com)
48   */

106     /***
107      * loads up the field adapter if the right classes exist,
108      * this allows us to support DynaBeans even though they may not be available 
109      */
110     protected void loadFieldAdapter() {
111         // load up the field adapter if possible
112         Class<?> dynaBean = ClassLoaderUtils.getClassFromString(DYNABEAN_CLASSNAME);
113         if (dynaBean != null) {
114             // assumes the adapter impl is in the same path as the interface
115             String path = FieldAdapter.class.getName();
116             path = path.replace(FieldAdapter.class.getSimpleName(), DYNABEAN_ADAPTER);
117             Class<?> adapterClass = ClassLoaderUtils.getClassFromString(path);
118             if (adapterClass == null) {
119                 throw new IllegalStateException("Could not find adapter class: " + DYNABEAN_ADAPTER);
120             }
121             try {
122                 fieldAdapter = (FieldAdapter) adapterClass.newInstance();
123             } catch (Exception e) {
124                 throw new RuntimeException("Failed to instantiate field adapter: " + adapterClass, e);
125             }
126         } else {
127             getFieldAdapter(); // loads the default
128         }
129     }

159      * @throws IllegalArgumentException if obj is null

161     @SuppressWarnings("unchecked")
162     public <T> ClassFields<T> analyzeObject(Object obj) {
163         if (obj == null) {
164             throw new IllegalArgumentException("obj cannot be null");
165         }
166         if (Class.class.equals(obj)) {
167             // this is a class so we should pass it over to the other method
168             return analyzeClass((Class)obj);
169         }
170         Class<T> cls = (Class<T>) obj.getClass();
171         return analyzeClass(cls);

174     /***
175      * Finds the type for a field based on the given class and the field name
176      * @param type any class type
177      * @param name the name of a field in this class (can be nested, indexed, mapped, etc.)
178      * @return the type of the field (will be {@link Object} if the type is indeterminate)
179      */
180     public Class<?> getFieldType(Class<?> type, String name) {
181         if (type == null || name == null) {
182             throw new IllegalArgumentException("type and name must not be null");
183         }
184         // get the nested class or return Object.class as a cop out
185         while (getResolver().hasNested(name)) {
186             String next = getResolver().next(name);
187             Class<?> nestedClass = null;
188             if (Object.class.equals(type)
189                     || Map.class.isAssignableFrom(type)
190                     || getResolver().isMapped(next)
191                     || getResolver().isIndexed(next) ) {
192                 // these can contain objects or it is Object so we bail out
193                 return Object.class;
194             } else {
195                 // a real class, hooray, analyze it
196                 ClassFields<?> cf = analyzeClass(type);
197                 nestedClass = cf.getFieldType(name);
198             }
199             type = nestedClass;
200             name = getResolver().remove(name);
201         }
202         String targetName = getResolver().getProperty(name); // simple name of target field
203         // get the type
204         Class<?> fieldType = null;
205         if ( ConstructorUtils.isClassObjectHolder(type) 
206                 || Object.class.equals(type) ) {
207             // special handling for the holder types, needed because attempting to analyze a map or other container will cause a failure
208             fieldType = Object.class;
209         } else {
210             // normal object
211             ClassFields<?> cf = analyzeClass(type);
212             fieldType = cf.getFieldType(targetName);
213         }
214         // special handling for indexed and mapped names
215         if (getResolver().isIndexed(name) || getResolver().isMapped(name)) {
216             if ( ConstructorUtils.isClassArray(fieldType) ) {
217                 // use the array component type
218                 fieldType = type.getComponentType();
219             } else {
220                 // default for contained type of holders
221                 fieldType = Object.class;
222             }
223         }
224         return fieldType;
225     }

227     /***
228      * Finds the type for a field based on the containing object and the field name
229      * @param obj any object
230      * @param name the name of a field in this object (can be nested, indexed, mapped, etc.)
231      * @return the type of the field (will be {@link Object} if the type is indeterminate)
232      * @throws FieldnameNotFoundException if the name is invalid for this obj
233      * @throws IllegalArgumentException if the params are null
234      */
235     @SuppressWarnings("unchecked")
236     public Class<?> getFieldType(Object obj, String name) {
237         if (obj == null || name == null) {
238             throw new IllegalArgumentException("obj and name must not be null");
239         }
240         if (Class.class.equals(obj)) {
241             // this is a class so we should pass it over to the other method
242             return getFieldType((Class<?>)obj, name); // EXIT
243         }
244         if ( Object.class.equals(obj.getClass()) ) {
245             return Object.class; // EXIT
246         }
247         // get the nested object or die
248         while (getResolver().hasNested(name)) {
249             String next = getResolver().next(name);
250             Object nestedBean = null;
251             if (Map.class.isAssignableFrom(obj.getClass())) {
252                 nestedBean = getValueOfMap((Map) obj, next);
253             } else if (getResolver().isMapped(next)) {
254                 nestedBean = getMappedValue(obj, next);
255             } else if (getResolver().isIndexed(next)) {
256                 nestedBean = getIndexedValue(obj, next);
257             } else {
258                 nestedBean = getSimpleValue(obj, next);
259             }
260             if (nestedBean == null) {
261                 // no auto create so we have to fail here
262                 throw new NullPointerException("Nested traversal failure: null field value for name (" 
263                         + name + ") on object class (" + obj.getClass() + ") for object: " + obj);
264             }
265             obj = nestedBean;
266             name = getResolver().remove(name);
267         }
268         String targetName = getResolver().getProperty(name); // simple name of target field
269         // get the type
270         Class<?> fieldType = null;
271         if (fieldAdapter.isAdaptableObject(obj)) {
272             fieldType = fieldAdapter.getFieldType(obj, targetName);
273         } else if ( ConstructorUtils.isClassObjectHolder(obj.getClass()) 
274                 || Object.class.equals(obj.getClass()) ) {
275             // special handling for the holder types, needed because attempting to analyze a map or other container will cause a failure
276             fieldType = Object.class;
277         } else {
278             // normal object
279             ClassFields<?> cf = analyzeObject(obj);
280             ClassProperty cp = cf.getClassProperty(targetName);
281             fieldType = cp.getType();
282         }
283         // special handling for indexed and mapped names
284         if (getResolver().isIndexed(name) || getResolver().isMapped(name)) {
285             if ( ConstructorUtils.isClassArray(fieldType) ) {
286                 // use the array component type
287                 fieldType = fieldType.getComponentType();
288             } else {
289                 // default for contained type of holders
290                 fieldType = Object.class;
291             }
292         }
293         return fieldType;
294     }

334     /***
335      * Get the values of all fields on an object but optionally filter the fields used
336      * @param obj any object
337      * @param filter (optional) indicates the fields to return the values for, can be null for defaults <br/>
338      * WARNING: getting the field values from settable only fields works as expected (i.e. you will an empty map)
339      * @param includeClassField if true then the value from the "getClass()" method is returned as part of the 
340      * set of object values with a type of {@link Class} and a field name of "class"
341      * @return a map of field name -> value
342      * @throws IllegalArgumentException if the obj is null
343      */
344     public Map<String, Object> getFieldValues(Object obj, FieldsFilter filter, boolean includeClassField) {
345         if (obj == null) {
346             throw new IllegalArgumentException("obj cannot be null");
347         }
348         Map<String, Object> values = new ArrayOrderedMap<String, Object>();
349         if (includeClassField) {
350             // add as the first field
351             values.put(ClassFields.FIELD_CLASS, obj.getClass());
352         }
353         if (fieldAdapter.isAdaptableObject(obj)) {
354             values.putAll( fieldAdapter.getFieldValues(obj, filter) );
355         } else {
356             Map<String, Class<?>> types = getFieldTypes(obj.getClass(), filter);
357             if (FieldsFilter.WRITEABLE.equals(filter)) {
358                 types.clear();
359             }
360             for (String name : types.keySet()) {
361                 try {
362                     Object o = getFieldValue(obj, name);
363                     values.put(name, o);
364                 } catch (RuntimeException e) {
365                     // failed to get the value so we will skip this one
366                     continue;
367                 }
368             }
369         }
370         return values;
371     }

373     /***
374      * Get the value of a field on an object,
375      * name can be nested, indexed, or mapped
376      * @param obj any object
377      * @param name the name of a field on this object
378      * @return the value of the field
379      * @throws FieldnameNotFoundException if this field name is invalid for this object
380      * @throws IllegalArgumentException if there is a failure getting the value
381      */
382     @SuppressWarnings("unchecked")
383     public Object getFieldValue(Object obj, String name) {
384         if (obj == null) {
385             throw new IllegalArgumentException("obj cannot be null");
386         }
387         if (name == null || "".equals(name)) {
388             throw new IllegalArgumentException("field name cannot be null or blank");
389         }
390 
391         // Resolve nested references
392         Holder holder = unpackNestedName(name, obj, false);
393         name = holder.getName();
394         obj = holder.getObject();
395 
396         Object value = null;
397         if (Map.class.isAssignableFrom(obj.getClass())) {
398             value = getValueOfMap((Map) obj, name);
399         } else if (getResolver().isMapped(name)) {
400             value = getMappedValue(obj, name);
401         } else if (getResolver().isIndexed(name)) {
402             value = getIndexedValue(obj, name);
403         } else {
404             value = getSimpleValue(obj, name);
405         }
406         return value;
407     }

423     /***
424      * Set the value of a field on an object (optionally convert the value to the field type),
425      * name can be nested, indexed, or mapped
426      * 
427      * @param obj any object
428      * @param name the name of a field on this object
429      * @param value the value to set the field to
430      * @param autoConvert if true then the value will be converted to the target value type if it is possible,
431      * otherwise the value must match the target type exactly
432      * @throws FieldnameNotFoundException if this field name is invalid for this object
433      * @throws IllegalArgumentException if there is a failure setting the value
434      * @throws UnsupportedOperationException if this value cannot be auto converted to the type specified
435      */
436     @SuppressWarnings("unchecked")
437     public void setFieldValue(Object obj, String name, Object value, boolean autoConvert) {
438         if (obj == null) {
439             throw new IllegalArgumentException("obj cannot be null");
440         }
441         if (name == null || "".equals(name)) {
442             throw new IllegalArgumentException("field name cannot be null or blank");
443         }
444 
445         // Resolve nested references
446         Holder holder = unpackNestedName(name, obj, true);
447         name = holder.getName();
448         obj = holder.getObject();
449 
450         if (autoConvert) {
451             // auto convert the value to the target type if possible
452             //            String targetName = getResolver().getProperty(name); // simple name of target field
453             // attempt to convert the value into the target type
454             Class<?> type = getFieldType(obj, name);
455             value = getConversionUtils().convert(value, type);
456         }
457 
458         // set the value
459         if (Map.class.isAssignableFrom(obj.getClass())) {
460             setValueOfMap((Map) obj, name, value);
461         } else if (getResolver().isMapped(name)) {
462             setMappedValue(obj, name, value);
463         } else if (getResolver().isIndexed(name)) {
464             setIndexedValue(obj, name, value);
465         } else {
466             setSimpleValue(obj, name, value);
467         }
468     }

470     /***
471      * For setting an indexed value on an indexed object directly,
472      * indexed objects are lists and arrays <br/>
473      * NOTE: If the supplied index is invalid for the array then this will fail
474      * 
475      * @param indexedObject any array or list
476      * @param index the index to put the value into (will append to the end of the list if index < 0), must be within the bounds of the array
477      * @param value any value, will be converted to the correct type for the array automatically
478      * @throws IllegalArgumentException if there is a failure because of an invalid index or null arguments
479      */
480     @SuppressWarnings("unchecked")
481     public void setIndexedValue(Object indexedObject, int index, Object value) {
482         if (indexedObject == null) {
483             throw new IllegalArgumentException("Invalid indexedObject, cannot be null");
484         }
485         if ( ConstructorUtils.isClassArray(indexedObject.getClass()) ) {
486             // this is an array
487             try {
488                 // set the value on the array
489                 // NOTE: cannot automatically expand the array
490                 Class<?> componentType = ArrayUtils.type((Object[])indexedObject);
491                 Object convert = ReflectUtils.getInstance().convert(value, componentType);
492                 Array.set(indexedObject, index, convert);
493             } catch (Exception e) {
494                 throw new IllegalArgumentException("Failed to set index ("+index+") for array of size ("+Array.getLength(indexedObject)+") to value: " + value, e);
495             }
496         } else if ( ConstructorUtils.isClassList(indexedObject.getClass()) ) {
497             // this is a list
498             List l = (List) indexedObject;
499             try {
500                 // set value on list
501                 if (index < 0) {
502                     l.add(value);
503                 } else {
504                     if (index >= l.size()) {
505                         // automatically expand the list
506                         int start = l.size();
507                         for (int i = start; i < (index+1); i++) {
508                             l.add(i, null);
509                         }
510                     }
511                     l.set(index, value);
512                 }
513             } catch (Exception e) {
514                 // catching the general exception is correct here, translate the exception
515                 throw new IllegalArgumentException("Failed to set index ("+index+") for list of size ("+l.size()+") to value: " + value, e);
516             }
517         } else {
518             // invalid
519             throw new IllegalArgumentException("Object does not appear to be indexed (not an array or a list): " 
520                     + (indexedObject == null ? "NULL" : indexedObject.getClass()) );
521         }
522     }

527     /***
528      * Traverses the nested name path to get to the requested name and object
529      * @param fullName the full path name (e.g. thing.field1.field2.stuff)
530      * @param object the object to traverse
531      * @param autoCreate if true then create the nested objects to force successful traversal, else will throw NPE
532      * @return a holder with the nested name (e.g. stuff) and the nested object
533      * @throws NullPointerException if the path cannot be traversed
534      * @throws IllegalArgumentException if the path is invalid
535      */
536     @SuppressWarnings("unchecked")
537     protected Holder unpackNestedName(final String fullName, final Object object, boolean autoCreate) {
538         String name = fullName;
539         Object obj = object;
540         Class<?> cls = object.getClass();
541         try {
542             // Resolve nested references
543             while (getResolver().hasNested(name)) {
544                 String next = getResolver().next(name);
545                 Object nestedBean = null;
546                 if (Map.class.isAssignableFrom(obj.getClass())) {
547                     nestedBean = getValueOfMap((Map) obj, next);
548                 } else if (getResolver().isMapped(next)) {
549                     nestedBean = getMappedValue(obj, next);
550                 } else if (getResolver().isIndexed(next)) {
551                     nestedBean = getIndexedValue(obj, next);
552                 } else {
553                     nestedBean = getSimpleValue(obj, next);
554                 }
555                 if (nestedBean == null) {
556                     // could not get the nested bean because it is unset
557                     if (autoCreate) {
558                         // create the nested bean
559                         try {
560                             Class<?> type = getFieldType(obj, next);
561                             if (Object.class.equals(type)) {
562                                 // indeterminate type so we will make a map
563                                 type = ArrayOrderedMap.class;
564                             }
565                             nestedBean = getConstructorUtils().constructClass(type);
566                             setFieldValue(obj, next, nestedBean, false); // need to put this new object into the parent
567                         } catch (RuntimeException e) {
568                             throw new IllegalArgumentException("Nested path failure: Could not create nested object ("
569                                     +cls.getName()+") in path ("+fullName+"): " + e.getMessage(), e);
570                         }
571                     } else {
572                         // no auto create so we have to fail here
573                         throw new NullPointerException("Nested traversal failure: null field value for name (" 
574                                 + name + ") in nestedName ("+fullName+") on object class (" + cls + ") for object: " + obj);
575                     }
576                 }
577                 obj = nestedBean;
578                 name = getResolver().remove(name);
579             }
580         } catch (FieldnameNotFoundException e) {
581             // convert field name failure into illegal argument
582             throw new IllegalArgumentException("Nested path failure: Invalid path name ("
583                     +fullName+") contains invalid field names: " + e.getMessage(), e);
584         }
585         return new Holder(name, obj);
586     }

588     /***
589      * For getting a value out of a bean based on a field name
590      * <br/>
591      * <b>WARNING: Cannot handle a nested/mapped/indexed name</b>
592      * @throws FieldnameNotFoundException if this field name is invalid for this object
593      * @throws IllegalArgumentException if there is failure
594      */
595     protected Object getSimpleValue(Object obj, String name) {
596         if (obj == null) {
597             throw new IllegalArgumentException("obj cannot be null");
598         }
599         if (name == null || "".equals(name)) {
600             throw new IllegalArgumentException("field name cannot be null or blank");
601         }
602 
603         Object value = null;
604         // Handle DynaBean instances specially
605         if (fieldAdapter.isAdaptableObject(obj)) {
606             value = fieldAdapter.getSimpleValue(obj, name);
607         } else {
608             // normal bean
609             ClassFields<?> cf = analyzeObject(obj);
610             ClassProperty cp = cf.getClassProperty(name);
611             value = findFieldValue(obj, cp);
612         }
613         return value;
614     }

616     /***
617      * For getting an indexed value out of an object based on field name,
618      * name must be like: fieldname[index],
619      * If the object is an array or a list then name can be the index only:
620      * "[1]" or "[0]" (brackets must be included)
621      * <br/>
622      * <b>WARNING: Cannot handle a nested/mapped/indexed name</b>
623      * @throws FieldnameNotFoundException if this field name is invalid for this object
624      * @throws IllegalArgumentException if there is a failure
625      * @throws FieldGetValueException if there is an internal failure getting the field
626      */
627     @SuppressWarnings("unchecked")
628     protected Object getIndexedValue(Object obj, String name) {
629         if (obj == null) {
630             throw new IllegalArgumentException("obj cannot be null");
631         }
632         if (name == null || "".equals(name)) {
