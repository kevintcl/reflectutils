168     @SuppressWarnings("unchecked")
169     protected static String toXML(Object object, String tagName, int level, int maxLevel, boolean humanOutput, boolean includeNulls, boolean includeClass, boolean includeClassField, Map<String, Object> properties) {
170         StringBuilder sb = new StringBuilder();
171 
172         if (object == null) {
173             if (includeNulls) {
174                 // nulls are empty tags always
175                 tagName = validate(tagName == null ? "null" : tagName);
176                 makeLevelSpaces(sb, level, humanOutput);
177                 sb.append(LT);
178                 sb.append(tagName);
179                 sb.append(SLASH);
180                 sb.append(GT);
181                 makeEOL(sb, humanOutput);
182             }
183         } else {
184             Class<?> type = ConstructorUtils.getWrapper(object.getClass());
185             if ( ConstructorUtils.isClassSimple(type) ) {
186                 // Simple (String, Number, etc.)
187                 tagName = validate(tagName == null ? makeElementName(type) : tagName);
188                 String value = "";
189                 makeLevelSpaces(sb, level, humanOutput);
190                 sb.append(LT);
191                 sb.append(tagName);
192                 if (Date.class.isAssignableFrom(type) || Calendar.class.isAssignableFrom(type)) {
193                     // date
194                     Date d = null;
195                     if (Date.class.isAssignableFrom(type)) {
196                         d = (Date) object;
197                     } else {
198                         d = ((Calendar) object).getTime();
199                     }
200                     value = d.getTime()+"";
201                     sb.append(" type='date' date='");
202                     sb.append( DateUtils.makeDateISO8601(d) );
203                     sb.append(APOS);
204                 } else if (Number.class.isAssignableFrom(type)) {
205                     // number
206                     sb.append(" type='number'");
207                     if (includeClass) {
208                         makeClassName(sb, type);
209                     }
210                     value = object.toString();
211                 } else if (Boolean.class.isAssignableFrom(type)) {
212                     // boolean
213                     value = object.toString();
214                     sb.append(" type='boolean'");
215                 } else {
216                     value = escapeForXML( object.toString() );
217                 }
218                 sb.append(GT);
219                 sb.append(value);
220                 sb.append(LT);
221                 sb.append(SLASH);
222                 sb.append(tagName);
223                 sb.append(GT);
224                 makeEOL(sb, humanOutput);
225             } else if ( ConstructorUtils.isClassArray(type) ) {
226                 // ARRAY
227                 tagName = validate(tagName == null ? "array" : tagName);
228                 int length = ArrayUtils.size((Object[])object);
229                 Class<?> elementType = ArrayUtils.type((Object[])object);
230                 makeLevelSpaces(sb, level, humanOutput);
231                 sb.append(LT);
232                 sb.append(tagName);
233                 sb.append(" type='array' length='");
234                 sb.append(length);
235                 sb.append(APOS);
236                 if (includeClass) {
237                     sb.append(" component='");
238                     sb.append( ConstructorUtils.getTypeFromInnerCollection(elementType).getName() );
239                     sb.append(APOS);
240                 }
241                 sb.append(GT);
242                 makeEOL(sb, humanOutput);
243                 for (int i = 0; i < length; ++i) {
244                     sb.append( toXML(Array.get(object, i), makeElementName(elementType), level+1, maxLevel, humanOutput, includeNulls, includeClass, includeClassField, properties) );
245                 }
246                 makeLevelSpaces(sb, level, humanOutput);
247                 sb.append(LT);
248                 sb.append(SLASH);
249                 sb.append(tagName);
250                 sb.append(GT);
251                 makeEOL(sb, humanOutput);
252             } else if ( ConstructorUtils.isClassCollection(type) ) {
253                 // COLLECTION
254                 tagName = validate(tagName == null ? "collection" : tagName);
255                 Collection<Object> collection = (Collection) object;
256                 makeLevelSpaces(sb, level, humanOutput);
257                 sb.append(LT);
258                 sb.append(tagName);
259                 sb.append(" type='collection' size='");
260                 sb.append(collection.size());
261                 sb.append(APOS);
262                 if (includeClass) {
263                     makeClassName(sb, ConstructorUtils.getTypeFromInnerCollection(type));
264                 }
265                 sb.append(GT);
266                 makeEOL(sb, humanOutput);
267                 for (Object element : collection) {
268                     Class<?> elementType = null;
269                     if (element != null) {
270                         elementType = element.getClass();
271                     }
272                     sb.append( toXML(element, makeElementName(elementType), level+1, maxLevel, humanOutput, includeNulls, includeClass, includeClassField, properties) );
273                 }
274                 makeLevelSpaces(sb, level, humanOutput);
275                 sb.append(LT);
276                 sb.append(SLASH);
277                 sb.append(tagName);
278                 sb.append(GT);
279                 makeEOL(sb, humanOutput);
280             } else {
281                 // must be a bean or map, make sure it is a map
282                 tagName = validate(tagName == null ? makeElementName(type) : tagName);
283                 if (maxLevel <= level) {
284                     // if the max level was reached then stop
285                     sb.append(LT);
286                     sb.append(tagName);
287                     sb.append(GT);
288                     sb.append( "MAX level reached (" );
289                     sb.append( level );
290                     sb.append( "):" );
291                     sb.append( escapeForXML(object.toString()) );
292                     sb.append(LT);
293                     sb.append(SLASH);
294                     sb.append(tagName);
295                     sb.append(GT);
296                     makeEOL(sb, humanOutput);
297                 } else {
298                     String xmlType = "bean";
299                     Map<String, Object> map = null;
300                     if (Map.class.isAssignableFrom(type)) {
301                         xmlType = "map";
302                         map = (Map<String, Object>) object;
303                     } else {
304                         // reflect over objects
305                         map = ReflectUtils.getInstance().getObjectValues(object, FieldsFilter.SERIALIZABLE, false);
306                     }
307                     // add in the optional properties if it makes sense to do so
308                     if (level == 0 && properties != null && ! properties.isEmpty()) {
309                         map.putAll(properties);
310                     }
311                     makeLevelSpaces(sb, level, humanOutput);
312                     sb.append(LT);
313                     sb.append(tagName);
314                     sb.append(" type='");
315                     sb.append(xmlType);
316                     sb.append(APOS);
317                     sb.append(" size='");
318                     sb.append(map.size());
319                     sb.append(APOS);
320                     if (includeClass) {
321                         makeClassName(sb, ConstructorUtils.getTypeFromInnerCollection(type));
322                     }
323                     sb.append(GT);
324                     makeEOL(sb, humanOutput);
325                     for (Entry<String, Object> entry : map.entrySet()) {
326                         if (entry.getKey() != null) {
327                             sb.append( toXML(entry.getValue(), entry.getKey().toString(), level+1, maxLevel, humanOutput, includeNulls, includeClass, includeClassField, properties) );
328                         }
329                     }
330                     makeLevelSpaces(sb, level, humanOutput);
331                     sb.append(LT);
332                     sb.append(SLASH);
333                     sb.append(tagName);
334                     sb.append(GT);
335                     makeEOL(sb, humanOutput);
336                 }
337             }
338         }
339         return sb.toString();
340     }

376     /***
377      * Escape a string for XML encoding: replace special characters with XML escapes:
378      * <pre>
379      * &amp; <small>(ampersand)</small> is replaced by &amp;amp;
380      * &lt; <small>(less than)</small> is replaced by &amp;lt;
381      * &gt; <small>(greater than)</small> is replaced by &amp;gt;
382      * &quot; <small>(double quote)</small> is replaced by &amp;quot;
383      * </pre>
384      * @param string The string to be escaped.
385      * @return The escaped string.
386      */
387     public static String escapeForXML(String string) {
388         StringBuilder sb = new StringBuilder();
389         for (int i = 0, len = string.length(); i < len; i++) {
390             char c = string.charAt(i);
391             switch (c) {
392             case AMP:
393                 sb.append("&amp;");
394                 break;
395             case LT:
396                 sb.append("&lt;");
397                 break;
398             case GT:
399                 sb.append("&gt;");
400                 break;
401             case QUOT:
402                 sb.append("&quot;");
403                 break;
404             default:
405                 sb.append(c);
406             }
407         }
408         return sb.toString();
409     }

411     /***
412      * Validates that a string contains no spaces and is non-null/non-empty
413      * Throw an exception if the string contains whitespace. 
414      * Whitespace is not allowed in tagNames and attributes.
415      * @param string any string
416      * @throws IllegalArgumentException
417      */
418     public static String validate(String string) {
419         if (string == null) {
420             throw new IllegalArgumentException("string is NULL");
421         }
422         int i, length = string.length();
423         if (length == 0) {
424             throw new IllegalArgumentException("Empty string.");
425         }
426         for (i = 0; i < length; i += 1) {
427             if (Character.isWhitespace(string.charAt(i))) {
428                 throw new IllegalArgumentException("'" + string + "' contains a space character.");
429             }
430         }
431         return string;
432     }

495         /***
496          * Adds the value to the container using the given key,
497          * if the key already exists in the container then the container needs to be switched
498          * over to a collection and its contents moved, then the stack needs to be updated,
499          * and finally the parent container needs to have it's value replaced
500          */
501         @SuppressWarnings("unchecked")
502         protected void add(Container container, String key, Object value) {
503             // first we need to make sure this container is on the stack
504 //            if (containerStack.peek() != container) {
505 //                containerStack.push( new Container(container.getContainer(), key, value) );
506 //            }
507             // now do the add
508             Class<?> type = container.getContainer().getClass();
509             if ( ConstructorUtils.isClassMap(type)) {
510                 Map<String, Object> m = (Map)container.getContainer();
511                 if (m.containsKey(key)) {
512                     // this should have been a collection so replace the map and move elements over to collection
513                     Collection collection = (Collection) makeContainerObject(Types.COLLECTION);
514                     for (Entry entry : m.entrySet()) {
515                         collection.add( entry.getValue());
516                     }
517                     collection.add(value);
518                     // now replace the container in the stack
519                     int endPosition = containerStack.size()-1;
520                     int containerPosition = endPosition;
521                     if (container != containerStack.peek() && containerPosition != 0) {
522                         containerPosition--;
523                     }
524                     Container current = containerStack.get(containerPosition);
525                     current.replaceContainer(collection); // update container and replace the value in the parent object in the container
526                     // finally we need to get the next thing in the stack to point back at the new parent
527                     if (containerPosition < endPosition) {
528                         // there is another container on the stack which needs to be replaced
529                         containerStack.set(endPosition, new Container(collection, 1, value) );
530                     }
531                 } else {
532                     m.put(key, value);
533                 }
534             } else if ( ConstructorUtils.isClassCollection(type)) {
535                 Collection collection = ((Collection)container.getContainer());
536                 collection.add(value);
537                 // make sure the parent index is correct
538                 if (container != containerStack.peek()) {
539                     containerStack.peek().updateIndex(collection.size() - 1);
540                 }
541             } else {
542                 // bean or something we hope
543                 try {
544                     ReflectUtils.getInstance().setFieldValue(container.getContainer(), key, value);
545                 } catch (RuntimeException e) {
546                     throw new RuntimeException("Unknown container type ("+type+") and could not set field on container: " + container, e);
547                 }
548             }
549         }

551         private Stack<String> tagStack = new Stack<String>();
552         private Stack<Container> containerStack = new Stack<Container>();
553 
554         private CharArrayWriter contents = new CharArrayWriter();
555         private Types currentType = null;
556         // this should be false when there are no contents to read
557         private boolean currentContents = false;
558 
559         // Event Handlers
560         @Override
561         public void startElement(String uri, String localName, String name, Attributes attributes) throws SAXException {
562             contents.reset();
563             tagStack.push(localName);
564             if (tagStack.size() > containerStack.size() + 1) {
565                 // add a new container to the stack, use the types info from the parent
566                 Container lastContainer = containerStack.peek();
567                 Object newContainerObject = makeContainerObject(currentType);
568                 String parentName = ( tagStack.size() > 1 ? tagStack.get(tagStack.size()-2) : tagStack.peek() );
569                 containerStack.push( new Container(lastContainer.getContainer(), parentName, newContainerObject) );
570                 add(lastContainer, parentName, newContainerObject);
571             }
572             currentType = getDataType(attributes);
573             currentContents = false;
574         }
575 
576         @Override
577         public void endElement(String uri, String localName, String name) throws SAXException {
578             if (tagStack.size() > containerStack.size()) {
579                 // only add data when we are above a container
580                 Object val = null;
581                 if (currentContents) {
582                     String content = unescapeXML(contents.toString().trim());
583                     val = content;
584                     if (Types.BOOLEAN.equals(currentType)) {
585                         val = Boolean.valueOf(content);
586                     } else if (Types.NUMBER.equals(currentType)) {
587                         try {
588                             val = number(content);
589                         } catch (NumberFormatException e) {
590                             val = content;
591                         }
592                     } else if (Types.DATE.equals(currentType)) {
593                         try {
594                             val = new Date(Long.valueOf(content));
595                         } catch (NumberFormatException e) {
596                             val = content;
597                         }
598                     }
599                 }
600                 // put the value into the current container
601                 add(containerStack.peek(), localName, val);
602             }
603             if (tagStack.isEmpty()) {
604                 throw new IllegalStateException("tag stack is out of sync, empty while still processing tags: " + localName);
605             } else {
606                 tagStack.pop();
607             }
608             // now we need to remove the current container if we are done with it
609             while (tagStack.size() < containerStack.size()) {
610                 if (containerStack.size() <= 1) break;
611                 containerStack.pop();
612             }
613             contents.reset();
614         }

618             // get the text out of the element
619             contents.write(ch, start, length);
620             currentContents = true;
621         }

635     protected static Types getDataType(Attributes attributes) {
636         Types elementType = Types.STRING;
637         String value = attributes.getValue("", "type");
638         if (value != null) {
639             if ("boolean".equals(value)) {
640                 elementType = Types.BOOLEAN;
641             } else if ("number".equals(value)) {
642                 elementType = Types.NUMBER;
643             } else if ("date".equals(value)) {
644                 elementType = Types.DATE;
645             } else if ("array".equals(value)) {
646                 elementType = Types.ARRAY;
647             } else if ("collection".equals(value)) {
648                 elementType = Types.COLLECTION;
649             } else if ("map".equals(value)) {
650                 elementType = Types.MAP;
651             } else if ("bean".equals(value)) {
652                 elementType = Types.BEAN;
653             }
654         }
655         return elementType;
656     }

682     /***
683      * Converts a string into a number
684      * @param s the string
685      * @return the number
686      * @throws NumberFormatException if the string is not a number
687      */
688     @SuppressWarnings("fallthrough")
689     protected static Number number(String s) {
690         int length = s.length();
691         boolean isFloatingPoint = false;
692 
693         for (int i = 0; i < s.length(); i++) {
694             char c = s.charAt(i);
695             switch (c) {
696             case '.':
697             case 'e':
698             case 'E':
699                 isFloatingPoint = true;
700             case '-':
701             case '+':
702                 length--;
703             }
704         }
705 
706         // more friendly handling of numbers
707         Number num = null;
708         if (isFloatingPoint) {
709             if (length < 10) {
710                 num = Float.valueOf(s);
711             } else if (length < 17) {
712                 num = Double.valueOf(s);
713             } else {
714                 num = new BigDecimal(s);
715             }
716         } else {
717             if (length < 10) {
718                 num = Integer.valueOf(s);
719             } else if (length < 19) {
720                 num = Long.valueOf(s);
721             } else {
722                 num = new BigInteger(s);
723             }
724         }
725         return num;
726     }
