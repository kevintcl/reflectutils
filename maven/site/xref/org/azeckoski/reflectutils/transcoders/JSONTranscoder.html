67      @SuppressWarnings("unchecked")
68      public Map<String, Object> decode(String string) {
69          Map<String, Object> decoded = null;
70          Object decode = new JsonReader().read(string);
71          if (decode instanceof Map) {
72              decoded = (Map<String, Object>) decode;
73          } else {
74              // for JSON if the result is not a map then simply put the result into a map
75              decoded = new ArrayOrderedMap<String, Object>();
76              decoded.put(Transcoder.DATA_KEY, decode);
77          }
78          return decoded;
79      }

164     @SuppressWarnings("unchecked")
165     protected static String toJSON(Object object, int level, int maxLevel, boolean humanOutput, boolean includeNulls, boolean includeClassField, Map<String, Object> properties) {
166         StringBuilder sb = new StringBuilder();
167 
168         if (object == null) {
169             if (includeNulls) {
170                 // nulls use the constant
171                 sb.append(NULL);
172             }
173         } else {
174             Class<?> type = ConstructorUtils.getWrapper(object.getClass());
175             if ( ConstructorUtils.isClassSimple(type) ) {
176                 // Simple (String, Number, etc.)
177                 if (Date.class.isAssignableFrom(type) || Calendar.class.isAssignableFrom(type)) {
178                     // date
179                     Date d = null;
180                     if (Date.class.isAssignableFrom(type)) {
181                         d = (Date) object;
182                     } else {
183                         d = ((Calendar) object).getTime();
184                     }
185                     sb.append(d.getTime());
186                 } else if (Number.class.isAssignableFrom(type)) {
187                     // number
188                     sb.append(object.toString());
189                 } else if (Boolean.class.isAssignableFrom(type)) {
190                     // boolean
191                     if ( ((Boolean)object).booleanValue() ) {
192                         sb.append(BOOLEAN_TRUE);
193                     } else {
194                         sb.append(BOOLEAN_FALSE);
195                     }
196                 } else {
197                     sb.append(QUOT);
198                     sb.append( escapeForJSON(object.toString()) );
199                     sb.append(QUOT);
200                 }
201             } else if ( ConstructorUtils.isClassArray(type) ) {
202                 // ARRAY
203                 int length = ArrayUtils.size((Object[])object);
204                 sb.append(ARRAY_BEG);
205                 if (length > 0) {
206                     for (int i = 0; i < length; ++i) {
207                         if (i > 0) {
208                             sb.append(JSON_SEP);
209                         }
210                         makeEOL(sb, humanOutput);
211                         makeLevelSpaces(sb, level+1, humanOutput);
212                         sb.append( toJSON(Array.get(object, i), level+1, maxLevel, humanOutput, includeNulls, includeClassField, properties) );
213                     }
214                     makeEOL(sb, humanOutput);
215                     makeLevelSpaces(sb, level, humanOutput);
216                 }
217                 sb.append(ARRAY_END);
218             } else if ( ConstructorUtils.isClassCollection(type) ) {
219                 // COLLECTION
220                 Collection<Object> collection = (Collection) object;
221                 sb.append(ARRAY_BEG);
222                 if (! collection.isEmpty()) {
223                     boolean first = true;
224                     for (Object element : collection) {
225                         if (first) {
226                             first = false;
227                         } else {
228                             sb.append(JSON_SEP);
229                         }
230                         makeEOL(sb, humanOutput);
231                         makeLevelSpaces(sb, level+1, humanOutput);
232                         sb.append( toJSON(element, level+1, maxLevel, humanOutput, includeNulls, includeClassField, properties) );
233                     }
234                     makeEOL(sb, humanOutput);
235                     makeLevelSpaces(sb, level, humanOutput);
236                 }
237                 sb.append(ARRAY_END);
238             } else {
239                 // must be a bean or map, make sure it is a map
240                 if (maxLevel <= level) {
241                     // if the max level was reached then stop
242                     sb.append(QUOT);
243                     sb.append( "MAX level reached (" );
244                     sb.append( level );
245                     sb.append( "):" );
246                     sb.append( escapeForJSON(object.toString()) );
247                     sb.append(QUOT);
248                 } else {
249                     Map<String, Object> map = null;
250                     if (Map.class.isAssignableFrom(type)) {
251                         map = (Map<String, Object>) object;
252                     } else {
253                         // reflect over objects
254                         map = ReflectUtils.getInstance().getObjectValues(object, FieldsFilter.SERIALIZABLE, includeClassField);
255                     }
256                     // add in the optional properties if it makes sense to do so
257                     if (level == 0 && properties != null && ! properties.isEmpty()) {
258                         map.putAll(properties);
259                     }
260                     sb.append(OBJ_BEG);
261                     boolean first = true;
262                     for (Entry<String, Object> entry : map.entrySet()) {
263                         if (entry.getKey() != null) {
264                             Object value = entry.getValue();
265                             if (value != null || includeNulls) {
266                                 if (first) {
267                                     first = false;
268                                 } else {
269                                     sb.append(JSON_SEP);
270                                 }
271                                 makeEOL(sb, humanOutput);
272                                 makeLevelSpaces(sb, level+1, humanOutput);
273                                 sb.append(QUOT);
274                                 sb.append(entry.getKey());
275                                 sb.append(QUOT);
276                                 sb.append(OBJ_SEP);
277                                 if (humanOutput) { sb.append(SPACE); }
278                                 sb.append( toJSON(value, level+1, maxLevel, humanOutput, includeNulls, includeClassField, properties) );
279                             }
280                         }
281                     }
282                     makeEOL(sb, humanOutput);
283                     makeLevelSpaces(sb, level, humanOutput);
284                     sb.append(OBJ_END);
285                 }
286             }
287         }
288         return sb.toString();
289     }

306     /***
307      * Escape a string for JSON encoding
308      * @param string any string
309      * @return the escaped string
310      */
311     public static String escapeForJSON(String string) {
312         StringBuilder sb = new StringBuilder();
313         if (string != null) {
314             for (int i = 0, len = string.length(); i < len; i++) {
315                 char c = string.charAt(i);
316                 switch (c) {
317                 case QUOT:
318                     sb.append("//\"");
319                     break;
320                 case BACK:
321                     sb.append("////");
322                     break;
323                 case SLASH:
324                     sb.append("///");
325                     break;
326                 case '\b':
327                     sb.append("//b");
328                     break;
329                 case '\f':
330                     sb.append("//f");
331                     break;
332                 case '\n':
333                     sb.append("//n");
334                     break;
335                 case '\r':
336                     sb.append("//r");
337                     break;
338                 case '\t':
339                     sb.append("//t");
340                     break;
341                 default:
342                     if (Character.isISOControl(c)) {
343                         sb.append("//u");
344                         int n = c;
345                         for (int j = 0; j < 4; ++j) {
346                             int digit = (n & 0xf000) >> 12;
347                             sb.append(hex[digit]);
348                             n <<= 4;
349                         }
350                     } else {
351                         sb.append(c);
352                     }
353                 }
354             }
355         }
356         return sb.toString();
357     }

384     /***
385      * Create Java objects from JSON (note that only simple java objects, maps, and arrays will be returned) <br/>
386      * Dates will come back in as UTC timecodes or possibly strings which you will need to parse manually <br/>
387      * Numbers will come in as int if they are small, long if they are big, and BigInteger if they are huge,
388      * floating point is handled similarly: float, double, BigDecimal <br/>
389      * JSON arrays come back as a List always, similarly, any collection or array that was output will come back as a list <br/>
390      * You can use the {@link ConversionUtils} to help with conversion if needed <br/>
391      * 
392      * Derived from code at:
393      * https://svn.sourceforge.net/svnroot/stringtree/trunk/src/delivery/java/org/stringtree/json/JSONWriter.java
394      */

413         public Object read(CharacterIterator ci, int start) {
414             it = ci;
415             switch (start) {
416             case FIRST:
417                 c = it.first();
418                 break;
419             case CURRENT:
420                 c = it.current();
421                 break;
422             case NEXT:
423                 c = it.next();
424                 break;
425             }
426             return read();
427         }

429         public Object read(CharacterIterator it) {
430             return read(it, NEXT);
431         }

433         public Object read(String string) {
434             return read(new StringCharacterIterator(string), FIRST);
435         }

437         private Object read() {
438             skipWhiteSpace();
439             char ch = c;
440             next();
441             switch (ch) {
442                 case '"': token = string(); break;
443                 case '[': token = array(); break;
444                 case ']': token = MARK_ARRAY_END; break;
445                 case ',': token = MARK_COMMA; break;
446                 case '{': token = object(); break;
447                 case '}': token = MARK_OBJECT_END; break;
448                 case ':': token = MARK_COLON; break;
449                 case 't':
450                     next(); next(); next(); // assumed r-u-e
451                     token = Boolean.TRUE;
452                     break;
453                 case'f':
454                     next(); next(); next(); next(); // assumed a-l-s-e
455                     token = Boolean.FALSE;
456                     break;
457                 case 'n':
458                     next(); next(); next(); // assumed u-l-l
459                     token = null;
460                     break;
461                 default:
462                     c = it.previous();
463                     if (Character.isDigit(c) || c == '-') {
464                         token = number();
465                     }
466             }
467             // System.out.println("token: " + token); // enable this line to see the token stream
468             return token;
469         }
470         
471         private Object object() {
472             Map<Object, Object> ret = new ArrayOrderedMap<Object, Object>();
473             Object key = read();
474             while (token != MARK_OBJECT_END) {
475                 read(); // should be a colon
476                 if (token != MARK_OBJECT_END) {
477                     ret.put(key, read());
478                     if (read() == MARK_COMMA) {
479                         key = read();
480                     }
481                 }
482             }
483 
484             return ret;
485         }
486 
487         private Object array() {
488             List<Object> ret = new ArrayList<Object>();
489             Object value = read();
490             while (token != MARK_ARRAY_END) {
491                 ret.add(value);
492                 if (read() == MARK_COMMA) {
493                     value = read();
494                 }
495             }
496             return ret;
497         }

499         private Object number() {
500             int length = 0;
501             boolean isFloatingPoint = false;
502             buf.setLength(0);
503             
504             if (c == '-') {
505                 add();
506             }
507             length += addDigits();
508             if (c == '.') {
509                 add();
510                 length += addDigits();
511                 isFloatingPoint = true;
512             }
513             if (c == 'e' || c == 'E') {
514                 add();
515                 if (c == '+' || c == '-') {
516                     add();
517                 }
518                 addDigits();
519                 isFloatingPoint = true;
520             }
521      
522             String s = buf.toString();
523             // more friendly handling of numbers
524             Object num = null;
525             if (isFloatingPoint) {
526                 if (length < 10) {
527                     num = Float.valueOf(s);
528                 } else if (length < 17) {
529                     num = Double.valueOf(s);
530                 } else {
531                     num = new BigDecimal(s);
532                 }
533             } else {
534                 if (length < 10) {
535                     num = Integer.valueOf(s);
536                 } else if (length < 19) {
537                     num = Long.valueOf(s);
538                 } else {
539                     num = new BigInteger(s);
540                 }
541             }
542             return num;
543         }
544      
545         private int addDigits() {
546             int ret;
547             for (ret = 0; Character.isDigit(c); ++ret) {
548                 add();
549             }
550             return ret;
551         }
552 
553         private Object string() {
554             buf.setLength(0);
555             while (c != '"') {
556                 if (c == '//') {
557                     next();
558                     if (c == 'u') {
559                         add(unicode());
560                     } else {
561                         Object value = escapes.get(new Character(c));
562                         if (value != null) {
563                             add(((Character) value).charValue());
564                         }
565                     }
566                 } else {
567                     add();
568                 }
569             }
570             next();
571 
572             return buf.toString();
573         }
574 
575         private void add(char cc) {
576             buf.append(cc);
577             next();
578         }
579 
580         private void add() {
581             add(c);
582         }

584         private char unicode() {
585             int value = 0;
586             for (int i = 0; i < 4; ++i) {
587                 switch (next()) {
588                 case '0': case '1': case '2': case '3': case '4': 
589                 case '5': case '6': case '7': case '8': case '9':
590                     value = (value << 4) + c - '0';
591                     break;
592                 case 'a': case 'b': case 'c': case 'd': case 'e': case 'f':
593                     value = (value << 4) + (c - 'a') + 10;
594                     break;
595                 case 'A': case 'B': case 'C': case 'D': case 'E': case 'F':
596                     value = (value << 4) + (c - 'A') + 10;
597                     break;
598                 }
599             }
600             return (char) value;
601         }
