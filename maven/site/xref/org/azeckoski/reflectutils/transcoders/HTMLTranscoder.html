40      public String encode(Object object, String name, Map<String, Object> properties) {
41          String encoded = "";
42          if (object != null) {
43              if (name == null || "".equals(name)) {

45              }
46          }

100     /***
101      * Convert an object into a well-formed, element-normal HTML string.
102      * @param object any object
103      * @param tagName (optional) enclosing root tag
104      * @param humanOutput true of human readable output
105      * @param includeNulls true to include null values when generating tags
106      * @return the HTML string version of the object
107      */
108     public static String makeHTML(Object object, String tagName, Map<String, Object> properties, boolean humanOutput, boolean includeNulls, boolean includeClassField) {
109         return "<table border='1'>\n" 
110             + toHTML(object, tagName, 0, humanOutput, includeNulls, includeClassField, properties) 
111             + "</table>\n";
112     }

114     @SuppressWarnings("unchecked")
115     protected static String toHTML(Object object, String tagName, int level, boolean humanOutput, boolean includeNulls, boolean includeClassField, Map<String, Object> properties) {
116         StringBuilder sb = new StringBuilder();
117 
118         if (object == null) {
119             if (includeNulls) {
120                 // nulls are empty tags always
121                 tagName = validate(tagName == null ? "null" : tagName);
122                 makeLevelSpaces(sb, level, humanOutput);
123                 sb.append("<tr><td>");
124                 sb.append(tagName);
125                 sb.append("</td><td>");
126                 sb.append("<i>NULL</i>");
127                 sb.append("</td></tr>");
128                 makeEOL(sb, humanOutput);
129             }
130         } else {
131             Class<?> type = ConstructorUtils.getWrapper(object.getClass());
132             if ( ConstructorUtils.isClassSimple(type) ) {
133                 // Simple (String, Number, etc.)
134                 tagName = validate(tagName == null ? makeElementName(type) : tagName);
135                 makeLevelSpaces(sb, level, humanOutput);
136                 String value = escapeForXML( object.toString() );
137                 sb.append("<tr><td>");
138                 sb.append(tagName);
139                 sb.append("</td><td>");
140                 sb.append(value);
141                 sb.append("</td></tr>");
142                 makeEOL(sb, humanOutput);
143             } else if ( ConstructorUtils.isClassArray(type) ) {
144                 // ARRAY
145                 tagName = validate(tagName == null ? "array" : tagName);
146                 int length = ArrayUtils.size((Object[])object);
147                 Class<?> elementType = ArrayUtils.type((Object[])object);
148                 makeLevelSpaces(sb, level, humanOutput);
149                 sb.append("<tr><td width='3%'>");
150                 sb.append(tagName);
151                 sb.append(" type=array");
152                 sb.append(" length="+length);
153                 sb.append("</td><td>");
154                 makeEOL(sb, humanOutput);
155                 makeLevelSpaces(sb, level+1, humanOutput);
156                 sb.append("<table border='1'>");
157                 makeEOL(sb, humanOutput);
158                 for (int i = 0; i < length; ++i) {
159                     sb.append( toHTML(Array.get(object, i), makeElementName(elementType), level+2, humanOutput, includeNulls, includeClassField, properties) );
160                 }
161                 makeLevelSpaces(sb, level+1, humanOutput);
162                 sb.append("</table>");
163                 makeEOL(sb, humanOutput);
164                 makeLevelSpaces(sb, level, humanOutput);
165                 sb.append("</td></tr>");
166                 makeEOL(sb, humanOutput);
167             } else if ( ConstructorUtils.isClassCollection(type) ) {
168                 // COLLECTION
169                 tagName = validate(tagName == null ? "collection" : tagName);
170                 Collection<Object> collection = (Collection) object;
171                 makeLevelSpaces(sb, level, humanOutput);
172                 sb.append("<tr><td width='3%'>");
173                 sb.append(tagName);
174                 sb.append(" type=collection");
175                 sb.append(" size="+collection.size());
176                 sb.append("</td><td>");
177                 makeEOL(sb, humanOutput);
178                 makeLevelSpaces(sb, level+1, humanOutput);
179                 sb.append("<table border='1'>");
180                 makeEOL(sb, humanOutput);
181                 for (Object element : collection) {
182                     Class<?> elementType = null;
183                     if (element != null) {
184                         elementType = element.getClass();
185                     }
186                     sb.append( toHTML(element, makeElementName(elementType), level+2, humanOutput, includeNulls, includeClassField, properties) );
187                 }
188                 makeLevelSpaces(sb, level+1, humanOutput);
189                 sb.append("</table>");
190                 makeEOL(sb, humanOutput);
191                 makeLevelSpaces(sb, level, humanOutput);
192                 sb.append("</td></tr>");
193                 makeEOL(sb, humanOutput);
194             } else {
195                 // must be a bean or map, make sure it is a map
196                 tagName = validate(tagName == null ? makeElementName(type) : tagName);
197                 String xmlType = "bean";
198                 Map<String, Object> map = null;
199                 if (Map.class.isAssignableFrom(type)) {
200                     xmlType = "map";
201                     map = (Map<String, Object>) object;
202                 } else {
203                     // reflect over objects
204                     map = ReflectUtils.getInstance().getObjectValues(object, FieldsFilter.SERIALIZABLE, includeClassField);
205                 }
206                 makeLevelSpaces(sb, level, humanOutput);
207                 sb.append("<tr><td width='3%'>");
208                 sb.append(tagName);
209                 sb.append(" type="+xmlType);
210                 sb.append(" size="+map.size());
211                 sb.append("</td><td>");
212                 makeEOL(sb, humanOutput);
213                 makeLevelSpaces(sb, level+1, humanOutput);
214                 sb.append("<table border='1'>");
215                 makeEOL(sb, humanOutput);
216                 for (Entry<String, Object> entry : map.entrySet()) {
217                     if (entry.getKey() != null) {
218                         sb.append( toHTML(entry.getValue(), entry.getKey().toString(), level+2, humanOutput, includeNulls, includeClassField, properties) );
219                     }
220                 }
221                 makeLevelSpaces(sb, level+1, humanOutput);
222                 sb.append("</table>");
223                 makeEOL(sb, humanOutput);
224                 makeLevelSpaces(sb, level, humanOutput);
225                 sb.append("</td></tr>");
226                 makeEOL(sb, humanOutput);
227             }
228         }
229         return sb.toString();
230     }

232     protected static String makeElementName(Class<?> type) {
233         String name = "element";
234         if (type != null) {
235             if (! Map.class.isAssignableFrom(type)) {
236                 name = type.getSimpleName();
237             }
238         }
239         return name;
240     }

248     protected static final String SPACES = "  ";
249     protected static void makeLevelSpaces(StringBuilder sb, int level, boolean includeEOL) {
250         level++;
251         if (includeEOL) {
252             for (int i = 0; i < level; i++) {
253                 sb.append(SPACES);
254             }
255         }

258     /***
259      * Escape a string for XML encoding: replace special characters with XML escapes:
260      * <pre>
261      * &amp; <small>(ampersand)</small> is replaced by &amp;amp;
262      * &lt; <small>(less than)</small> is replaced by &amp;lt;
263      * &gt; <small>(greater than)</small> is replaced by &amp;gt;
264      * &quot; <small>(double quote)</small> is replaced by &amp;quot;
265      * </pre>
266      * @param string The string to be escaped.
267      * @return The escaped string.
268      */
269     public static String escapeForXML(String string) {
270         StringBuilder sb = new StringBuilder();
271         for (int i = 0, len = string.length(); i < len; i++) {
272             char c = string.charAt(i);
273             switch (c) {
274             case AMP:
275                 sb.append("&amp;");
276                 break;
277             case LT:
278                 sb.append("&lt;");
279                 break;
280             case GT:
281                 sb.append("&gt;");
282                 break;
283             case QUOT:
284                 sb.append("&quot;");
285                 break;
286             default:
287                 sb.append(c);
288             }
289         }
290         return sb.toString();
291     }

293     /***
294      * Validates that a string contains no spaces and is non-null/non-empty
295      * Throw an exception if the string contains whitespace. 
296      * Whitespace is not allowed in tagNames and attributes.
297      * @param string any string
298      * @throws IllegalArgumentException
299      */
300     public static String validate(String string) {
301         if (string == null) {
302             throw new IllegalArgumentException("string is NULL");
303         }
304         int i, length = string.length();
305         if (length == 0) {
306             throw new IllegalArgumentException("Empty string.");
307         }
308         for (i = 0; i < length; i += 1) {
309             if (Character.isWhitespace(string.charAt(i))) {
310                 throw new IllegalArgumentException("'" + string + "' contains a space character.");
311             }
312         }
313         return string;
314     }
