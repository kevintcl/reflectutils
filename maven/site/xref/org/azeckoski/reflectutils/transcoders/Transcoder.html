19  import org.azeckoski.reflectutils.ReflectUtils;

22  /***
23   * The transcoder can convert from a java objects => a format => java objects (simple) <br/>
24   * Note that conversion to simple objects is all that is generally supported,
25   * conversions to complex objects require use of the {@link ReflectUtils} to handle
26   * the conversion using {@link ReflectUtils#populate(Object, Map)} or {@link ReflectUtils#convert(Object, Class)}
27   * <br/>
28   * The ability to append optional properties is also supported <br/>
29   * The various transcoders will also have optional configuration parameters which should be controlled
30   * via the constructors. <br/>
31   * Transcoders should be written to be created once and used many times without problems.
32   * 
33   * @author Aaron Zeckoski (azeckoski @ gmail.com)
34   */
35  public interface Transcoder {
36  
37      /***
38       * default map key to use if the output data is not actually a map but is a list or array or whatever
39       * when it is being decoded into a map
40       */
41      public static final String DATA_KEY = "data";
42  
43      /***
44       * @return the handled format (xml, json, etc.)
45       */
46      public String getHandledFormat();
47  
48      /***
49       * @param object any java object (should not be null)
50       * @param name (optional) the name of the data, will be handled differently by the encoders (null for default: {@link #DATA_KEY})
51       * @param properties (optional) additional properties which will be added to the encoding 
52       * as if it were a property of the bean or map being encoded, if the object being encoded is not a bean
53       * or a map then the properties are ignored
54       * @return the object encoded into the handled format
55       */
56      public String encode(Object object, String name, Map<String, Object> properties);
57  
58      /***
59       * Decode the data string into a map of java objects (typically simple objects like String, Integer, etc.)
60       * @param string a string in the handled format
61       * @return the map of java objects based on the data string, this will not attempt to convert XML into beans
62       * but will simply convert data into maps or lists or simple java objects (strings, etc.), if there
63       * is only a single simple value then it will be placed into the map with the key {@value #DATA_KEY}
64       */
65      public Map<String, Object> decode(String string);
66  
67  }
