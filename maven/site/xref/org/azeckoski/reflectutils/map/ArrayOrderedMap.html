29  /***
30   * A map which keeps track of the order the entries are added 
31   * and allows retrieval of entries in the order they were entered as well,
32   * this is NOT safe for multi-threaded access,
33   * this is backed by a {@link HashMap} and {@link ArrayList}
34   * 
35   * @author Aaron Zeckoski (azeckoski @ gmail.com)
36   */
37  public class ArrayOrderedMap<K, V> extends HashMap<K, V> implements OrderedMap<K, V> {
38      public static final long serialVersionUID = 1l;
39  
40      private ArrayList<K> list;
41  
42      public ArrayOrderedMap() {
43          this(10);
44      }
45  
46      public ArrayOrderedMap(int initialCapacity) {
47          super(initialCapacity);
48          list = new ArrayList<K>(initialCapacity);
49      }
50  
51      public ArrayOrderedMap(Map<K, V> map) {
52          this(map.size());
53          for (Entry<K, V> entry : map.entrySet()) {
54              this.put(entry.getKey(), entry.getValue());
55          }
56      }

84      /***
85       * @return a list of all the entries in this map in the order they were created
86       */
87      public List<Entry<K, V>> getEntries() {
88          ArrayList<Entry<K, V>> entries = new ArrayList<Entry<K,V>>();
89          for (K key : list) {
90              Entry<K, V> entry = new SimpleEntry<K,V>(key, this.get(key));
91              entries.add(entry);
92          }
93          return entries;
94      }

96      /***
97       * Get an entry based on the position it is in the map (based on the order entries were created)
98       * @param position the position in the map (must be less that the size)
99       * @return the entry at that position
100      * @throws IllegalArgumentException if the position is greater than the map size
101      */
102     public Entry<K, V> getEntry(int position) {
103         if (position >= list.size()) {
104             throw new IllegalArgumentException("Value is too large for the map size: " + list.size());
105         }
106         K key = list.get(position);
107         Entry<K, V> entry = new SimpleEntry<K,V>(key, this.get(key));
108         return entry;
109     }

111     @Override
112     public V put(K key, V value) {
113         V v = super.put(key, value);
114         if (v != null) {
115             // displaced
116             list.remove(key);
117         }
118         list.add(key);
119         return v;
120     }
121 
122     @Override
123     public V remove(Object key) {
124         V v = super.remove(key);
125         if (v != null) {
126             list.remove(key);
127         }
128         return v;
129     }

170     abstract class CoreIterator {
171 
172         int currentPos = -1;
173         Entry<K, V> lastReturned = null;
174         private List<Entry<K, V>> entries = ArrayOrderedMap.this.getEntries();
175 
176         public boolean hasMore() {
177             if ((currentPos + 1) < entries.size()) {
178                 return true;
179             }
180             return false;
181         }
182 
183         public Entry<K, V> getNext() {
184             currentPos++;
185             try {
186                 lastReturned = entries.get(currentPos);
187             } catch (RuntimeException e) {
188                 throw new NoSuchElementException("There are no more items available to get, the last one was reached");
189             }
190             return lastReturned;
191         }
192 
193         public void removeCurrent() {
194             if (currentPos < 0) {
195                 throw new IllegalArgumentException("Have not called next yet, cannot remove from this iterator");
196             }
197             entries.remove(currentPos);
198             ArrayOrderedMap.this.remove(lastReturned.getKey());
199         }
200 
201         // shared methods
202         public boolean hasNext() {          return hasMore(); }
203         public boolean hasMoreElements() {  return hasMore(); }
204         public void remove() {              removeCurrent(); }
205 
206     }

282     @SuppressWarnings("unchecked")
283     final class EntrySet extends AbstractSet<Map.Entry<K,V>> {
284         public Iterator<Map.Entry<K,V>> iterator() {
285             return new EntryIterator();
286         }
287         public boolean contains(Object o) {
288             if (!(o instanceof Map.Entry))
289                 return false;
290             Map.Entry<K,V> e = (Map.Entry<K,V>)o;
291             V v = ArrayOrderedMap.this.get(e.getKey());
292             return v != null && v.equals(e.getValue());
293         }
294         public boolean remove(Object o) {
295             if (!(o instanceof Map.Entry))
296                 return false;
297             Map.Entry<K,V> e = (Map.Entry<K,V>)o;
298             return ArrayOrderedMap.this.remove(e.getKey()) != null;
299         }
300         public int size() {
301             return ArrayOrderedMap.this.size();
302         }
303         public void clear() {
304             ArrayOrderedMap.this.clear();
305         }
306         public Object[] toArray() {
307             // Since we don't ordinarily have distinct Entry objects, we
308             // must pack elements using exportable SimpleEntry
309             Collection<Map.Entry<K,V>> c = new ArrayList<Map.Entry<K,V>>(size());
310             for (Iterator<Map.Entry<K,V>> i = iterator(); i.hasNext(); )
311                 c.add(new SimpleEntry<K,V>(i.next()));
312             return c.toArray();
313         }
314         public <T> T[] toArray(T[] a) {
315             Collection<Map.Entry<K,V>> c = new ArrayList<Map.Entry<K,V>>(size());
316             for (Iterator<Map.Entry<K,V>> i = iterator(); i.hasNext(); )
317                 c.add(new SimpleEntry<K,V>(i.next()));
318             return c.toArray(a);
319         }

324     /***
325      * This duplicates java.util.AbstractMap.SimpleEntry until this class
326      * is made accessible.
327      */
328     static final class SimpleEntry<K,V> implements Entry<K,V> {
329         K key;
330         V value;
331 
332         public SimpleEntry(K key, V value) {
333             this.key   = key;
334             this.value = value;
335         }
336 
337         public SimpleEntry(Entry<K,V> e) {
338             this.key   = e.getKey();
339             this.value = e.getValue();
340         }
341 
342         public K getKey() {
343             return key;
344         }
345 
346         public V getValue() {
347             return value;
348         }
349 
350         public V setValue(V value) {
351             V oldValue = this.value;
352             this.value = value;
353             return oldValue;
354         }
355 
356         @SuppressWarnings("unchecked")
357         public boolean equals(Object o) {
358             if (!(o instanceof Map.Entry))
359                 return false;
360             Map.Entry e = (Map.Entry)o;
361             return eq(key, e.getKey()) && eq(value, e.getValue());
362         }
363 
364         public int hashCode() {
365             return ((key   == null)   ? 0 :   key.hashCode()) ^
366             ((value == null)   ? 0 : value.hashCode());
367         }
368 
369         public String toString() {
370             return key + "=" + value;
371         }
372 
373         static boolean eq(Object o1, Object o2) {
374             return (o1 == null ? o2 == null : o1.equals(o2));
375         }
376     }
