21  /***
22   * Utilities to help with operations that deal in {@link ClassLoader} and {@link Class}
23   * 
24   * @author Aaron Zeckoski (azeckoski@gmail.com)
25   */
26  public class ClassLoaderUtils {
27  
28      /***
29       * @param className a fully qualified class name (e.g. org.dspace.MyClass)
30       * @return the Class if it can be found in the context or current {@link ClassLoader} OR null if no class can be found
31       */
32      public static Class<?> getClassFromString(String className) {
33          Class<?> c = null;
34          try {
35              ClassLoader cl = getCurrentClassLoader();
36              c = Class.forName(className, true, cl);
37          } catch (ClassNotFoundException e) {
38              try {
39                  ClassLoader cl = ClassLoaderUtils.class.getClassLoader();
40                  c = Class.forName(className, true, cl);
41              } catch (ClassNotFoundException e1) {
42                  try {
43                      c = Class.forName(className);
44                  } catch (ClassNotFoundException e2) {
45                      c = null;
46                  }
47              }
48          }
49          return c;
50      }
51  
52      /***
53       * @return the current context {@link ClassLoader} or the nearest thing that can be found
54       */
55      public static ClassLoader getCurrentClassLoader() {
56          ClassLoader cl = Thread.currentThread().getContextClassLoader();
57          if (cl == null) {
58              cl = ClassLoaderUtils.class.getClassLoader();
59          }
60          return cl;
61      }

64      /***
65       * Returns a list of all superclasses and implemented interfaces by the supplied class,
66       * recursively to the base, up to but excluding Object.class. These will be listed in order from
67       * the supplied class, all concrete superclasses in ascending order, and then finally all
68       * interfaces in recursive ascending order.<br/>
69       * This will include duplicates if any superclasses implement the same classes 
70       * 
71       * Taken from PonderUtilCore around version 1.2.2
72       *  - Antranig Basman (antranig@caret.cam.ac.uk)
73       */
74      public static List<Class<?>> getSuperclasses(Class<?> clazz) {
75          List<Class<?>> accumulate = new ArrayList<Class<?>>();
76          while (clazz != Object.class) {
77              accumulate.add(clazz);
78              clazz = clazz.getSuperclass();
79          }
80          int supers = accumulate.size();
81          for (int i = 0; i < supers; ++i) {
82              appendSuperclasses(accumulate.get(i), accumulate);
83          }
84          return accumulate;
85      }
86  
87      /***
88       * Taken from PonderUtilCore around version 1.2.2
89       *  - Antranig Basman (antranig@caret.cam.ac.uk)
90       */
91      private static void appendSuperclasses(Class<?> clazz, List<Class<?>> accrete) {
92          Class<?>[] interfaces = clazz.getInterfaces();
93          for (int i = 0; i < interfaces.length; ++i) {
94              accrete.add(interfaces[i]);
95          }
96          for (int i = 0; i < interfaces.length; ++i) {
97              appendSuperclasses(interfaces[i], accrete);
98          }
99      }

102     /***
103      * Finds a class type that is in the containing collection,
104      * will always return something (failsafe to Object.class)
105      * @param collection any collection (List, Set, etc.)
106      * @return the class type contained in this collecion
107      */
108     @SuppressWarnings("unchecked")
109     public static Class<?> getClassFromCollection(Collection collection) {
110         // try to get the type of entities out of this collection
111         Class<?> c = Object.class;
112         if (collection != null) {
113             if (! collection.isEmpty()) {
114                 c = collection.iterator().next().getClass();
115             } else {
116                 // this always gets Object.class -AZ
117                 //c = collection.toArray().getClass().getComponentType();
118                 c = Object.class;
119             }
120         }
121         return c;
122     }
