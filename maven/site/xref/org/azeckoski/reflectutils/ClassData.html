31  /***
32   * @param <T> the class type
33   * A class which retrieves and holds all the reflected class data related to a class 
34   * (annotations, constructors, fields, methods), this is immutable<br/>
35   * WARNING: this is NOT cheap and the results need to be cached badly,
36   * you should get this from the {@link ClassDataCacher} rather than constructing it yourself<br/>
37   */
38  public class ClassData<T> {
39  
40      private final Class<T> type;
41      private final List<Annotation> annotations;
42      private final List<Constructor<T>> constructors;
43      private final List<Field> fields;
44      private final List<Method> methods;
45      private final List<Class<?>> interfaces;
46      private final List<Class<?>> superclasses;
47      private final List<T> enumConstants;
48  
49      public ClassData(Class<T> type) {
50          if (type == null) {
51              throw new IllegalArgumentException("ClassData requires a valid class, type is null");
52          }
53          if (type.isPrimitive() || type.isArray()) {
54              throw new IllegalArgumentException("Invalid type to make reflection cache for ("+type.getName()+"), cannot reflect over primitives or arrays");
55          }
56          this.type = type;
57          T[] ecs = type.getEnumConstants();
58          if (ecs != null) {
59              enumConstants = new ArrayList<T>(ecs.length);
60              for (T t : ecs) {
61                  enumConstants.add(t);
62              }
63          } else {
64              enumConstants = new ArrayList<T>(0);
65          }
66          annotations = new ArrayList<Annotation>(5);
67          getAllAnnotations(type, annotations);
68          constructors = new ArrayList<Constructor<T>>(5);
69          fields = new ArrayList<Field>(5);
70          methods = new ArrayList<Method>(5);
71          interfaces = new ArrayList<Class<?>>(5);
72          superclasses = new ArrayList<Class<?>>(5);
73          getAllThings(type, fields, methods, constructors, interfaces, superclasses);
74          // sort the fields,methods,constructors
75          Collections.sort(fields, new MemberComparator());
76          Collections.sort(methods, new MemberComparator());
77          Collections.sort(constructors, new MemberComparator());
78          // remove duplicates from the list of interfaces
79          ArrayUtils.removeDuplicates(interfaces);
80      }

82      /***
83       * @return only the public constructors in the class this data represents
84       */
85      public List<Constructor<T>> getPublicConstructors() {
86          ArrayList<Constructor<T>> pubConstructors = new ArrayList<Constructor<T>>();
87          for (Constructor<T> constructor : constructors) {
88              if (Modifier.isPublic(constructor.getModifiers())) {
89                  pubConstructors.add(constructor);
90              }
91          }
92          return pubConstructors;
93      }

121     protected void getAllAnnotations(final Class<?> type, final List<Annotation> list) {
122         // get only public annotations
123         Annotation[] annotations = type.getAnnotations();
124         for (Annotation annotation : annotations) {
125             if (annotation != null) {
126                 list.add(annotation);
127             }
128         }
129     }

131     @SuppressWarnings("unchecked")
132     protected void getAllThings(final Class<?> type, final List<Field> fList, final List<Method> mList, final List<Constructor<T>> cList, final List<Class<?>> iList, final List<Class<?>> sList) {
133         // get the fields for the current class
134         Field[] fields = type.getDeclaredFields();
135         for (final Field field : fields) {
136             if (field != null) {
137                 int modifiers = field.getModifiers();
138                 if (Modifier.isPublic(modifiers)) {
139                     fList.add(field);
140                 } else {
141                     try {
142                         AccessController.doPrivileged(new PrivilegedAction<T>() {
143                             public T run() {
144                                 field.setAccessible(true);
145                                 return null;
146                             }
147                         });
148                         fList.add(field);
149                     } catch (SecurityException e) {
150                         // oh well, this does not get added then
151                     }
152                 }
153             }
154         }
155         Method[] methods = type.getDeclaredMethods();
156         for (final Method method : methods) {
157             if (method != null) {
158                 int modifiers = method.getModifiers();
159                 if (Modifier.isPublic(modifiers)) {
160                     mList.add(method);
161                 } else {
162                     try {
163                         AccessController.doPrivileged(new PrivilegedAction<T>() {
164                             public T run() {
165                                 method.setAccessible(true);
166                                 return null;
167                             }
168                         });
169                         mList.add(method);
170                     } catch (SecurityException e) {
171                         // oh well, this does not get added then
172                     }
173                 }
174             }
175         }
176         Constructor<T>[] constructors = type.getDeclaredConstructors();
177         for (final Constructor<T> constructor : constructors) {
178             // need to avoid the Object constructor
179             if (!Object.class.equals(type) && constructor != null) {
180                 int modifiers = constructor.getModifiers();
181                 if (Modifier.isPublic(modifiers)) {
182                     cList.add(constructor);
183                 } else {
184                     try {
185                         AccessController.doPrivileged(new PrivilegedAction<T>() {
186                             public T run() {
187                                 constructor.setAccessible(true);
188                                 return null;
189                             }
190                         });
191                         cList.add(constructor);
192                     } catch (SecurityException e) {
193                         // oh well, this does not get added then
194                     }
195                 }
196             }
197         }
198         // now we recursively go through the interfaces and super classes
199         Class<?>[] interfaces = type.getInterfaces();
200         for (Class<?> iface : interfaces) {
201             iList.add(iface); // add to the interfaces list
202         }
203         for (int i = 0; i < interfaces.length; i++) {
204             getAllThings(interfaces[i], fList, mList, cList, iList, sList);
205         }
206         Class<?> superClass = type.getSuperclass();
207         if (superClass != null) {
208             sList.add(superClass); // add to superclasses list
209             getAllThings(superClass, fList, mList, cList, iList, sList);
210         }
211     }

269     /***
270      * Sorts the members by visibility and name order
271      */
272     public static final class MemberComparator implements Comparator<Member>, Serializable {
273         public static final long serialVersionUID = 1l;
274         public int compare(Member o1, Member o2) {
275             String c1 = getModifierPrefix(o1.getModifiers()) + o1.getName();
276             String c2 = getModifierPrefix(o2.getModifiers()) + o2.getName();
277             return c1.compareTo(c2);
278         }
279     }
280 
281     public static final String getModifierPrefix(int modifier) {
282         String prefix = "0public-";
283         if (Modifier.isProtected(modifier)) {
284             prefix = "1protected-";
285         } else if (Modifier.isPrivate(modifier)) {
286             prefix = "2private-";
287         }
288         return prefix;
289     }
