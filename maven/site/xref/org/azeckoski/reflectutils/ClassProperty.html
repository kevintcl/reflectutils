31  /***
32   * Simple class for holding the values we care about for a single field,
33   * if it uses getters and setters then they must be like this:<br/>
34   * If the fieldName is "thing" then: <br/>

59      private OrderedMap<Class<? extends Annotation>, Annotation> propertyAnnotations; // contains all annotations on this property
60      protected void addAnnotation(Annotation annotation) {
61          if (annotation != null) {
62              if (propertyAnnotations == null) {
63                  propertyAnnotations = new ArrayOrderedMap<Class<? extends Annotation>, Annotation>();
64              }
65              Class<? extends Annotation> c = annotation.annotationType();
66              if (! propertyAnnotations.containsKey(c)) {
67                  // only add an annotation in the first time it is encountered
68                  propertyAnnotations.put(c, annotation);
69              }
70              if (ReflectTransient.class.equals(c)) {
71                  transientField = true; // pretend to be transient
72              }
73          }
74      }
75      protected Collection<Annotation> getAnnotationsCollection() {
76          Collection<Annotation> ans = null;
77          if (propertyAnnotations == null 
78                  || propertyAnnotations.isEmpty()) {
79              ans = new ArrayList<Annotation>();
80          } else {
81              ans = propertyAnnotations.values();
82          }
83          return ans;
84      }
85      /***
86       * @param annotationType the annotation type to look for on this field
87       * @return the annotation of this type for this field OR null if none found
88       */
89      public Annotation getAnnotation(Class<? extends Annotation> annotationType) {
90          Annotation a = null;
91          if (propertyAnnotations != null) {
92              a = propertyAnnotations.get(annotationType);
93          }
94          return a;
95      }
96      /***
97       * @return the annotations present on the property
98       */
99      public Annotation[] getAnnotations() {
100         Collection<Annotation> c = getAnnotationsCollection();
101         return c.toArray(new Annotation[c.size()]);
102     }

271     /***
272      * Sets the fieldModifiers based on this field,
273      * will not reset them once they are set except to clear them if a null field is set
274      */
275     protected void setModifiers(Field field) {
276         if (field == null) {
277             fieldModifiers = -1;
278             transientField = false;
279             finalField = false;
280             publicField = false;
281             staticField = false;
282         } else {
283             if (fieldModifiers < 0) {
284                 fieldModifiers = field.getModifiers();
285                 transientField = Modifier.isTransient(fieldModifiers);
286                 finalField = Modifier.isFinal(fieldModifiers);
287                 publicField = Modifier.isPublic(fieldModifiers);
288                 staticField = Modifier.isStatic(fieldModifiers);
289             }
290         }
291     }
292     private void makeType() {
293         if (getGetter() != null) {
294             Method m = getGetter();
295             setType( m.getReturnType() );
296         } else if (getSetter() != null) {
297             Method m = getSetter();
298             Class<?>[] params = m.getParameterTypes();
299             if (params != null) {
300                 if (params.length == 1) {
301                     setType( params[0] ); // normal setter
302                 } else if (params.length == 2) {
303                     setType( params[1] ); // indexed/mapped setter
304                 }
305             }
306         } else {
307             Field f = getField();
308             setType( f.getType() );
309         }
310         indexed = false;
311         arrayed = false;
312         mapped = false;
313         Class<?> type = getType();
314         if (type != null) {
315             if (type.isArray()) {
316                 indexed = true;
317                 arrayed = true;
318             }
319             if (Map.class.isAssignableFrom(type)) {
320                 mapped = true;
321             }
322             if (List.class.isAssignableFrom(type)) {
323                 indexed = true;
324             }
325         }

328     public String toString() {
329         return fieldName + "(" + (type == null ? "" : type.getSimpleName()) + ")["
330         +(field == null?"-":"F")+(getter == null?"-":"G")+(setter == null?"-":"S")+":"
331         +(mapped?"M":"-")+(indexed?"I":"-")+(arrayed?"A":"-")+":"
332         +(finalField?"F":"-")+(transientField?"T":"-")+(publicField?"P":"-")+(staticField?"S":"-")+"]";
333     }
