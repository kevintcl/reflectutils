31  /***
32   * Simple class for holding the values we care about for a single field,
33   * if it uses getters and setters then they must be like this:<br/>
34   * If the fieldName is "thing" then: <br/>

59      private OrderedMap<Class<? extends Annotation>, Annotation> propertyAnnotations; // contains all annotations on this property
60      protected void addAnnotation(Annotation annotation) {
61          if (annotation != null) {
62              if (propertyAnnotations == null) {
63                  propertyAnnotations = new ArrayOrderedMap<Class<? extends Annotation>, Annotation>();
64              }
65              Class<? extends Annotation> c = annotation.annotationType();
66              if (! propertyAnnotations.containsKey(c)) {
67                  // only add an annotation in the first time it is encountered
68                  propertyAnnotations.put(c, annotation);
69              }
70              // note that we compare the simple name to avoid issues with cross classloader types
71              if (ReflectTransient.class.getSimpleName().equals(c.getSimpleName())) {
72                  transientField = true; // pretend to be transient
73              }
74          }
75      }
76      protected Collection<Annotation> getAnnotationsCollection() {
77          Collection<Annotation> ans = null;
78          if (propertyAnnotations == null 
79                  || propertyAnnotations.isEmpty()) {
80              ans = new ArrayList<Annotation>();
81          } else {
82              ans = propertyAnnotations.values();
83          }
84          return ans;
85      }
86      /***
87       * @param annotationType the annotation type to look for on this field
88       * @return the annotation of this type for this field OR null if none found
89       */
90      public Annotation getAnnotation(Class<? extends Annotation> annotationType) {
91          Annotation a = null;
92          if (propertyAnnotations != null) {
93              a = propertyAnnotations.get(annotationType);
94          }
95          return a;
96      }

98       * @return the annotations present on the property
99       */
100     public Annotation[] getAnnotations() {
101         Collection<Annotation> c = getAnnotationsCollection();
102         return c.toArray(new Annotation[c.size()]);
103     }

272     /***
273      * Sets the fieldModifiers based on this field,
274      * will not reset them once they are set except to clear them if a null field is set
275      */
276     protected void setModifiers(Field field) {
277         if (field == null) {
278             fieldModifiers = -1;
279             transientField = false;
280             finalField = false;
281             publicField = false;
282             staticField = false;
283         } else {
284             if (fieldModifiers < 0) {
285                 fieldModifiers = field.getModifiers();
286                 transientField = Modifier.isTransient(fieldModifiers);
287                 finalField = Modifier.isFinal(fieldModifiers);
288                 publicField = Modifier.isPublic(fieldModifiers);
289                 staticField = Modifier.isStatic(fieldModifiers);
290             }
291         }

293     private void makeType() {
294         if (getGetter() != null) {
295             Method m = getGetter();
296             setType( m.getReturnType() );
297         } else if (getSetter() != null) {
298             Method m = getSetter();
299             Class<?>[] params = m.getParameterTypes();
300             if (params != null) {
301                 if (params.length == 1) {
302                     setType( params[0] ); // normal setter
303                 } else if (params.length == 2) {
304                     setType( params[1] ); // indexed/mapped setter
305                 }
306             }
307         } else {
308             Field f = getField();
309             setType( f.getType() );
310         }
311         indexed = false;
312         arrayed = false;
313         mapped = false;
314         Class<?> type = getType();
315         if (type != null) {
316             if (type.isArray()) {
317                 indexed = true;
318                 arrayed = true;
319             }
320             if (Map.class.isAssignableFrom(type)) {
321                 mapped = true;
322             }
323             if (List.class.isAssignableFrom(type)) {
324                 indexed = true;
325             }
326         }

329     public String toString() {
330         return fieldName + "(" + (type == null ? "" : type.getSimpleName()) + ")["
331         +(field == null?"-":"F")+(getter == null?"-":"G")+(setter == null?"-":"S")+":"
332         +(mapped?"M":"-")+(indexed?"I":"-")+(arrayed?"A":"-")+":"
333         +(finalField?"F":"-")+(transientField?"T":"-")+(publicField?"P":"-")+(staticField?"S":"-")+"]";
334     }
