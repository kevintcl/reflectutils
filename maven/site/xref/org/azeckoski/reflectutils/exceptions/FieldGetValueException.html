18  /***
19   * Indicates that there was a failure attempting to get a field value,
20   * probably caused by a security exception or obscure java failure reading or writing the value
21   * (this indicates the arguments were correct and the field was found but there was a failure at the time of the operation)
22   * 
23   * @author Aaron Zeckoski (azeckoski @ gmail.com)
24   */
25  public class FieldGetValueException extends RuntimeException {
26  
27     /***
28      * the field name we were trying to get the value from 
29      */
30     public String fieldName;
31     /***
32      * the object with the field we were trying to get a value from
33      */
34     public Object object;
35  
36     public FieldGetValueException(String fieldName, Object object, Throwable cause) {
37        super("Failed to get field ("+fieldName+") value from object ("+object+"), cause=" + cause, cause);
38        this.fieldName = fieldName;
39        this.object = object;
40     }
41     public FieldGetValueException(String message, String fieldName, Object object, Throwable cause) {
42        super(message, cause);
43        this.fieldName = fieldName;
44        this.object = object;
45     }
46     
47  }
