15  package org.azeckoski.reflectutils.exceptions;
16  
17  /***
18   * Indicates that the fieldname could not be found
19   * 
20   * @author Aaron Zeckoski (azeckoski@gmail.com)
21   */
22  public class FieldnameNotFoundException extends RuntimeException {
23  
24     public String fieldName;
25  
26     public FieldnameNotFoundException(String fieldName) {
27        this(fieldName, null);
28     }
29  
30     public FieldnameNotFoundException(String fieldName, Throwable cause) {
31        this("Could not find fieldName ("+fieldName+") on object", fieldName, cause);
32     }
33  
34     public FieldnameNotFoundException(String message, String fieldName, Throwable cause) {
35        super(message, cause);
36        this.fieldName = fieldName;
37     }
38  
39  }
