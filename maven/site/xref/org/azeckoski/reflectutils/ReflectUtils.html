31  /***
32   * Reflection utilities and utilities related to working with classes and their fields<br/>
33   * <br/>
34   * These are built to be compatible with <a href="http://commons.apache.org/beanutils/">Apache Commons BeanUtils</a>
35   * and the nesting structure works the same, refer to the apache BeanUtils project docs for details<br/>
36   * <br/>
37   * Handles field operations for properties (getters and setters), partial properties (only getter or only setter), and fields.
38   * This improves upon the BeanUtils limitation of handling only properties or the Google utilities limitation of handling
39   * only fields.<br/>
40   * <br/>
41   * Getting and setting fields supports simple, nested, indexed, and mapped values<br/>
42   * <b>Simple:</b> Get/set a field in a bean (or map), Example: "title", "id"<br/>
43   * <b>Nested:</b> Get/set a field in a bean which is contained in another bean, Example: "someBean.title", "someBean.id"<br/>
44   * <b>Indexed:</b> Get/set a list/array item by index in a bean, Example: "myList[1]", "anArray[2]"<br/>
45   * <b>Mapped:</b> Get/set a map entry by key in a bean, Example: "myMap(key)", "someMap(thing)"<br/>
46   * <br/>
47   * Includes support for dealing with annotations and working with field which have annotations on them. Methods
48   * for finding fields with an annotation and finding all annotations in a class or on a fields are included.
49   * <br/>
50   * Includes support for deep cloning, deep copying, and populating objects using auto-conversion. Also
51   * includes support for fuzzy copies where object data can be copied from one object to another without
52   * the objects being the same type.<br/>
53   * <br/>
54   * Also includes an extendable conversion system for converting between java types. This system also handles
55   * conversions between arrays, maps, collections, and scalars and improves upon the apache system by
56   * handling more types and handling object holders.<br/>
57   * <br/>
58   * Support for construction of any class and a set of utilities for determining what types of objects
59   * you are working with are also included. A method for executing a specific constructor can be used if more control if needed.
60   * <br/>
61   * The utilities cache reflection data for high performance operation but uses weak/soft caching to avoid holding open
62   * ClassLoaders and causing the caches to exist in memory permanently. The ability to override the caching mechanism
63   * with your own is supported.<br/>
64   * <br/>
65   * The utilities are modular and are meant to be extendable and overridable. All methods are protected or public so
66   * that the various utility classes can be easily overridden if needed.
67   * 
68   * @author Aaron Zeckoski (azeckoski@gmail.com)
69   */

217     /***
218      * Get the value of a field or getter method from an object allowing an annotation to override<br/>
219      * Getting fields supports simple, nested, indexed, and mapped values:<br/>
220      * <b>Simple:</b> Get/set a field in a bean (or map), Example: "title", "id"<br/>
221      * <b>Nested:</b> Get/set a field in a bean which is contained in another bean, Example: "someBean.title", "someBean.id"<br/>
222      * <b>Indexed:</b> Get/set a list/array item by index in a bean, Example: "myList[1]", "anArray[2]"<br/>
223      * <b>Mapped:</b> Get/set a map entry by key in a bean, Example: "myMap(key)", "someMap(thing)"<br/>
224      * 
225      * @param object any object
226      * @param fieldName the name of the field (property) to get the value of or the getter method without the "get" and lowercase first char
227      * @param annotationClass if the annotation class is set then we will attempt to get the value from the annotated field or getter method first
228      * @return the value of the field OR null if the value is null
229      * @throws FieldnameNotFoundException if this fieldName does not exist on the object and no annotation was found
230      * @throws IllegalArgumentException if a failure occurs while getting the field value
231      */
232     public Object getFieldValue(Object object, String fieldName, Class<? extends Annotation> annotationClass) {
233         Object value = null;
234         Class<?> elementClass = object.getClass();
235         if (annotationClass != null) {
236             // try to find annotation first
237             String annotatedField = getFieldNameWithAnnotation(elementClass, annotationClass);
238             if (annotatedField != null) {
239                 fieldName = annotatedField;
240             }
241         }
242         value = getFieldValue(object, fieldName);
243         return value;
244     }

447      * @param ignoreTransient if true then all transient fields will be skipped, useful when serializing
448      * @param initialKey (optional) the initial key to use when converting simpler objects to maps

600     /***
601      * @param text string to make MD5 hash from
602      * @param maxLength
603      * @return an MD5 hash no longer than maxLength
604      */
605     public static String makeMD5(String text, int maxLength) {
606         MessageDigest md;
607         try {
608             md = MessageDigest.getInstance("MD5");
609         } catch (NoSuchAlgorithmException e) {
610             throw new RuntimeException("Stupid java sucks for MD5", e);
611         }
612         md.update(text.getBytes());
613 
614         // convert the binary md5 hash into hex
615         StringBuilder sb = new StringBuilder();
616         byte[] b_arr = md.digest();
617 
618         for (int i = 0; i < b_arr.length; i++) {
619             // convert the high nibble
620             byte b = b_arr[i];
621             b >>>= 4;
622             b &= 0x0f; // this clears the top half of the byte (intentional)
623             sb.append( Integer.toHexString(b) );
624 
625             // convert the low nibble
626             b = b_arr[i];
627             b &= 0x0F;
628             sb.append( Integer.toHexString(b) );
629         }
630         String md5 = sb.toString();
631         if (maxLength > 0 && md5.length() > maxLength) {
632             md5 = md5.substring(0, maxLength);
633         }
634         return md5;
635     }

637     /***
638      * Finds a map value for a key (or set of keys) if it exists in the map and returns the string value of it
639      * @param map any map with strings as keys
640      * @param keys an array of keys to try to find in order
641      * @return the string value OR null if it could not be found for any of the given keys
642      */
643     public static String findMapStringValue(Map<String, ?> map, String[] keys) {
644         if (map == null || keys == null) {
645             return null;
646         }
647         String value = null;
648         try {
649             for (int i = 0; i < keys.length; i++) {
650                 String key = keys[i];
651                 if (map.containsKey(key)) {
652                     Object oVal = map.get(key);
653                     if (oVal != null) {
654                         value = oVal.toString();
655                         break;
656                     }
657                 }
658             }
659         } catch (RuntimeException e) {
660             // in case the given map is not actually of the right types at runtime
661             value = null;
662         }
663         return value;
664     }
