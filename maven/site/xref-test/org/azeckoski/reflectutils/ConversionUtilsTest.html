39      @SuppressWarnings("unchecked")
40      public void testConvertObject() {
41          ConversionUtils convertUtils = new ConversionUtils();
42  
43          String[] abc = new String[] {"A","B","C"};
44  
45          // equivalent conversion also should be instantaneous since no conversion happens
46          assertEquals(123, (int) convertUtils.convert(123, int.class) );
47          assertEquals(123, (int) convertUtils.convert(new Integer(123), int.class) );
48          assertEquals("123", convertUtils.convert("123", String.class) );
49          assertEquals(abc, convertUtils.convert(abc, String[].class) );
50  
51          // object conversion should be instant
52          assertEquals(new Integer(123), (Integer) convertUtils.convert(123, Object.class) );
53          assertEquals("123", convertUtils.convert("123", Object.class) );
54  
55          // now do actual conversions
56          assertEquals(123, (int) convertUtils.convert("123", int.class) );
57          assertEquals("123", convertUtils.convert("123", String.class) );
58          assertEquals(new Integer(123), convertUtils.convert("123", Integer.class) );
59  
60          // array to scalar
61          assertEquals(123, (int) convertUtils.convert(new int[] {123,456}, int.class) );
62          assertEquals(123, (int) convertUtils.convert(new String[] {"123","456"}, int.class) );
63  
64          // scalar to array
65          int[] nums = new int[] {123};
66          nums = convertUtils.convert(123, int[].class);
67          assertEquals(123, nums[0]);
68          nums = convertUtils.convert("123", int[].class);
69          assertEquals(123, nums[0]);
70  
71          // array to string
72          assertEquals("A,B,C", convertUtils.convert(abc, String.class) );
73  
74          // array to list and vice versa
75          List<String> l = convertUtils.convert(new String[] {"A","B","C"}, List.class);
76          assertEquals(l.get(0), abc[0]);
77          assertEquals(l.get(1), abc[1]);
78          assertEquals(l.get(2), abc[2]);
79  
80          LinkedList<String> ll = convertUtils.convert(new String[] {"A","B","C"}, LinkedList.class);
81          assertEquals(ll.get(0), abc[0]);
82          assertEquals(ll.get(1), abc[1]);
83          assertEquals(ll.get(2), abc[2]);
84  
85          String[] sarray = convertUtils.convert(l, String[].class);
86          assertEquals(sarray[0], l.get(0));
87          assertEquals(sarray[1], l.get(1));
88          assertEquals(sarray[2], l.get(2));
89  
90          // array to collection and map
91          Set<String> s = convertUtils.convert(new String[] {"A","B","C"}, Set.class);
92          assertNotNull(s);
93          assertEquals(3, s.size());
94          assertTrue(s.contains("A"));
95          assertTrue(s.contains("B"));
96          assertTrue(s.contains("C"));
97  
98          Map<String, Object> map = convertUtils.convert(new String[] {"A","B","C"}, Map.class);
99          assertNotNull(map);
100         assertEquals(3, map.size());
101     }
