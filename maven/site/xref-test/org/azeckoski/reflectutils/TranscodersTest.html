363     @SuppressWarnings("unchecked")
364     public void testJSONDecode() {
365         Transcoder transcoder = new JSONTranscoder();
366         Map<String, Object> decoded = null;
367 
368         // TEST a few json test strings
369         String json = "{\"id\":123,\"thing\":\"AZ\"}";
370         decoded = transcoder.decode(json);
371         assertNotNull(decoded);
372         assertEquals(123, decoded.get("id"));
373         assertEquals("AZ", decoded.get("thing"));
374 
375         json = "{\"id\":123,\"thing\":\"AZ\",\"map\":{\"name\":\"aaron\",\"date\":1221493247004,\"num\":456,\"array\":[\"A\",\"B\",\"C\"]}}";
376         decoded = transcoder.decode(json);
377         assertNotNull(decoded);
378         assertEquals(3, decoded.size());
379         assertEquals(123, decoded.get("id"));
380         assertEquals("AZ", decoded.get("thing"));
381         Map<String, Object> m2d = (Map<String, Object>) decoded.get("map");
382         assertEquals(4, m2d.size());
383         assertEquals("aaron", m2d.get("name"));
384         assertEquals(456, m2d.get("num"));
385 
386         // TEST in and out conversion
387         String encoded = null;
388 
389         // test simple cases
390         encoded = transcoder.encode(null, null, null);
391         assertNotNull(encoded);
392         assertEquals("null", encoded);
393         decoded = transcoder.decode(encoded);
394         assertNotNull(decoded);
395         assertEquals(null, decoded.get(Transcoder.DATA_KEY));
396 
397         encoded = transcoder.encode("AaronZ", null, null);
398         assertNotNull(encoded);
399         assertTrue(encoded.contains("AaronZ"));
400         decoded = transcoder.decode(encoded);
401         assertNotNull(decoded);
402         assertEquals("AaronZ", decoded.get(Transcoder.DATA_KEY));
403 
404         encoded = transcoder.encode(1234, null, null);
405         assertNotNull(encoded);
406         assertTrue(encoded.contains("1234"));
407         decoded = transcoder.decode(encoded);
408         assertNotNull(decoded);
409         assertEquals(1234, decoded.get(Transcoder.DATA_KEY));
410 
411         encoded = transcoder.encode(true, null, null);
412         assertNotNull(encoded);
413         assertTrue(encoded.contains("true"));
414         decoded = transcoder.decode(encoded);
415         assertNotNull(decoded);
416         assertEquals(true, decoded.get(Transcoder.DATA_KEY));
417 
418         // test arrays
419         TestBean[] array = new TestBean[0];
420         encoded = transcoder.encode(array, null, null);
421         assertNotNull(encoded);
422         assertEquals("[]", encoded); //json
423         decoded = transcoder.decode(encoded);
424         assertNotNull(decoded);
425         assertEquals(0, ((List)decoded.get(Transcoder.DATA_KEY)).size());
426 
427         array = new TestBean[] {new TestBean(888), new TestBean(777)};
428         encoded = transcoder.encode(array, null, null);
429         assertNotNull(encoded);
430         assertTrue(encoded.contains("888"));
431         assertTrue(encoded.contains("777"));
432         decoded = transcoder.decode(encoded);
433         assertNotNull(decoded);
434         List<Map<String, Object>> decodeArray = ((List)decoded.get(Transcoder.DATA_KEY));
435         assertEquals(2, decodeArray.size());
436         assertEquals(array[0].getMyInt(), decodeArray.get(0).get("myInt"));
437         assertEquals(array[1].getMyString(), decodeArray.get(1).get("myString"));
438 
439         List<TestPea> list = new ArrayList<TestPea>();
440         list.add( new TestPea("AZ","AaronZ"));
441         list.add( new TestPea("BZ","BeckyZ"));
442 
443         // test maps
444         Map<String, Object> m = new ArrayOrderedMap<String, Object>();
445         m.put("id", 123);
446         m.put("thing", "AZ");
447         encoded = transcoder.encode(m, null, null);
448         assertNotNull(encoded);
449         assertTrue(encoded.contains("123"));
450         assertTrue(encoded.contains("AZ"));
451         decoded = transcoder.decode(encoded);
452         assertNotNull(decoded);
453         assertEquals(2, decoded.size());
454         assertEquals(123, decoded.get("id"));
455         assertEquals("AZ", decoded.get("thing"));
456 
457         Map<String, Object> m2 = new ArrayOrderedMap<String, Object>();
458         m2.put("name", "aaron");
459         m2.put("date", new Date());
460         m2.put("num", 456);
461         m2.put("array", new String[] {"A","B","C"});
462         m.put("map", m2);
463 
464         encoded = transcoder.encode(m, null, null);
465         assertNotNull(encoded);
466         assertTrue(encoded.contains("123"));
467         assertTrue(encoded.contains("AZ"));
468         assertTrue(encoded.contains("aaron"));
469         assertTrue(encoded.contains("456"));
470         decoded = transcoder.decode(encoded);
471         assertNotNull(decoded);
472         assertEquals(3, decoded.size());
473         assertEquals(123, decoded.get("id"));
474         assertEquals("AZ", decoded.get("thing"));
475         m2d = (Map<String, Object>) decoded.get("map");
476         assertEquals(4, m2d.size());
477         assertEquals("aaron", m2d.get("name"));
478         assertEquals(456, m2d.get("num"));
479 
480         // test beans
481         TestCompound tc = new TestCompound();
482         encoded = transcoder.encode(tc, "az-root", null);
483         assertNotNull(encoded);
484 //        assertTrue(encoded.contains("az-root"));
485         assertTrue(encoded.contains("fieldInt"));
