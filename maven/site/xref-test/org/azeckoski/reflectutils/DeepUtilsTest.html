46       * Test method for {@link org.sakaiproject.ReflectUtils.util.reflect.ReflectUtil#deepClone(java.lang.Object, int, java.lang.String[])}.
47       */
48      public void testClone() {
49          DeepUtils deepUtils = new DeepUtils();
50  
51          TestBean tb = new TestBean();
52          tb.setMyInt(100);
53          tb.setMyString("1000");
54          TestBean tbClone = deepUtils.deepClone(tb, 0, null);
55          assertNotNull(tbClone);
56          assertEquals(tb.getMyInt(), tbClone.getMyInt());
57          assertEquals(tb.getMyString(), tbClone.getMyString());
58  
59          // test skipping values
60          tbClone = deepUtils.deepClone(tb, 0, new String[] {"myInt"});
61          assertNotNull(tbClone);
62          assertTrue(tb.getMyInt() != tbClone.getMyInt());
63          assertEquals(tb.getMyString(), tbClone.getMyString());
64  
65          tbClone = deepUtils.deepClone(tb, 5, null);
66          assertNotNull(tbClone);
67          assertEquals(tb.getMyInt(), tbClone.getMyInt());
68          assertEquals(tb.getMyString(), tbClone.getMyString());
69  
70          // test cloning maps
71          Map<String, Object> sm = new HashMap<String, Object>();
72          sm.put("A", "AZ");
73          sm.put("B", "BZ");
74          sm.put("C", "CZ");
75          Map<String, Object> cm = deepUtils.deepClone(sm, 1, null);
76          assertNotNull(cm);
77          assertEquals(3, cm.size());
78          assertEquals("AZ", cm.get("A"));
79          assertEquals("BZ", cm.get("B"));
80          assertEquals("CZ", cm.get("C"));
81  
82          Map<String, Object> cm2 = deepUtils.deepClone(sm, 1, new String[] {"A","C"});
83          assertNotNull(cm2);
84          assertEquals(1, cm2.size());
85          assertEquals("BZ", cm2.get("B"));
86  
87          // test cloning collections
88          List<String> l = new ArrayList<String>();
89          l.add("A");
90          l.add("B");
91          l.add("C");
92          List<String> cl = deepUtils.deepClone(l, 1, null);
93          assertNotNull(cl);
94          assertEquals(3, cl.size());
95  
96          List<TestPea> ltp = new ArrayList<TestPea>();
97          ltp.add( new TestPea("A","AZ") );
98          ltp.add( new TestPea("B","BZ") );
99          List<TestPea> cltp = deepUtils.deepClone(ltp, 1, null);
100         assertNotNull(cltp);
101         assertEquals(2, cltp.size());
102         assertNotSame(ltp.get(0), cltp.get(0)); // object was cloned
103         assertEquals(ltp.get(0).id, cltp.get(0).id); // values are same though
104         assertEquals(ltp.get(0).entityId, cltp.get(0).entityId);
105 
106         // test cloning nested objects
107         TestNesting tn = new TestNesting(123, "123", new String[] {"1","2","3"});
108         TestNesting ctn = deepUtils.deepClone(tn, 3, null);
109         assertNotNull(ctn);
110         assertNotSame(ctn, tn);
111         assertEquals(ctn.getExtra(), tn.getExtra());
112         assertEquals(ctn.getId(), tn.getId());
113         assertEquals(ctn.getTitle(), tn.getTitle());
114         assertEquals(ctn.getTestBean(), tn.getTestBean());
115         assertEquals(ctn.getTestEntity().getId(), tn.getTestEntity().getId());
116         assertEquals(ctn.getTestEntity().getBool(), tn.getTestEntity().getBool());
117 
118     }

424     public void testPopulateFromNestedMap() {
425         DeepUtils deepUtils = new DeepUtils();
426

428         m.put("A", "AZ");
429         m.put("id", "999");
430         m.put("entityId", "AZ");

436 
437         TestCompound tc = new TestCompound();

439         m.put("fieldInt", "123");
440         m.put("myInt", "234");
441         m.put("myString", "string");
442         deepUtils.populate(tc, m);
443         assertEquals("F1", tc.myField);
444         assertEquals(new Integer(123), tc.fieldInt);
445         assertEquals(234, tc.getMyInt());
446         assertEquals("string", tc.getMyString());

449         sm.put("A", "AZ");
450         sm.put("B", "BZ");
451         sm.put("C", "CZ");
452 
453         List<String> l = new ArrayList<String>();
454         l.add("A");
455         l.add("B");
456         l.add("C");
457 
458         String[] sa = new String[] {"A","B","C"};
459 
460         Set<String> set = new HashSet<String>();
461         set.add("A");
462         set.add("B");
463         set.add("C");
464 
465         TestCollections tcoll = new TestCollections(false);
466         m.clear();

485         assertNotNull(tcoll.map);
486         assertNotNull(tcoll.set);

491     }
